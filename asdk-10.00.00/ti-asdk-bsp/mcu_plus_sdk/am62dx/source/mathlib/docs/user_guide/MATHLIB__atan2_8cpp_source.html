54 // this method performs a/b division for atan2 computation
55 template <typename vecType> static inline vecType divspMod_atan2_i(vecType a, vecType b)
56 {
57  vecType res, r0, d0, d1, p0, p1, Two;
58 
59  Two = (vecType) 2.0;
60 
61  p0 = __recip(b);
62  d0 = p0 * b;
63  r0 = p0;
64 
65  p1 = Two - d0;
66  d1 = r0 * p1;
67 
68  res = a * d1;
69 
70  return res;
71 }

74 template <typename vecType, typename vecBool>
75 static inline vecType atan22f_sr1i_atan2_i(vecType g1, vecType pih, vecBool s, vecBool bn, vecBool an)
76 {
77 
78  /*************************************************************************************/
79  /* Create and assign values for constants and variables for polynomial approximation */
80  /*************************************************************************************/
81  vecType coef, negativeCoef, pi, pol, Zero, g2, g4, g6, g8, g10, g12, tmp1, tmp2, c1, c2, c3, c4, c5, c6, c7, c8, res,
82  res_minus, res_plus;
83  vecBool FALSE;
84 
85  FALSE = (vecBool) 0;
86  Zero = (vecType) 0.0;
87  pi = (vecType) 3.1415927;
88  // pih = (vecType) 1.57079632679;
89 
90  c1 = (vecType) 0.00230158202;
91  c2 = (vecType) -0.01394551000;
92  c3 = (vecType) 0.03937087815;
93  c4 = (vecType) -0.07235669163;
94  c5 = (vecType) 0.10521499322;
95  c6 = (vecType) -0.14175076797;
96  c7 = (vecType) 0.19989300877;
97  c8 = (vecType) -0.33332930041;
98 
99  coef = pih;
100 
101  // get coef based on the flags
102  // check 'swap' flag
103  // if (s == FALSE) {
104  // coef = pi;
105  // }
106  __vpred cmp_swap = __cmp_eq_pred(s, FALSE);
107  coef = __select(cmp_swap, pi, coef);
108  // if ((s == FALSE) && (bn == FALSE)) {
109  // coef = 0.0f;
110 
111  __vpred cmp_negB = __cmp_eq_pred(bn, FALSE);
112  __vpred and_negB_swap = __and(cmp_negB, cmp_swap);
113 
114  coef = __select(and_negB_swap, Zero, coef);
115 
116  // check if input to atan2sp_i is negative
117  // if (an != FALSE) {
118  // coef = -coef;
119  // }
120  negativeCoef = -coef;
121  __vpred cmp_sign = __cmp_eq_pred(an, FALSE);
122  coef = __select(cmp_sign, coef, negativeCoef);
123 
124  // calculate polynomial
125  g2 = g1 * g1;
126  g4 = g2 * g2;
127  g6 = g2 * g4;
128  g8 = g4 * g4;
129  g10 = g6 * g4;
130  g12 = g8 * g4;
131 
132  tmp1 = ((c5 * g8) + (c6 * g6)) + ((c7 * g4) + (c8 * g2));
133  tmp2 = ((((c1 * g4) + (c2 * g2)) + c3) * g12) + (c4 * g10);
134 
135  pol = tmp1 + tmp2;
136  pol = (pol * g1) + g1;
137 
138  // return ((s != FALSE) ? (coef - pol) : (coef + pol));
139  res_minus = coef - pol;
140  res_plus = coef + pol;
141  __vpred cmp_res = __cmp_eq_pred(s, FALSE);
142  res = __select(cmp_res, res_plus, res_minus);
143 
144  return res;
145 }

147 // this method performs arc-tangent2 computation of input vector
148 template <typename T> MATHLIB_STATUS MATHLIB_atan2(size_t length, T *pSrc0, T *pSrc1, T *pDst)
149 {
150 
151  // variables
152  MATHLIB_STATUS status = MATHLIB_SUCCESS; // return function status
153  size_t numBlocks = 0; // compute loop's iteration count
154  size_t remNumBlocks = 0; // when numBlocks is not a multiple of SIMD width
155 
156  // derive c7x vector type from template typename
157  typedef typename c7x::make_full_vector<T>::type vec;
158  __SE_TEMPLATE_v1 se0Params = __gen_SE_TEMPLATE_v1();
159  __SA_TEMPLATE_v1 sa0Params = __gen_SA_TEMPLATE_v1();
160 
161  status = MATHLIB_checkParams(length, pSrc0, pSrc1, pDst);
162 
163  if (status == MATHLIB_SUCCESS) {
164  MATHLIB_SE0SA01DSequentialInit(&se0Params, &sa0Params, length, pSrc0, pDst);
165 
166  // calculate compute loop's iteration counter
167  numBlocks = length / c7x::element_count_of<vec>::value;
168  remNumBlocks = length % c7x::element_count_of<vec>::value;
169  if (remNumBlocks) {
170  numBlocks++;
171  }
172 
173  // open SE0, SE1, and SA0 for reading and writing operands
174  MATHLIB_SE0SE1SA0Open(&se0Params, &sa0Params, pSrc0, pSrc1);
175 
176  /***********************************************************************/
177  /* Create and assign values for constants employed on atan2 computation */
178  /***********************************************************************/
179  vec g, res, Zero, temp, abs_a, abs_b;
180  c7x::char_vec an, bn, s, TRUE, FALSE;
181  vec pih = (vec) 1.570796327;
182  vec pi = (vec) 3.141592741;
183  vec MAX = std::numeric_limits<T>::max();
184 
185  TRUE = (c7x::char_vec) 1;
186  FALSE = (c7x::char_vec) 0;
187  Zero = (vec) 0.0;
188 
189  // compute loop to perform vector atan2
190  for (size_t i = 0; i < numBlocks; i++) {
191  vec a = c7x::strm_eng<0, vec>::get_adv();
192  vec b = c7x::strm_eng<1, vec>::get_adv();
193 
194  vec x = a;
195  vec y = b;
196 
197  /* if (i == 0) { */
198  /* x.print(); */
199  /* y.print(); */
200  /* } */
201 
202  s = FALSE;
203  __vpred cmp_negA = __cmp_lt_pred(a, Zero);
204  an = __select(cmp_negA, TRUE, FALSE);
205  __vpred cmp_negB = __cmp_lt_pred(b, Zero);
206  bn = __select(cmp_negB, TRUE, FALSE);
207 
208  // swap a and b before calling division sub routine if a > b
209  // if (_fabsf(a) > _fabsf(b) {
210  // temp = b;
211  // b = a;
212  // a = temp;
213  // s = TRUE;
214  // }
215 
216  abs_a = __abs(a);
217  abs_b = __abs(b);
218 
219  __vpred cmp_AgtB = __cmp_lt_pred(abs_b, abs_a);
220  temp = __select(cmp_AgtB, b, Zero);
221  b = __select(cmp_AgtB, a, b);
222  a = __select(cmp_AgtB, temp, a);
223  s = __select(cmp_AgtB, TRUE, s);
224  /***********************************************************************/
225  /* Division computation for atan2 */
226  /***********************************************************************/
227  g = divspMod_atan2_i<vec>(a, b);
228 
229  /***********************************************************************/
230  /* Polynomial computation for atan2 */
231  /***********************************************************************/
232  res = atan22f_sr1i_atan2_i<vec, c7x::char_vec>(g, pih, s, bn, an);
233 
234  /***********************************************************************/
235  /* Bounds Checking */
236  /***********************************************************************/
237 
238  // if (x == 0.0f) {
239  // res = y >= 0.0f ? 0.0f : pi;
240  // }
241 
242  __vpred cmp_zeroX = __cmp_eq_pred(x, Zero);
243 
244  __vpred cmp_zeroY = __cmp_le_pred(Zero, y);
245  vec resY = __select(cmp_zeroY, Zero, pi);
246 
247  res = __select(cmp_zeroX, resY, res);
248 
249  // if (g > MAX) {
250 
251  // res = pih;
252  // }
253 
254  __vpred gMax = __cmp_lt_pred(MAX, g);
255  res = __select(gMax, pih, res);
256 
257  // if (g < -MAX) {
258  // res = -pih;
259  // }
260 
261  vec MIN = -MAX;
262  vec negativepih = -pih;
263 
264  __vpred gMin = __cmp_lt_pred(g, MIN);
265  res = __select(gMin, negativepih, res);
266 
267  vec outVec = res;
268 
269  __vpred tmp = c7x::strm_agen<0, vec>::get_vpred();
270  vec *addr = c7x::strm_agen<0, vec>::get_adv(pDst);
271  __vstore_pred(tmp, addr, outVec);
272  }
273 
274  MATHLIB_SE0SE1SA0Close();
275  }
276 
277  return status;
278 }
