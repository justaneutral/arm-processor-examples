52 // this method performs sine coputation of input vector
53 template <typename T> MATHLIB_STATUS MATHLIB_sin(size_t length, T *pSrc, T *pDst)
54 {
55 
56  // variables
57  MATHLIB_STATUS status = MATHLIB_SUCCESS; // return function status
58  size_t numBlocks = 0; // compute loop's iteration count
59  size_t remNumBlocks = 0; // when numBlocks is not a multiple of SIMD width
60 
61  // derive c7x vector type from template typename
62  typedef typename c7x::make_full_vector<T>::type vec;
63 
64  // Compile-time decision: float_vec => int_vec and double_vec=> long_vec
65  typedef
66  typename std::conditional<ELEMENT_COUNT(c7x::float_vec) == ELEMENT_COUNT(vec), c7x::int_vec, c7x::long_vec>::type
67  vec_type;
68 
69  __SE_TEMPLATE_v1 se0Params = __gen_SE_TEMPLATE_v1();
70  __SA_TEMPLATE_v1 sa0Params = __gen_SA_TEMPLATE_v1();
71 
72  status = MATHLIB_checkParams(length, pSrc, pDst);
73 
74  if (status == MATHLIB_SUCCESS) {
75 
76  MATHLIB_SE0SA01DSequentialInit(&se0Params, &sa0Params, length, pSrc, pDst);
77 
78  // calculate compute loop's iteration counter
79  numBlocks = length / c7x::element_count_of<vec>::value;
80  remNumBlocks = length % c7x::element_count_of<vec>::value;
81  if (remNumBlocks) {
82  numBlocks++;
83  }
84 
85  // open SE0, SE1, and SA0 for reading and writing operands
86  MATHLIB_SE0SA0Open(&se0Params, &sa0Params, pSrc);
87 
88  /**********************************************************************/
89  /* Create and assign values for constants employed on cos computation */
90  /**********************************************************************/
91 
92  vec InvPI, One, MAX, Zero, s1, s2, s3, s4, C1, C2, X, Y, Z, F, G, R, negativeOne, Sign;
93  vec_type N, int_one;
94 
95  InvPI = (vec) 0.318309886183791;
96  One = (vec) 1.0;
97  MAX = (vec) 1048576.0;
98  Zero = (vec) 0.0;
99  s1 = (vec) -1.666665668e-1;
100  s2 = (vec) 8.333025139e-3;
101  s3 = (vec) -1.980741872e-4;
102  s4 = (vec) 2.601903036e-6;
103  C1 = (vec) 3.140625;
104  C2 = (vec) 9.67653589793e-4;
105  negativeOne = (vec) -1;
106 
107  int_one = (vec_type) 1;
108 
109  Sign = One;
110 
111  // compute loop to perform vector sin
112  for (size_t i = 0; i < numBlocks; i++) {
113  vec a = c7x::strm_eng<0, vec>::get_adv();
114  Y = a;
115 
116  // if (__abs(Y) > (vec)MAX) {
117  // Y = Zero;
118  // }
119  __vpred cmp_gt = __cmp_lt_pred((vec) MAX, __abs(Y));
120  Y = __select(cmp_gt, Zero, Y);
121 
122  // X = Y * (1/PI)
123  X = Y * InvPI;
124 
125  N = __float_to_int(X);
126  Z = c7x::convert<vec>(N);
127 
128  /**********************************************************************/
129  /* Sign checking for quadrant 3 or 4 */
130  /**********************************************************************/
131 
132  // if ((N % 2) != 0) {
133  // Sign = -Sign;
134  // }
135  vec_type andN = N & int_one;
136  vec convert_andN = c7x::convert<vec>(andN);
137  __vpred cmp_mod = __cmp_le_pred(convert_andN, Zero);
138  vec Sign_vec = __select(cmp_mod, Sign, negativeOne);
139 
140  F = (Y - (Z * C1)) - (Z * C2);
141  G = F * F;
142 
143  R = ((((((s4 * G) + s3) * G) + s2) * G) + s1) * G;
144 
145  vec output = ((F + (F * R)) * Sign_vec);
146  __vpred tmp = c7x::strm_agen<0, vec>::get_vpred();
147  vec *VB1 = c7x::strm_agen<0, vec>::get_adv(pDst);
148 
149  __vstore_pred(tmp, VB1, output);
150  }
151 
152  MATHLIB_SE0SA0Close();
153  }
154 
155  return status;
156 }
