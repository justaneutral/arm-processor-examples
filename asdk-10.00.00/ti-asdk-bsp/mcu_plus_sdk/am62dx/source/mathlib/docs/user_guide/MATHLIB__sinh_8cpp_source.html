54 template <typename T> static inline void MATHLIB_sinh_scalar(size_t length, T *restrict pSrc, T *restrict pDst)
55 {
56  // Call scalar code
57  for (uint32_t i = 0; i < length; i++) {

63 template <typename T> MATHLIB_STATUS MATHLIB_sinh(size_t length, T *pSrc, T *pDst)
64 {
65  MATHLIB_STATUS status = MATHLIB_SUCCESS; // return function status
66 
67  // check for null pointers and non-zero length
68  status = MATHLIB_checkParams(length, pSrc, pDst);
69 
70  if (status == MATHLIB_SUCCESS) {
71  T *restrict pSrcLocal = pSrc;
72  T *restrict pDstLocal = pDst;
73  MATHLIB_sinh_scalar(length, pSrcLocal, pDstLocal);
74  }
75  return status;
76 }
