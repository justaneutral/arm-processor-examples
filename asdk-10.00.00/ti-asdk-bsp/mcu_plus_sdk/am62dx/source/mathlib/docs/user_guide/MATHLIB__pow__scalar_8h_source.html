67 static inline float MATHLIB_log_pow_scalar_ci(float a)
68 {
69  double ln2 = 0.693147180559945;
70  float MAXe = 3.402823466E+38f;
71  float c1 = -0.2302894f;
72  float c2 = 0.1908169f;
73  float c3 = -0.2505905f;
74  float c4 = 0.3333164f;
75  float c5 = -0.5000002f;
76  float pol, r1, r2, r3, r4, res;
77  double dr, frcpax, rcp, T;
78  int N, T_index;
79 
80  /* r = x * frcpa(x) -1 */
81  rcp = _rcpdp((double) a);
82  frcpax = _itod(_clr(_hi(rcp), 0u, 16u), 0u);
83  dr = (frcpax * (double) a) - 1.0;
84 
85  /* Polynomial p(r) that approximates ln(1+r) - r */
86  r1 = (float) dr;
87  r2 = r1 * r1;
88  r3 = r1 * r2;
89  r4 = r2 * r2;
90 
91  pol = (c5 * r2) + (c4 * r3) + (((c2 * r1) + c3 + (c1 * r2)) * r4);
92 
93  /* Reconstruction: result = T + r + p(r) */
94  N = (int) _extu(_hi(frcpax), 1u, 21u) - 1023;
95  T_index = (int) _extu(_hi(frcpax), 12u, 29u);
96  T = MATHLIB_logTable[T_index] - (ln2 * (double) N);
97  res = (float) (dr + T + (double) pol);
98 
99  if (a > MAXe) {
100  res = 88.72283905313f;
101  }
102 
103  return (res);
104 }

109 static inline float MATHLIB_exp_pow_scalar_ci(float a)
110 {
111  float log2_base_x16 = 1.442695041f * 16.0f;
112  float Halfe = 0.5f;
113  float LnMine = -87.33654475f;
114  float LnMaxe = 88.72283905f;
115  float Maxe = 3.402823466E+38f;
116  float c0 = 0.1667361910f;
117  float c1 = 0.4999999651f;
118  float c2 = 0.9999998881f;
119  float P1 = 0.04331970214844f;
120  float P2 = 1.99663646e-6f;
121  float pol, r, r2, r3, res;
122  unsigned int Ttemp;
123  unsigned int J, K;
124  float Nf;
125  int N;
126  double dT;
127 
128  /* Get N such that |N - x*16/ln(2)| is minimized */
129  Nf = (a * log2_base_x16) + Halfe;
130  N = (int) Nf; /* Cast from intermediate variable to appease MISRA */
131 
132  if ((a * log2_base_x16) < -Halfe) {
133  N--;
134  }
135 
136  /* Argument reduction, r, and polynomial approximation pol(r) */
137  r = (a - (P1 * (float) N)) - (P2 * (float) N);
138  r2 = r * r;
139  r3 = r * r2;
140 
141  pol = (r * c2) + ((r3 * c0) + (r2 * c1));
142 
143  /* Get index for ktable and jtable */
144  K = _extu((unsigned int ) N, 28u, 30u);
145  J = (unsigned int ) N & 0x3u;
146  dT = MATHLIB_kTable[K] * MATHLIB_jTable[J];
147 
148  /* Scale exponent to adjust for 2^M */
149  Ttemp = _hi(dT) + (((unsigned int) N >> 4) << 20);
150  dT = _itod(Ttemp, _lo(dT));
151 
152  res = (float) (dT * (1.0 + (double) pol));
153 
154  /* Early exit for small a */
155  if (_extu(_ftoi(a), 1u, 24u) < 114u) {
156  res = 1.0f + a;
157  }
158 
159  /* < LnMin returns 0 */
160  if (a < LnMine) {
161  res = 0.0f;
162  }
163 
164  /* > LnMax returns MAX */
165  if (a > LnMaxe) {
166  res = Maxe;
167  }
168 
169  return (res);
170 }

176 static inline float MATHLIB_pow_scalar_ci(float a, float b)
177 {
178  float arg, W, X2;
179  int y, Sign = 1;
180 
183  y = _spint(b);
184  if ((a < 0.0) && (b == (float) y)) {
185  if ((y % 2 ) != 0) {
186  Sign = -1;
187  }
188  }
189 
190  arg = _fabsf(a);
191  W = b * MATHLIB_log_pow_scalar_ci(arg);
192 
193  if (arg == 1.0f) {
194  W = 0.0f;
195  }
196 
197  X2 = (float ) Sign * MATHLIB_exp_pow_scalar_ci(W);
198 
199  if ((a < 0.0f) && (b != (float) y)) {
200  X2 = _itof(0x7fffffffu);
201  }
202 
203  if (a == 0.0f) {
204  X2 = (b >= 0.0f) ? 0.0f : _itof(0x7F800000u);
205  }
206 
207  if (b == 0.0f) {
208  X2 = 1.0f;
209  }
210 
211  return (X2);
212 }
