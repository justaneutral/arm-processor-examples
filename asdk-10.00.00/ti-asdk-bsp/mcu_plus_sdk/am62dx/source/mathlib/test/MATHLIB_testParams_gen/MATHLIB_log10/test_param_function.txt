/*
 *  Sends the test parameter structure and number of tests
 */
void MATHLIB_log10_getTestParams (
    MATHLIB_log10_testParams_t **params,
    int32_t *                                 numTests)
{
   *params = testParams;
   *numTests =
       sizeof (testParams) / sizeof (MATHLIB_log10_testParams_t);
}
