82 template <uint32_t dTypeIn0, uint32_t dTypeIn1, uint32_t dTypeOut>
83 VXLIB_STATUS VXLIB_multiply_init_ci(VXLIB_kernelHandle handle,
84  const VXLIB_bufParams2D_t * bufParamsIn0,
85  const VXLIB_bufParams2D_t * bufParamsIn1,
86  const VXLIB_bufParams2D_t * bufParamsOut,
87  const VXLIB_multiply_InitArgs *pKerInitArgs)
88 {
89  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
90 
91  // typecast handle (void) to struct pointer type associated to kernel
92  VXLIB_multiply_PrivArgs *pKerPrivArgs = (VXLIB_multiply_PrivArgs *) handle;
93 
94  uint8_t *pBlock = pKerPrivArgs->bufPblock; // address to retrieve to store SE/SA params
95 
96  // structs to hold SE and SA parameters
97  __SE_TEMPLATE_v1 se0Params = __gen_SE_TEMPLATE_v1();
98  __SE_TEMPLATE_v1 se1Params = __gen_SE_TEMPLATE_v1();
99  __SA_TEMPLATE_v1 sa0Params = __gen_SA_TEMPLATE_v1();
100 
101  // set SE0, SE1, and SA0 params
102  size_t numBlocks = VXLIB_SE0SE1SA0_init<dTypeIn0, dTypeIn1, dTypeOut>(&se0Params, &se1Params, &sa0Params,
103  bufParamsIn0, bufParamsIn1, bufParamsOut);
104 
105  if (dTypeOut == VXLIB_UINT8) {
106  pKerPrivArgs->numBlocks = VXLIB_ceilingDiv(numBlocks, VXLIB_MULTIPLY_UNROLL_FACTOR);
107  }
108  else {
109  pKerPrivArgs->numBlocks = numBlocks;
110  }
111  /**************************/
112  /* Store SE and SA params */
113  /**************************/
114 
115  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_PARAM_OFFSET) = se0Params;
116  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE1_PARAM_OFFSET) = se1Params;
117  *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_PARAM_OFFSET) = sa0Params;
118 
119  /* -------------------------------------------------------------------------------
120  CHECK for multiply mode - (1/2^n) OR (Integer) OR (Generic)
121  --------------------------------------------------------------------------------- */
122  float scale = pKerPrivArgs->pKerInitArgs.scale;
123 
124  /* Calculate if "scale" is 1/2^n */
125  float recipScaleF = (1 / scale) * (float) (1U << DECIMAL_BITS);
126  uint32_t recipScaleU = (uint32_t)(recipScaleF);
127  uint32_t msb = (recipScaleU == 0) ? 0 : (31U - __leftmost_bit_detect_one(recipScaleU));
128  pKerPrivArgs->shiftVal = msb - DECIMAL_BITS;
129 
130  /* Calculate if "scale" is integer */
131  float convertScaleF = scale * (float) (1U << DECIMAL_BITS);
132  uint32_t convertScaleU = (uint32_t) convertScaleF;
133  uint32_t integerMask = (1U << DECIMAL_BITS) - 1U;
134 
135  if ((msb > DECIMAL_BITS) && (recipScaleU == (1u << msb))) {
136  pKerPrivArgs->multiplyMode = VXLIB_MULTIPLY_DIV_2_N;
137  }
138  else if (((convertScaleU & integerMask) == 0) && (convertScaleU > 0)) {
139  pKerPrivArgs->multiplyMode = VXLIB_MULTIPLY_INTEGER;
140  }
141  else {
142  pKerPrivArgs->multiplyMode = VXLIB_MULTIPLY_GENERIC;
143  }
144 
145  return status;
146 }

188 static inline void MultiplySat(uint8_t *pOutLocal, c7x::float_vec vscale)
189 {
190  typedef typename c7x::make_full_vector<uint8_t>::type vec;
191 
192  c7x::ushort_vec vEven, vOdd;
193  c7x::double_vec scaledD1, scaledD2, scaledD3, scaledD4, scaledD5, scaledD6, scaledD7, scaledD8;
194 
195  vec a = c7x::strm_eng<0, vec>::get_adv(); // fetch input 0 via SE0
196  vec b = c7x::strm_eng<1, vec>::get_adv(); // fetch input 1 via SE1
197 
198  __vmpyubh_vvw((a), (b), vOdd, vEven);
199  // short to float conversion
200  c7x::float_vec unscaledF1 = __low_half_int_to_float(vOdd);
201  c7x::float_vec unscaledF2 = __high_half_int_to_float(vOdd);
202  c7x::float_vec unscaledF3 = __low_half_int_to_float(vEven);
203  c7x::float_vec unscaledF4 = __high_half_int_to_float(vEven);
204 
205  __vmpysp2dp_vvw(vscale, unscaledF1, scaledD1, scaledD2);
206  __vmpysp2dp_vvw(vscale, unscaledF2, scaledD3, scaledD4);
207  __vmpysp2dp_vvw(vscale, unscaledF3, scaledD5, scaledD6);
208  __vmpysp2dp_vvw(vscale, unscaledF4, scaledD7, scaledD8);
209 
210  c7x::int_vec scaledI1E = __double_to_int_rtz(scaledD1);
211  c7x::int_vec scaledI1O = __double_to_int_rtz(scaledD2);
212  c7x::int_vec scaledI2E = __double_to_int_rtz(scaledD3);
213  c7x::int_vec scaledI2O = __double_to_int_rtz(scaledD4);
214  c7x::int_vec scaledI3E = __double_to_int_rtz(scaledD5);
215  c7x::int_vec scaledI3O = __double_to_int_rtz(scaledD6);
216  c7x::int_vec scaledI4E = __double_to_int_rtz(scaledD7);
217  c7x::int_vec scaledI4O = __double_to_int_rtz(scaledD8);
218 
219  c7x::int_vec scaledI1 = __vpackw_vvv(scaledI1O, scaledI1E);
220  c7x::int_vec scaledI2 = __vpackw_vvv(scaledI2O, scaledI2E);
221  c7x::int_vec scaledI3 = __vpackw_vvv(scaledI3O, scaledI3E);
222  c7x::int_vec scaledI4 = __vpackw_vvv(scaledI4O, scaledI4E);
223 
224  c7x::short_vec scaledS1 = __pack_sat(scaledI2, scaledI1);
225  c7x::short_vec scaledS2 = __pack_sat(scaledI4, scaledI3);
226  scaledS1 = c7x::as_short_vec(__permute(vpermMaskSat, c7x::as_uchar_vec(scaledS1)));
227  scaledS2 = c7x::as_short_vec(__permute(vpermMaskSat, c7x::as_uchar_vec(scaledS2)));
228 
229  c7x::uchar_vec saturatedOut = __vspackuhb_vvv(c7x::as_short_vec(scaledS2), c7x::as_short_vec(scaledS1));
230 
231  // store output via SA0
232  __vpred tmp = c7x::strm_agen<0, vec>::get_vpred(); // generate predicate
233  vec * addr = c7x::strm_agen<0, vec>::get_adv(pOutLocal); // generate address
234  __vstore_pred(tmp, addr, saturatedOut); // store value
235 }

284 template <typename dTypeIn0, typename dTypeIn1, typename dTypeOut>
285 inline void VXLIB_multiply_util_sat(VXLIB_kernelHandle handle, void *pOut, size_t size)
286 {
287  VXLIB_DEBUGPRINTFN(0, "%s\n", "Enter function <dTypeIn0, dTypeIn1, dTypeOut>");
288  typedef typename c7x::make_full_vector<dTypeOut>::type vec;
289 
290  dTypeOut * pOutLocal = (dTypeOut *) pOut;
291  VXLIB_multiply_PrivArgs *pKerPrivArgs = (VXLIB_multiply_PrivArgs *) handle;
292  VXLIB_F32 scale = pKerPrivArgs->pKerInitArgs.scale;
293  uint32_t shiftVal = pKerPrivArgs->shiftVal;
294  size_t multiplyMode = pKerPrivArgs->multiplyMode;
295 
296  c7x::int_vec vEven, vOdd;
297 
298  /* For 1 / 2^n */
299  if (multiplyMode == VXLIB_MULTIPLY_DIV_2_N) {
300  c7x::int_vec rightShiftVec = (c7x::int_vec) shiftVal;
301 
302  uint32_t neg = (1u << shiftVal) - 1u;
303  c7x::int_vec negVec = (c7x::int_vec) neg;
304 
305  for (uint32_t counter = 0; counter < size; counter++) {
306  vec a = c7x::strm_eng<0, vec>::get_adv(); // fetch input 0 via SE0
307  vec b = c7x::strm_eng<1, vec>::get_adv(); // fetch input 1 via SE1
308 
309  __vmpyhw_vvw(a, b, vEven, vOdd);
310 
311  c7x::int_vec vEvenMask = (__shift_right(vEven, (c7x::int_vec) 31) & negVec) + vEven;
312  c7x::int_vec vOddMask = (__shift_right(vOdd, (c7x::int_vec) 31) & negVec) + vOdd;
313 
314  c7x::int_vec vEvenScaled = __shift_right(vEvenMask, rightShiftVec);
315  c7x::int_vec vOddScaled = __shift_right(vOddMask, rightShiftVec);
316 
317  c7x::short_vec vOut = __pack_sat(vOddScaled, vEvenScaled);
318 
319  __vpred pred = c7x::strm_agen<0, vec>::get_vpred();
320  vec * pStore = c7x::strm_agen<0, vec>::get_adv(pOutLocal);
321  __vstore_pred(pred, pStore, vOut);
322  }
323  }
324  /* For Integer */
325  else if (multiplyMode == VXLIB_MULTIPLY_INTEGER) {
326  int32_t intScale = (int32_t) scale;
327  c7x::int_vec vScale = (c7x::int_vec) intScale;
328  c7x::long_vec scaled1, scaled2, scaled3, scaled4;
329 
330  uint32_t loopCount = size;
331  for (uint32_t counter = 0; counter < loopCount; counter++) {
332  vec a = c7x::strm_eng<0, vec>::get_adv(); // fetch input 0 via SE0
333  vec b = c7x::strm_eng<1, vec>::get_adv(); // fetch input 1 via SE1
334 
335  __vmpyhw_vvw(a, b, vEven, vOdd);
336 
337  __vmpywd_vvw(vEven, vScale, scaled1, scaled2);
338  __vmpywd_vvw(vOdd, vScale, scaled3, scaled4);
339 
340  c7x::int_vec scaledI1 = __pack_sat(scaled2, scaled1);
341  c7x::int_vec scaledI2 = __pack_sat(scaled4, scaled3);
342 
343  c7x::short_vec vOut = __pack_sat(scaledI2, scaledI1);
344 
345  __vpred pred = c7x::strm_agen<0, vec>::get_vpred(); // generate predicate
346  vec * pStore = c7x::strm_agen<0, vec>::get_adv(pOutLocal); // generate address
347  __vstore_pred(pred, pStore, vOut); // store value
348  }
349  }
350  /* Generic */
351  else {
352  c7x::double_vec vScaleD = (c7x::double_vec) scale;
353  for (uint32_t counter = 0; counter < size; counter++) {
354 
355  vec a = c7x::strm_eng<0, vec>::get_adv(); // fetch input 0 via SE0
356  vec b = c7x::strm_eng<1, vec>::get_adv(); // fetch input 1 via SE1
357 
358  __vmpyhw_vvw(a, b, vEven, vOdd);
359 
360  c7x::double_vec unscaledD1 = __low_int_to_double(vEven);
361  c7x::double_vec unscaledD2 = __high_int_to_double(vEven);
362  c7x::double_vec unscaledD3 = __low_int_to_double(vOdd);
363  c7x::double_vec unscaledD4 = __high_int_to_double(vOdd);
364 
365  c7x::double_vec scaledD1 = unscaledD1 * vScaleD;
366  c7x::double_vec scaledD2 = unscaledD2 * vScaleD;
367  c7x::double_vec scaledD3 = unscaledD3 * vScaleD;
368  c7x::double_vec scaledD4 = unscaledD4 * vScaleD;
369 
370  c7x::int_vec scaledI1E = __double_to_int_rtz(scaledD1);
371  c7x::int_vec scaledI1O = __double_to_int_rtz(scaledD2);
372  c7x::int_vec scaledI2E = __double_to_int_rtz(scaledD3);
373  c7x::int_vec scaledI2O = __double_to_int_rtz(scaledD4);
374 
375  c7x::int_vec scaledI1 = __vpackw_vvv(scaledI1O, scaledI1E);
376  c7x::int_vec scaledI2 = __vpackw_vvv(scaledI2O, scaledI2E);
377 
378  vec saturatedOut = __pack_sat(scaledI2, scaledI1);
379 
380  __vpred tmp = c7x::strm_agen<0, vec>::get_vpred(); // generate predicate
381  vec * addr = c7x::strm_agen<0, vec>::get_adv(pOutLocal); // generate address
382  __vstore_pred(tmp, addr, saturatedOut); // store value
383  }
384  }
385 
386  VXLIB_DEBUGPRINTFN(0, "%s\n", "Exit function");
387  return;
388 }
