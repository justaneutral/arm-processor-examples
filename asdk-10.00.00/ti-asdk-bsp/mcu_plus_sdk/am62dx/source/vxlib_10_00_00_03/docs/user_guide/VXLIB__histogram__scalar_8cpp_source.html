73 static void VXLIB_histogramSimple_i8u_o32u_core(const uint8_t even_img[restrict],
74  const uint8_t odd_img[restrict],
75  int32_t width,
76  uint32_t _hist0[],
77  uint32_t _hist1[],
78  uint32_t _hist2[],
79  uint32_t _hist3[],
80  uint32_t hist0[restrict],
81  uint32_t hist1[restrict],
82  uint32_t hist2[restrict],
83  uint32_t hist3[restrict],
84  uint32_t *minValue)
85 {
86  int32_t x, j;
87  uint32_t ha0o, bn0e, hs0e, ha0e, bn0o, hs0o;
88  uint32_t ha1o, bn1e, hs1e, ha1e, bn1o, hs1o;
89  uint32_t ha2o, bn2e, hs2e, ha2e, bn2o, hs2o;
90  uint32_t ha3o, bn3e, hs3e, ha3e, bn3o, hs3o;
91  uint32_t pix7654, pix3210;
92  int64_t pix2a, pix2b;
93  __x128_t pix4a, pix4b;
94  uint32_t minVal = *minValue;
95 
96  /* Case 1: Img pointers are not aligned to 4 byte boundaries, process of 8 pixels at a time */
97  /* I/O bound max would be 9/8, but there is a compiler bug when I achieved this ... not sure where */
98  /* 10/8 = 1.25 cycles per pixel */
99 
100  /* ---------------------------------------------------------------- */
101  /* Seed 'previous odd pixel' with bogus bin values that won't */
102  /* match the first even pixels. */
103  /* ---------------------------------------------------------------- */
104  bn3o = 1U;
105  bn2o = 1U;
106  bn1o = 1U;
107  bn0o = 1U;
108 
109  /* ---------------------------------------------------------------- */
110  /* Prefetch the initial bins for each of our four histograms. */
111  /* This is needed for properly handling our forwarding logic. */
112  /* ---------------------------------------------------------------- */
113  ha3o = _hist3[bn3o]; /* read previous odd data */
114  ha2o = _hist2[bn2o]; /* read previous odd data */
115  ha1o = _hist1[bn1o]; /* read previous odd data */
116  ha0o = _hist0[bn0o]; /* read previous odd data */
117 
118  /*------------------------------------------------------------------*/
119  /* An iteration is peeled off here using the non-restrict qualified */
120  /* _hist[0-3] to account for bn[0-3]o being 1 on the first */
121  /* iteration since it would be invalid to use restrict for the first*/
122  /* iteration. All other iterations uses index values that are a */
123  /* multiple of 4, so we can correctly use the restrict qualified */
124  /* hist[0-3] pointers to get full performance. */
125  /*------------------------------------------------------------------*/
126  x = j = 0;
127  if (x < ((width / 8) * 8)) {
128  /* ------------------------------------------------------------ */
129  /* Load 4 pixels from the even side of the image. Extract */
130  /* these so that our bin address is 4 * pixel_value. (The */
131  /* factor of 4 accounts for the interleave.) */
132  /* ------------------------------------------------------------ */
133  pix3210 = _mem4_const(&even_img[j * 4]);
134  minVal = _minu4(minVal, pix3210);
135  pix2a = _mpyu4ll(pix3210, 0x04040404U);
136  pix4a = _dmpy2(pix2a, 0x0001000100010001);
137  bn3e = _get32_128(pix4a, 3U); /* extract byte 3 */
138  bn2e = _get32_128(pix4a, 2U); /* extract byte 2 */
139  bn1e = _get32_128(pix4a, 1U); /* extract byte 1 */
140  bn0e = _get32_128(pix4a, 0); /* extract byte 0 */
141 
142  /* ------------------------------------------------------------ */
143  /* For each of the four even bins, do the following steps: */
144  /* -- Read the current value of the bin for even pixel. */
145  /* -- Store the update for the previous odd pixel. */
146  /* -- Increment the value we read for even pixel's bin. */
147  /* -- Add one more if even pixel equals previous odd pixel. */
148  /* (Except on first iteration, since there is no prev.) */
149  /* ------------------------------------------------------------ */
150  hs3e = _hist3[bn3e]; /* Get even bin. */
151  hs2e = _hist2[bn2e]; /* Get even bin. */
152  hs1e = _hist1[bn1e]; /* Get even bin. */
153  hs0e = _hist0[bn0e]; /* Get even bin. */
154 
155  _hist3[bn3o] = ha3o; /* Save previous odd bin. */
156  _hist2[bn2o] = ha2o; /* Save previous odd bin. */
157  _hist1[bn1o] = ha1o; /* Save previous odd bin. */
158  _hist0[bn0o] = ha0o; /* Save previous odd bin. */
159 
160  ha3e = 1U + hs3e; /* Update even bin. */
161  ha2e = 1U + hs2e; /* Update even bin. */
162  ha1e = 1U + hs1e; /* Update even bin. */
163  ha0e = 1U + hs0e; /* Update even bin. */
164 
165  ha3e += (uint32_t) (bn3e == bn3o); /* Add forwarding. */
166  ha2e += (uint32_t) (bn2e == bn2o); /* Add forwarding. */
167  ha1e += (uint32_t) (bn1e == bn1o); /* Add forwarding. */
168  ha0e += (uint32_t) (bn0e == bn0o); /* Add forwarding. */
169 
170  /* Load 4 pixels from the odd side of the image. */
171  pix7654 = _mem4_const(&odd_img[j * 4]);
172  minVal = _minu4(minVal, pix7654);
173  pix2b = _mpyu4ll(pix7654, 0x04040404U);
174  pix4b = _dmpy2(pix2b, 0x0001000100010001);
175  bn3o = _get32_128(pix4b, 3U); /* extract byte 3 */
176  bn2o = _get32_128(pix4b, 2U); /* extract byte 2 */
177  bn1o = _get32_128(pix4b, 1U); /* extract byte 1 */
178  bn0o = _get32_128(pix4b, 0); /* extract byte 0 */
179 
180  /* ------------------------------------------------------------ */
181  /* For each of the four odd bins, do the following steps: */
182  /* -- Read the current value of the bin for odd pixel. */
183  /* -- Store the update for the previous even pixel. */
184  /* -- Increment the value we read for odd pixel's bin. */
185  /* -- Add one more if odd pixel equals previous even pixel. */
186  /* ------------------------------------------------------------ */
187  hs3o = _hist3[bn3o]; /* Get odd bin. */
188  hs2o = _hist2[bn2o]; /* Get odd bin. */
189  hs1o = _hist1[bn1o]; /* Get odd bin. */
190  hs0o = _hist0[bn0o]; /* Get odd bin. */
191 
192  _hist3[bn3e] = ha3e; /* Save previous even bin. */
193  _hist2[bn2e] = ha2e; /* Save previous even bin. */
194  _hist1[bn1e] = ha1e; /* Save previous even bin. */
195  _hist0[bn0e] = ha0e; /* Save previous even bin. */
196 
197  ha3o = 1U + hs3o; /* Update odd bin. */
198  ha2o = 1U + hs2o; /* Update odd bin. */
199  ha1o = 1U + hs1o; /* Update odd bin. */
200  ha0o = 1U + hs0o; /* Update odd bin. */
201 
202  ha3o += (uint32_t) (bn3o == bn3e); /* Add forwarding. */
203  ha2o += (uint32_t) (bn2o == bn2e); /* Add forwarding. */
204  ha1o += (uint32_t) (bn1o == bn1e); /* Add forwarding. */
205  ha0o += (uint32_t) (bn0o == bn0e); /* Add forwarding. */
206 
207  j += 2;
208  x += 8;
209  }
210  {
211  for (; x < ((width / 8) * 8); x += 8) {
212  /* ------------------------------------------------------------ */
213  /* Load 4 pixels from the even side of the image. Extract */
214  /* these so that our bin address is 4 * pixel_value. (The */
215  /* factor of 4 accounts for the interleave.) */
216  /* ------------------------------------------------------------ */
217  pix3210 = _mem4_const(&even_img[j * 4]);
218  minVal = _minu4(minVal, pix3210);
219  pix2a = _mpyu4ll(pix3210, 0x04040404U);
220  pix4a = _dmpy2(pix2a, 0x0001000100010001);
221  bn3e = _get32_128(pix4a, 3U); /* extract byte 3 */
222  bn2e = _get32_128(pix4a, 2U); /* extract byte 2 */
223  bn1e = _get32_128(pix4a, 1U); /* extract byte 1 */
224  bn0e = _get32_128(pix4a, 0); /* extract byte 0 */
225 
226  /* ------------------------------------------------------------ */
227  /* For each of the four even bins, do the following steps: */
228  /* -- Read the current value of the bin for even pixel. */
229  /* -- Store the update for the previous odd pixel. */
230  /* -- Increment the value we read for even pixel's bin. */
231  /* -- Add one more if even pixel equals previous odd pixel. */
232  /* (Except on first iteration, since there is no prev.) */
233  /* ------------------------------------------------------------ */
234  hs3e = hist3[bn3e]; /* Get even bin. */
235  hs2e = hist2[bn2e]; /* Get even bin. */
236  hs1e = hist1[bn1e]; /* Get even bin. */
237  hs0e = hist0[bn0e]; /* Get even bin. */
238 
239  hist3[bn3o] = ha3o; /* Save previous odd bin. */
240  hist2[bn2o] = ha2o; /* Save previous odd bin. */
241  hist1[bn1o] = ha1o; /* Save previous odd bin. */
242  hist0[bn0o] = ha0o; /* Save previous odd bin. */
243 
244  ha3e = 1U + hs3e; /* Update even bin. */
245  ha2e = 1U + hs2e; /* Update even bin. */
246  ha1e = 1U + hs1e; /* Update even bin. */
247  ha0e = 1U + hs0e; /* Update even bin. */
248 
249  ha3e += (uint32_t) (bn3e == bn3o); /* Add forwarding. */
250  ha2e += (uint32_t) (bn2e == bn2o); /* Add forwarding. */
251  ha1e += (uint32_t) (bn1e == bn1o); /* Add forwarding. */
252  ha0e += (uint32_t) (bn0e == bn0o); /* Add forwarding. */
253 
254  /* Load 4 pixels from the odd side of the image. */
255  pix7654 = _mem4_const(&odd_img[j * 4]);
256  minVal = _minu4(minVal, pix7654);
257  pix2b = _mpyu4ll(pix7654, 0x04040404U);
258  pix4b = _dmpy2(pix2b, 0x0001000100010001);
259  bn3o = _get32_128(pix4b, 3U); /* extract byte 3 */
260  bn2o = _get32_128(pix4b, 2U); /* extract byte 2 */
261  bn1o = _get32_128(pix4b, 1U); /* extract byte 1 */
262  bn0o = _get32_128(pix4b, 0); /* extract byte 0 */
263 
264  /* ------------------------------------------------------------ */
265  /* For each of the four odd bins, do the following steps: */
266  /* -- Read the current value of the bin for odd pixel. */
267  /* -- Store the update for the previous even pixel. */
268  /* -- Increment the value we read for odd pixel's bin. */
269  /* -- Add one more if odd pixel equals previous even pixel. */
270  /* ------------------------------------------------------------ */
271  hs3o = hist3[bn3o]; /* Get odd bin. */
272  hs2o = hist2[bn2o]; /* Get odd bin. */
273  hs1o = hist1[bn1o]; /* Get odd bin. */
274  hs0o = hist0[bn0o]; /* Get odd bin. */
275 
276  hist3[bn3e] = ha3e; /* Save previous even bin. */
277  hist2[bn2e] = ha2e; /* Save previous even bin. */
278  hist1[bn1e] = ha1e; /* Save previous even bin. */
279  hist0[bn0e] = ha0e; /* Save previous even bin. */
280 
281  ha3o = 1U + hs3o; /* Update odd bin. */
282  ha2o = 1U + hs2o; /* Update odd bin. */
283  ha1o = 1U + hs1o; /* Update odd bin. */
284  ha0o = 1U + hs0o; /* Update odd bin. */
285 
286  ha3o += (uint32_t) (bn3o == bn3e); /* Add forwarding. */
287  ha2o += (uint32_t) (bn2o == bn2e); /* Add forwarding. */
288  ha1o += (uint32_t) (bn1o == bn1e); /* Add forwarding. */
289  ha0o += (uint32_t) (bn0o == bn0e); /* Add forwarding. */
290 
291  j += 2;
292  }
293 
294  /* Store final odd-pixel bin values. */
295  hist3[bn3o] = ha3o;
296  hist2[bn2o] = ha2o;
297  hist1[bn1o] = ha1o;
298  hist0[bn0o] = ha0o;
299 
300 /* Case 2: Process remaining pixels beyond multiple of 8 */
301 /* 7 cycles per pixel */
302 #pragma MUST_ITERATE(0, 8, 1)
303  for (; x < width; x++) {
304  hist0[even_img[x] * 4U]++;
305 
306  // Commenting the code to achieve coverage
307  // if ((uint32_t) even_img[x] < (minVal & 0xffU)) {
308  // minVal = (minVal & 0xffffff00U) | (uint32_t) even_img[x];
309  // }
310  }
311  }
312 
313  *minValue = minVal;
314 }

316 void VXLIB_histogramSimple_i8u_o32u(const uint8_t src[],
317  VXLIB_kernelHandle handle,
318  uint32_t dist[],
319  uint32_t scratch[],
320  uint32_t *minValue,
321  uint8_t lastBlock)
322 {
323 
324  int32_t x;
325  uint32_t min4 = *minValue;
326  uint32_t min2, min1;
327 
328  VXLIB_histogram_PrivArgs *pKerPrivArgs = (VXLIB_histogram_PrivArgs *) handle;
329 
330  if (pKerPrivArgs->width == pKerPrivArgs->strideInElements) {
331  VXLIB_histogramSimple_i8u_o32u_core(src, &src[4], pKerPrivArgs->width * pKerPrivArgs->height, &scratch[0],
332  &scratch[1], &scratch[2], &scratch[3], &scratch[0], &scratch[1], &scratch[2],
333  &scratch[3], &min4);
334  }
335  else {
336 
337  int32_t y;
338 
339  for (y = 0; y < pKerPrivArgs->height; y++) {
340 
341  const uint8_t *even = (const uint8_t *) &src[y * pKerPrivArgs->strideInElements];
342  const uint8_t *odd = (const uint8_t *) &src[(y * pKerPrivArgs->strideInElements) + 4];
343 
344  VXLIB_histogramSimple_i8u_o32u_core(even, odd, pKerPrivArgs->width, &scratch[0], &scratch[1], &scratch[2],
345  &scratch[3], &scratch[0], &scratch[1], &scratch[2], &scratch[3], &min4);
346  }
347  }
348 
349  *minValue = min4;
350 
351  /* Don't waste time updating the dist array if working through blocks of an image */
352  if (lastBlock != 0) {
353  /* 1.5 cycles per bin */
354  _nassert(((uintptr_t) scratch % 8U) == 0);
355  for (x = 0; x < 256; x++) {
356  dist[x] = scratch[(x * 4) + 0] + scratch[(x * 4) + 2] + scratch[(x * 4) + 1] + scratch[(x * 4) + 3];
357  // printf("value: %d\n", dist[x]);
358  }
359 
360  /* Reduce 4 lanes down to a single min value */
361  min2 = _minu4(min4, min4 >> 16);
362  min1 = _minu4(min2, min2 >> 8) & 0xffU;
363  *minValue = min1;
364  }
365 }

369 VXLIB_STATUS
370 VXLIB_histogram_exec_scalar_ci(VXLIB_kernelHandle handle, void *restrict pIn, void *restrict pOut, void *restrict pTemp)
371 {
372 
373  VXLIB_STATUS status = VXLIB_SUCCESS;
374  /* This function doesn't need minValue, so setting it to a dummy value */
375  uint32_t minValue = 0;
376 
377  uint8_t *restrict src = (uint8_t *) pIn;
378  uint32_t *restrict dist = (uint32_t *) pOut;
379  uint32_t *restrict scratch = (uint32_t *) pTemp;
380 
381  VXLIB_histogram_PrivArgs *pKerPrivArgs = (VXLIB_histogram_PrivArgs *) handle;
382  uint8_t offset = pKerPrivArgs->pKerInitArgs.offset;
383  uint16_t range = pKerPrivArgs->pKerInitArgs.range;
384  uint16_t numBins = pKerPrivArgs->pKerInitArgs.numBins;
385  uint8_t lastBlock = pKerPrivArgs->pKerInitArgs.lastBlock;
386 
387  /* Call the optimized simple histogram function, which gives ~1.25 cycles per input pixel */
388  /* Use the top of the scratch buffer to also be the intermediate dist buffer, since it doesn't
389  * get written until the lastBlock = 1 anyway */
390  VXLIB_histogramSimple_i8u_o32u(src, handle, scratch, scratch, &minValue, lastBlock);
391 
392  /* If calling the function on the lastBlock, then redistribute the histogram according to
393  * function parameters (offset/range/numBins) */
394  if (lastBlock == 1U) {
395 
396  uint16_t index;
397  uint16_t x;
398  uint16_t offset_16 = (uint16_t) offset;
399 
400  /* 7 cycles per input bin (range bins) */
401  /* Max range == 256 (256*7 = 1792 cycles) */
402  for (x = offset_16; x < (offset_16 + range); x++) {
403  index = ((x - offset_16) * numBins) / range;
404  dist[index] += scratch[x];
405  // printf("value: %d\n", dist[index]);
406  }
407  }
408 
409  return status;
410 }
