148  size = 3;
149  }
150  else if ((type == (uint32_t) VXLIB_INT32) || (type == (uint32_t) VXLIB_UINT32) ||
151  (type == (uint32_t) VXLIB_FLOAT32)) {
152  size = 4;
153  }
154  else if ((type == (uint32_t) VXLIB_INT64) || (type == (uint32_t) VXLIB_UINT64) ||
155  (type == (uint32_t) VXLIB_FLOAT64)) {
156  size = 8;
157  }
158  else if ((type == (uint32_t) VXLIB_INT128) || (type == (uint32_t) VXLIB_UINT128)) {
159  size = 16;
160  }
161  else {
162  size = 1;
163  }
164
