61 // this method checks the initialization parameters for the VXLIB_thresholdRange kernel
62 VXLIB_STATUS
63 VXLIB_thresholdRange_init_checkParams(VXLIB_kernelHandle handle,
64  const VXLIB_bufParams2D_t * bufParamsIn,
65  const VXLIB_bufParams2D_t * bufParamsOut,
66  const VXLIB_thresholdRange_InitArgs *pKerInitArgs)
67 {
68  VXLIB_STATUS status = VXLIB_SUCCESS;
69 
70  if ((handle == NULL) || (bufParamsIn == NULL) || (bufParamsOut == NULL) || (pKerInitArgs == NULL)) {
71  status = VXLIB_ERR_NULL_POINTER;
72  }
73 
74  if (status == VXLIB_SUCCESS) {
75  // check for dimensions and datatype combinations
76  // obtain input1 buffer parameters
77  uint32_t dTypeIn = bufParamsIn->data_type;
78  uint32_t widthIn = bufParamsIn->dim_x;
79  uint32_t heightIn = bufParamsIn->dim_y;
80  uint32_t strideIn = bufParamsIn->stride_y;
81 
82  // obtain output buffer parameters
83  uint32_t dTypeOut = bufParamsOut->data_type;
84  uint32_t widthOut = bufParamsOut->dim_x;
85  uint32_t heightOut = bufParamsOut->dim_y;
86  uint32_t strideOut = bufParamsOut->stride_y;
87 
88  uint32_t strideInElements = strideIn / VXLIB_sizeof(dTypeIn);
89  uint32_t strideOutElements = strideOut / VXLIB_sizeof(dTypeOut);
90 
91  if ((widthIn != widthOut) || (heightIn != heightOut)) {
92  status = VXLIB_ERR_INVALID_DIMENSION;
93  }
94  else if ((strideInElements < widthIn) || (strideOutElements < widthOut)) {
95  status = VXLIB_ERR_NOT_EQUAL_WIDTH_STRIDE;
96  }
97  else if (!(VXLIB_THRESHOLDRANGE_I8U_O8U) && !(VXLIB_THRESHOLDRANGE_I8S_O8S) &&
98  !(VXLIB_THRESHOLDRANGE_I16U_O16U) && !(VXLIB_THRESHOLDRANGE_I16S_O16S)) {
99  status = VXLIB_ERR_INVALID_TYPE;
100  }
101  else {
102  status = VXLIB_SUCCESS;
103  }
104  }
105 
106  return status;
107 }

146 // this method is the user-level initialization function for the VXLIB_thresholdRange kernel
147 VXLIB_STATUS VXLIB_thresholdRange_init(VXLIB_kernelHandle handle,
148  VXLIB_bufParams2D_t * bufParamsIn,
149  VXLIB_bufParams2D_t * bufParamsOut,
150  const VXLIB_thresholdRange_InitArgs *pKerInitArgs)
151 {
152  VXLIB_STATUS status = VXLIB_SUCCESS;
153  VXLIB_thresholdRange_PrivArgs *pKerPrivArgs = (VXLIB_thresholdRange_PrivArgs *) handle;
154 
155  // copy pKerinitArgs into pKerPrivargs
156  pKerPrivArgs->pKerInitArgs = *pKerInitArgs;
157 
158  // set width and height of image
159  pKerPrivArgs->width = bufParamsIn->dim_x;
160  pKerPrivArgs->height = bufParamsIn->dim_y;
161 
162  // compute stride in elements as SE/SA params are set to work with given element type
163  pKerPrivArgs->strideInElements = bufParamsIn->stride_y / VXLIB_sizeof(bufParamsIn->data_type);
164  pKerPrivArgs->strideOutElements = bufParamsOut->stride_y / VXLIB_sizeof(bufParamsOut->data_type);
165 
166 #if VXLIB_DEBUGPRINT
167  printf("VXLIB_DEBUGPRINT Enter VXLIB_thresholdRange_init\n");
168 #endif
169 
170  // obtain buffer datatypes
171  uint32_t dTypeIn = bufParamsIn->data_type;
172  uint32_t dTypeOut = bufParamsOut->data_type;
173 
174  // determine natural C vs optimized function call
175  if (pKerInitArgs->funcStyle == VXLIB_FUNCTION_NATC) {
176 
177  // set function pointer for natural C function with appropriate template parameters based on datatypes
178  if (VXLIB_THRESHOLDRANGE_I8U_O8U) {
179  pKerPrivArgs->execute = VXLIB_thresholdRange_exec_cn<VXLIB_THRESHOLDRANGE_TYPENAME_I8U_O8U>;
180  }
181  else if (VXLIB_THRESHOLDRANGE_I8S_O8S) {
182  pKerPrivArgs->execute = VXLIB_thresholdRange_exec_cn<VXLIB_THRESHOLDRANGE_TYPENAME_I8S_O8S>;
183  }
184  else if (VXLIB_THRESHOLDRANGE_I16U_O16U) {
185  pKerPrivArgs->execute = VXLIB_thresholdRange_exec_cn<VXLIB_THRESHOLDRANGE_TYPENAME_I16U_O16U>;
186  }
187  else if (VXLIB_THRESHOLDRANGE_I16S_O16S) {
188  pKerPrivArgs->execute = VXLIB_thresholdRange_exec_cn<VXLIB_THRESHOLDRANGE_TYPENAME_I16S_O16S>;
189  }
190  else {
191  status = VXLIB_ERR_INVALID_TYPE;
192  }
193  }
194  else { // Optimized function
195 
196  // set function pointer for natural C function with appropriate template parameters based on datatypes
197  if (VXLIB_THRESHOLDRANGE_I8U_O8U) {
198  pKerPrivArgs->execute = VXLIB_thresholdRange_exec_ci<VXLIB_THRESHOLDRANGE_TYPENAME_I8U_O8U>;
199  status = VXLIB_thresholdRange_init_ci<VXLIB_THRESHOLDRANGE_DTYPE_I8U_O8U>(handle, bufParamsIn, bufParamsOut,
200  pKerInitArgs);
201  }
202  else if (VXLIB_THRESHOLDRANGE_I8S_O8S) {
203  pKerPrivArgs->execute = VXLIB_thresholdRange_exec_ci<VXLIB_THRESHOLDRANGE_TYPENAME_I8S_O8S>;
204  status = VXLIB_thresholdRange_init_ci<VXLIB_THRESHOLDRANGE_DTYPE_I8S_O8S>(handle, bufParamsIn, bufParamsOut,
205  pKerInitArgs);
206  }
207  else if (VXLIB_THRESHOLDRANGE_I16U_O16U) {
208  pKerPrivArgs->execute = VXLIB_thresholdRange_exec_ci<VXLIB_THRESHOLDRANGE_TYPENAME_I16U_O16U>;
209  status = VXLIB_thresholdRange_init_ci<VXLIB_THRESHOLDRANGE_DTYPE_I16U_O16U>(handle, bufParamsIn, bufParamsOut,
210  pKerInitArgs);
211  }
212  else if (VXLIB_THRESHOLDRANGE_I16S_O16S) {
213  pKerPrivArgs->execute = VXLIB_thresholdRange_exec_ci<VXLIB_THRESHOLDRANGE_TYPENAME_I16S_O16S>;
214  status = VXLIB_thresholdRange_init_ci<VXLIB_THRESHOLDRANGE_DTYPE_I16S_O16S>(handle, bufParamsIn, bufParamsOut,
215  pKerInitArgs);
216  }
217  else {
218  status = VXLIB_ERR_INVALID_TYPE;
219  }
220  }
221 
222  return status;
223 }
