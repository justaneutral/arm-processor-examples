111 // this method initializes the kernel-specific parameters
112 // mainly, the streaming engine and streaming address generators
113 template <uint32_t dType>
114 VXLIB_STATUS VXLIB_minMaxLoc_init_ci(VXLIB_kernelHandle handle,
115  VXLIB_bufParams2D_t * bufParamsIn,
116  VXLIB_bufParams1D_t * bufParamsMinLoc,
117  VXLIB_bufParams1D_t * bufParamsMaxLoc,
118  const VXLIB_minMaxLoc_InitArgs *pKerInitArgs)
119 {
120  VXLIB_DEBUGPRINTFN(0, "%s\n", "Enter Function");
121  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
122 
123  // typecast handle (void) to struct pointer type associated to kernel
124  VXLIB_minMaxLoc_PrivArgs *pKerPrivArgs = (VXLIB_minMaxLoc_PrivArgs *) handle;
125 
126  // obtain image size and compute number of blocks to process
127  uint32_t width = pKerPrivArgs->width;
128  uint32_t height = pKerPrivArgs->height;
129  uint32_t strideInElements = pKerPrivArgs->strideInElements;
130  uint32_t eleCount = VXLIB_max_simd<dType>::value;
131 
132  uint8_t *pBlock = pKerPrivArgs->bufPblock; // address to retrieve to store SE/SA params
133 
134  typedef typename VXLIB_vec_type<dType>::type vec;
135 
136  __SE_ELETYPE SE_ELETYPE = c7x::se_eletype<vec>::value;
137  __SE_VECLEN SE_VECLEN = c7x::se_veclen<vec>::value;
138  __SA_VECLEN SA_VECLEN = c7x::sa_veclen<vec>::value;
139 
140  /* ---------------------------------------------------------------------------------------------------
141  SE PARAM SETTINGS FOR MAX/MIN VALUE
142  ------------------------------------------------------------------------------------------------------- */
143 
144  // structs to hold SE and SA parameters
145  __SE_TEMPLATE_v1 se0Params = __gen_SE_TEMPLATE_v1();
146  __SE_TEMPLATE_v1 se1Params = __gen_SE_TEMPLATE_v1();
147 
148  uint32_t oneVecBlocks = width / eleCount;
149  uint32_t firstBlockWidth = oneVecBlocks * eleCount;
150  uint32_t remainingCols = width - firstBlockWidth;
151  predSelectMask = generateSelectMask<dType>(remainingCols);
152  uint32_t firstBlockLoopCnt = oneVecBlocks * (height / 2);
153  uint32_t secondBlockLoopCnt = VXLIB_ceilingDiv(remainingCols, eleCount) * (height / 2);
154 
155  pKerPrivArgs->oneVecBlocks = oneVecBlocks;
156  pKerPrivArgs->remainingCols = remainingCols;
157  pKerPrivArgs->firstBlockWidth = firstBlockWidth;
158  pKerPrivArgs->firstBlockLoopCnt = firstBlockLoopCnt;
159  pKerPrivArgs->secondBlockLoopCnt = secondBlockLoopCnt;
160  pKerPrivArgs->isOddHeight = ((height % 2) == 0) ? 0u : 1u;
161 
162  se0Params.ICNT0 = (oneVecBlocks > 0) ? firstBlockWidth : eleCount;
163  se0Params.ICNT1 = VXLIB_ceilingDiv(height, 2);
164  se0Params.DIM1 = strideInElements * 2;
165  se0Params.DIMFMT = __SE_DIMFMT_2D;
166  se0Params.ELETYPE = SE_ELETYPE;
167  se0Params.VECLEN = SE_VECLEN;
168 
169  se1Params.ICNT0 = eleCount;
170  se1Params.ICNT1 = VXLIB_ceilingDiv(height, 2);
171  se1Params.DIM1 = strideInElements * 2;
172  se1Params.DIMFMT = __SE_DIMFMT_2D;
173  se1Params.ELETYPE = SE_ELETYPE;
174  se1Params.VECLEN = SE_VECLEN;
175  se1Params.DECDIM2 = __SE_DECDIM_DIM2;
176  se1Params.DECDIM2_WIDTH = remainingCols;
177 
178  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_PARAM_OFFSET) = se0Params;
179  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE1_PARAM_OFFSET) = se1Params;
180 
181  /* ---------------------------------------------------------------------------------------------------
182  SE/SA PARAM SETTINGS FOR MAX/MIN COUNT
183  ------------------------------------------------------------------------------------------------------- */
184 
185  __SE_TEMPLATE_v1 se0CountParams = __gen_SE_TEMPLATE_v1();
186  se0CountParams.ICNT0 = width;
187  se0CountParams.ICNT1 = height;
188  se0CountParams.DIM1 = strideInElements;
189  se0CountParams.ICNT2 = 2;
190  se0CountParams.DIM2 = 0;
191  se0CountParams.DIMFMT = __SE_DIMFMT_3D;
192  se0CountParams.ELETYPE = SE_ELETYPE;
193  se0CountParams.VECLEN = SE_VECLEN;
194 
195  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_COUNT_PARAM_OFFSET) = se0CountParams;
196 
197  __SA_TEMPLATE_v1 sa0CountParams = __gen_SA_TEMPLATE_v1();
198  sa0CountParams.ICNT0 = width;
199  sa0CountParams.ICNT1 = height;
200  sa0CountParams.DIM1 = strideInElements;
201  sa0CountParams.ICNT2 = 2;
202  sa0CountParams.DIM2 = 0;
203  sa0CountParams.DIMFMT = __SA_DIMFMT_3D;
204  sa0CountParams.VECLEN = SA_VECLEN;
205 
206  *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_COUNT_PARAM_OFFSET) = sa0CountParams;
207 
208  /* ---------------------------------------------------------------------------------------------------
209  SE/SA PARAM SETTINGS FOR MAX/MIN LOCATION
210  ------------------------------------------------------------------------------------------------------- */
211 
212  typedef typename VXLIB_vec_type<VXLIB_UINT32>::type vecIndexType;
213 
214  __SE_VECLEN SE_VECLEN_32BIT = c7x::se_veclen<vecIndexType>::value;
215  __SA_VECLEN SA_VECLEN_32BIT = c7x::sa_veclen<vecIndexType>::value;
216 
217  __SE_TEMPLATE_v1 se0NewLocParams = __gen_SE_TEMPLATE_v1();
218  se0NewLocParams.ICNT0 = width;
219  se0NewLocParams.ICNT1 = height;
220  se0NewLocParams.DIM1 = strideInElements;
221  se0NewLocParams.ICNT2 = 2;
222  se0NewLocParams.DIM2 = 0;
223  se0NewLocParams.DIMFMT = __SE_DIMFMT_3D;
224  se0NewLocParams.ELETYPE = SE_ELETYPE;
225  se0NewLocParams.VECLEN = SE_VECLEN_32BIT;
226  setSEPromoteMode<dType>(&se0NewLocParams);
227 
228  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_LOC_PARAM_OFFSET) = se0NewLocParams;
229 
230  __SA_TEMPLATE_v1 sa0NewLocParams = __gen_SA_TEMPLATE_v1();
231  sa0NewLocParams.ICNT0 = width;
232  sa0NewLocParams.ICNT1 = height;
233  sa0NewLocParams.DIM1 = strideInElements;
234  sa0NewLocParams.ICNT2 = 2;
235  sa0NewLocParams.DIM2 = 0;
236  sa0NewLocParams.DIMFMT = __SA_DIMFMT_3D;
237  sa0NewLocParams.VECLEN = SA_VECLEN_32BIT;
238 
239  *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_LOC_PARAM_OFFSET) = sa0NewLocParams;
240 
241  return status;
242 }

288 template <typename INVECTYPE>
289 static inline void locUpdateMinMax(INVECTYPE vIn,
290  __vpred predSA,
291  INVECTYPE numericLimitVec32BIT,
292  INVECTYPE comparisonVec,
293  c7x::uint_vec rowIdxTrackVec,
294  c7x::uint_vec colIdxTrackVec,
295  uint32_t halfEleCount,
296  uint32_t * locCapacity,
297  uint32_t *& pLocStore)
298 {
299  typedef typename c7x::element_type_of<INVECTYPE>::type signType;
300 
301  vIn = __select(predSA, vIn, numericLimitVec32BIT);
302 
303  __vpred predMin = __cmp_eq_pred(comparisonVec, vIn);
304  uint32_t minEleFound = countEle<signType>(predMin);
305 
306  c7x::uint_vec rowIdxMin = c7x::as_uint_vec(__gather_set_bits(predMin, c7x::as_char_vec(rowIdxTrackVec)));
307  c7x::uint_vec idx_lo = __shuffle_stride2_low_low(colIdxTrackVec, rowIdxMin);
308  c7x::uint_vec idx_hi = __shuffle_stride2_high_high(colIdxTrackVec, rowIdxMin);
309 
310  uint32_t minEleCount_lo = __min(minEleFound, halfEleCount);
311  minEleCount_lo = __min(minEleCount_lo, *locCapacity);
312  *locCapacity = *locCapacity - minEleCount_lo;
313 
314  uint32_t minEleCount_hi = minEleFound - minEleCount_lo;
315  minEleCount_hi = __min(minEleCount_hi, *locCapacity);
316  *locCapacity = *locCapacity - minEleCount_hi;
317 
318  __vpred predMin_lo = __mask_int(minEleCount_lo * 2u);
319  __vpred predMin_hi = __mask_int(minEleCount_hi * 2u);
320 
321  __vstore_pred(predMin_lo, (c7x::uint_vec *) pLocStore, idx_lo);
322  pLocStore = pLocStore + (minEleCount_lo * 2u);
323 
324  __vstore_pred(predMin_hi, (c7x::uint_vec *) pLocStore, idx_hi);
325  pLocStore = pLocStore + (minEleCount_hi * 2u);
326 }

328 // this method performs pixel-wise addition of two input images
329 template <typename dType>
330 VXLIB_STATUS VXLIB_minMaxLoc_exec_ci(VXLIB_kernelHandle handle,
331  void *restrict pIn,
332  void *restrict pMinVal,
333  void *restrict pMaxVal,
334  void *restrict pMinCount,
335  void *restrict pMaxCount,
336  void *restrict pMinLocCapacity,
337  void *restrict pMaxLocCapacity,
338  void *restrict pMinLoc,
339  void *restrict pMaxLoc,
340  void *restrict pStartX,
341  void *restrict pStartY)
342 {
343  VXLIB_DEBUGPRINTFN(0, "%s\n", "Entering function");
344 
345  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
346 
347  // typecast handle (void) to struct pointer type associated to kernel
348  VXLIB_minMaxLoc_PrivArgs *pKerPrivArgs = (VXLIB_minMaxLoc_PrivArgs *) handle;
349 
350  typedef typename c7x::make_full_vector<dType>::type vec;
351 
352  uint32_t eleCount = c7x::element_count_of<vec>::value;
353  // create local pointers
354  dType restrict *pInLocal = (dType *) pIn;
355  dType restrict *pMinValLocal = (dType *) pMinVal;
356  dType restrict *pMaxValLocal = (dType *) pMaxVal;
357  uint32_t restrict *pMinCountLocal = (uint32_t *) pMinCount;
358  uint32_t restrict *pMaxCountLocal = (uint32_t *) pMaxCount;
359  uint32_t restrict *pMinLocLocal = (uint32_t *) pMinLoc;
360  uint32_t restrict *pMaxLocLocal = (uint32_t *) pMaxLoc;
361  uint32_t restrict *pStartXLocal = (uint32_t *) pStartX;
362  uint32_t restrict *pStartYLocal = (uint32_t *) pStartY;
363  uint32_t restrict *minLocCapacity = (uint32_t *) pMinLocCapacity;
364  uint32_t restrict *maxLocCapacity = (uint32_t *) pMaxLocCapacity;
365 
366  dType origMinVal = *pMinValLocal;
367  dType origMaxVal = *pMaxValLocal;
368  uint32_t origMinCount = 0;
369  uint32_t origMaxCount = 0;
370 
371  // obtain dimensional parameters
372  uint32_t width = pKerPrivArgs->width;
373  uint32_t height = pKerPrivArgs->height;
374  uint32_t strideInElements = pKerPrivArgs->strideInElements;
375  uint32_t oneVecBlocks = pKerPrivArgs->oneVecBlocks;
376  uint32_t remainingCols = pKerPrivArgs->remainingCols;
377  uint32_t firstBlockWidth = pKerPrivArgs->firstBlockWidth;
378  uint32_t firstBlockLoopCnt = pKerPrivArgs->firstBlockLoopCnt;
379  uint32_t secondBlockLoopCnt = pKerPrivArgs->secondBlockLoopCnt;
380  bool isOddHeight = pKerPrivArgs->isOddHeight;
381 
382  // address of SE and SA parameters
383  uint8_t *pBlock = pKerPrivArgs->bufPblock;
384 
385  // retrieve SE and SA parameters
386  __SE_TEMPLATE_v1 se0Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_PARAM_OFFSET);
387  __SE_TEMPLATE_v1 se1Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE1_PARAM_OFFSET);
388 
389  /* ----------------------------------------------------------------------------------
390  FIND MIN MAX VALUE
391  ---------------------------------------------------------------------------------- */
392 
393  __vpred predSelectLocal = predSelectMask;
394  vec numericMaxVec = (vec) std::numeric_limits<dType>::max();
395  vec numericMinVec = (vec) std::numeric_limits<dType>::min();
396  vec minVecSE0 = (vec)(*pMinValLocal);
397  vec maxVecSE0 = (vec)(*pMaxValLocal);
398  vec minVecSE1 = (vec)(*pMinValLocal);
399  vec maxVecSE1 = (vec)(*pMaxValLocal);
400  vec minVecSE0_rem = (vec)(*pMinValLocal);
401  vec maxVecSE0_rem = (vec)(*pMaxValLocal);
402  vec minVecSE1_rem = (vec)(*pMinValLocal);
403  vec maxVecSE1_rem = (vec)(*pMaxValLocal);
404 
405  if (oneVecBlocks > 0) {
406  __SE0_OPEN(pInLocal, se0Params);
407  if (height > 1) {
408  se0Params.ICNT1 = height / 2;
409  __SE1_OPEN(pInLocal + strideInElements, se0Params);
410  }
411 
412  for (uint32_t i = 0; i < firstBlockLoopCnt; i++) {
413  vec vSE0 = c7x::strm_eng<0, vec>::get_adv();
414  vec vSE1 = c7x::strm_eng<1, vec>::get_adv();
415 
416  minVecSE0 = __min(minVecSE0, vSE0);
417  maxVecSE0 = __max(maxVecSE0, vSE0);
418  minVecSE1 = __min(minVecSE1, vSE1);
419  maxVecSE1 = __max(maxVecSE1, vSE1);
420  }
421 
422  if (isOddHeight) {
423  for (uint32_t j = 0; j < oneVecBlocks; j++) {
424  vec vSE0 = c7x::strm_eng<0, vec>::get_adv();
425 
426  minVecSE0 = __min(minVecSE0, vSE0);
427  maxVecSE0 = __max(maxVecSE0, vSE0);
428  }
429  }
430  __SE0_CLOSE();
431  __SE1_CLOSE();
432  }
433  if (remainingCols > 0) {
434  __SE0_OPEN(pInLocal + firstBlockWidth, se1Params);
435  if (height > 1) {
436  se1Params.ICNT1 = height / 2;
437  __SE1_OPEN(pInLocal + strideInElements + firstBlockWidth, se1Params);
438  }
439  for (uint32_t i = 0; i < secondBlockLoopCnt; i++) {
440  vec vSE0 = c7x::strm_eng<0, vec>::get_adv();
441  vec vSE1 = c7x::strm_eng<1, vec>::get_adv();
442 
443  minVecSE0_rem = __min(minVecSE0_rem, vSE0);
444  maxVecSE0_rem = __max(maxVecSE0_rem, vSE0);
445  minVecSE1_rem = __min(minVecSE1_rem, vSE1);
446  maxVecSE1_rem = __max(maxVecSE1_rem, vSE1);
447  }
448 
449  if (isOddHeight) {
450  vec vSE0 = c7x::strm_eng<0, vec>::get_adv();
451 
452  minVecSE0_rem = __min(minVecSE0_rem, vSE0);
453  maxVecSE0_rem = __max(maxVecSE0_rem, vSE0);
454  }
455  minVecSE0_rem = __select(predSelectLocal, minVecSE0_rem, numericMaxVec);
456  maxVecSE0_rem = __select(predSelectLocal, maxVecSE0_rem, numericMinVec);
457  minVecSE1_rem = __select(predSelectLocal, minVecSE1_rem, numericMaxVec);
458  maxVecSE1_rem = __select(predSelectLocal, maxVecSE1_rem, numericMinVec);
459 
460  __SE0_CLOSE();
461  __SE1_CLOSE();
462  }
463 
464  /* ---------- FIND MIN VAL ---------- */
465  {
466  vec minVec1 = __min(minVecSE0, minVecSE1);
467  vec minVec2 = __min(minVecSE0_rem, minVecSE1_rem);
468  vec minVecFinal = __min(minVec1, minVec2);
469  *pMinValLocal = VXLIB_horizontalMin<dType, vec>(minVecFinal);
470  }
471 
472  /* ---------- FIND MAX VAL ---------- */
473  {
474  vec maxVec1 = __max(maxVecSE0, maxVecSE1);
475  vec maxVec2 = __max(maxVecSE0_rem, maxVecSE1_rem);
476  vec maxVecFinal = __max(maxVec1, maxVec2);
477  *pMaxValLocal = VXLIB_horizontalMax<dType, vec>(maxVecFinal);
478  }
479 
480  VXLIB_DEBUGPRINTFN(0, "\norigMinVal: 0x%08X \nnew minVal: 0x%08X \norigMaxVal: 0x%08X \nnew maxVal: 0x%08X\n",
481  origMinVal, *pMinValLocal, origMaxVal, *pMaxValLocal);
482 
483  /* Only if the minCount or maxCount are available for counting we proceed*/
484  if (pMinCountLocal || pMaxCountLocal) {
485  __SE_TEMPLATE_v1 se0CountParams = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_COUNT_PARAM_OFFSET);
486  __SA_TEMPLATE_v1 sa0CountParams = *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_COUNT_PARAM_OFFSET);
487  __SE0_OPEN(pInLocal, se0CountParams);
488  __SA0_OPEN(sa0CountParams);
489 
490  uint32_t widthBlocks = VXLIB_ceilingDiv(width, eleCount);

517  }
