57 // this method computes M x N dilate via natural C code
58 template <typename dTypeIn, typename dTypeOut>
59 VXLIB_STATUS VXLIB_dilate_exec_cn(VXLIB_kernelHandle handle, void *restrict pIn, void *restrict pOut)
60 {
61 
62  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
63 
64 #if VXLIB_DEBUGPRINT
65  printf("Enter VXLIB_dilate_exec_cn\n");
66 #endif
67 
68  // typecast handle (void) to struct pointer type associated to kernel
69  VXLIB_dilate_PrivArgs *pKerPrivArgs = (VXLIB_dilate_PrivArgs *) handle;
70 
71  // obtain image parameters and overflow policy
72  size_t width = pKerPrivArgs->width;
73  size_t height = pKerPrivArgs->height;
74  size_t strideInElements = pKerPrivArgs->strideInElements;
75  size_t strideOutElements = pKerPrivArgs->strideOutElements;
76 
77  size_t filterWidth = pKerPrivArgs->pKerInitArgs.filterWidth;
78  size_t filterHeight = pKerPrivArgs->pKerInitArgs.filterHeight;
79 
80  // create local pointers
81  dTypeIn *restrict pInLocal = (dTypeIn *) pIn;
82  dTypeOut *restrict pOutLocal = (dTypeOut *) pOut;
83 
84 #if VXLIB_DEBUGPRINT
85  printf("In VXLIB_dilate_exec_cn, width: %d, height: %d\n", width, height);
86 #endif
87 
88  int32_t x, y, i, j;
89  dTypeIn max_val;
90  int32_t outWidth = (width - filterWidth + 1);
91  int32_t outHeight = (height - filterHeight + 1);
92 
93  for (x = 0; x < outHeight; x++) {
94  for (y = 0; y < outWidth; y++) {
95 
96  max_val = 0;
97 
98  for (i = 0; i < filterHeight; i++) {
99  for (j = 0; j < filterWidth; j++) {
100 
101  max_val = std::max((pInLocal[(x + i) * strideInElements + (y + j)]), max_val);
102  }
103  }
104 
105  pOutLocal[x * strideOutElements + y] = max_val;
106  }
107  }
108 
109  return (status);
110 }
