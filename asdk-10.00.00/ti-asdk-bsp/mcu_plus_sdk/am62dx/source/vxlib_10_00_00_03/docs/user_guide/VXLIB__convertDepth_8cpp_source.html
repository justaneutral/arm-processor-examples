44 VXLIB_convertDepth_init_checkParams(VXLIB_kernelHandle handle,
45  const VXLIB_bufParams2D_t *bufParamsIn0,
46  const VXLIB_bufParams2D_t *bufParamsOut,
47  const VXLIB_convertDepth_InitArgs *pKerInitArgs)
48 {
49  VXLIB_STATUS status = VXLIB_SUCCESS;
50 
51 #if VXLIB_DEBUGPRINT
52  printf("Enter VXLIB_convertDepth_init_checkParams\n");
53 #endif
54  if (handle == NULL || bufParamsIn0 == NULL || bufParamsOut == NULL || pKerInitArgs == NULL) {
55  status = VXLIB_ERR_NULL_POINTER;
56  }
57 
58  if (status == VXLIB_SUCCESS) {
59  if (((bufParamsIn0->data_type != VXLIB_INT16) || (bufParamsOut->data_type != VXLIB_UINT8)) &&
60  ((bufParamsIn0->data_type != VXLIB_UINT8) || (bufParamsOut->data_type != VXLIB_INT16))) {
61  status = VXLIB_ERR_INVALID_TYPE;
62  }
63  else {
64  /* Nothing to do here */
65  }
66  }
67 
68  if (status == VXLIB_SUCCESS) {
69 
70  if ((bufParamsIn0->dim_x != bufParamsOut->dim_x) || (bufParamsIn0->dim_y != bufParamsOut->dim_y) ||
71  (bufParamsIn0->stride_y / VXLIB_sizeof(bufParamsIn0->data_type) < bufParamsIn0->dim_x) ||
72  (bufParamsOut->stride_y / VXLIB_sizeof(bufParamsOut->data_type) < bufParamsOut->dim_x)) {
73  status = VXLIB_ERR_INVALID_DIMENSION;
74  }
75  }
76 
77  return status;
78 }

98 VXLIB_STATUS VXLIB_convertDepth_init(VXLIB_kernelHandle handle,
99  VXLIB_bufParams2D_t *bufParamsIn0,
100  VXLIB_bufParams2D_t *bufParamsOut,
101  const VXLIB_convertDepth_InitArgs *pKerInitArgs)
102 {
103  VXLIB_STATUS status = VXLIB_SUCCESS;
104  VXLIB_convertDepth_PrivArgs *pKerPrivArgs = (VXLIB_convertDepth_PrivArgs *) handle;
105 
106  uint32_t height = bufParamsOut->dim_y;
107  uint32_t width = bufParamsOut->dim_x;
108 
109  // obtain buffer datatypes
110  uint32_t dTypeIn0 = bufParamsIn0->data_type;
111  uint32_t dTypeOut = bufParamsOut->data_type;
112  uint32_t strideIn0Elements = bufParamsIn0->stride_y / VXLIB_sizeof(dTypeIn0);
113  uint32_t strideOutElements = bufParamsOut->stride_y / VXLIB_sizeof(dTypeOut);
114 
115  pKerPrivArgs->height = height;
116  pKerPrivArgs->width = width;
117  pKerPrivArgs->strideIn0Elements = strideIn0Elements;
118  pKerPrivArgs->strideOutElements = strideOutElements;
119  pKerPrivArgs->data_type = dTypeIn0;
120 
121  size_t SIMD_WIDTH = VXLIB_max_simd<VXLIB_INT16>::value;
122 
123  pKerPrivArgs->shift = pKerInitArgs->shift;
124  pKerPrivArgs->overFlowPolicy = pKerInitArgs->overFlowPolicy;
125 
126  if (width == strideIn0Elements) {
127  pKerPrivArgs->sePolicy = VXLIB_CONVERTDEPTH_SE_1D;
128  }
129  else if (width <= SIMD_WIDTH) {
130  pKerPrivArgs->sePolicy = VXLIB_CONVERTDEPTH_SE_NARROW;
131  }
132  else {
133  pKerPrivArgs->sePolicy = VXLIB_CONVERTDEPTH_SE_WIDE;
134  }
135 
136 #if VXLIB_DEBUGPRINT
137  printf("VXLIB_DEBUGPRINT Enter VXLIB_convertDepth_init\n");
138 #endif
139 
140  if (pKerInitArgs->funcStyle == VXLIB_FUNCTION_NATC) {
141  if (bufParamsIn0->data_type == VXLIB_UINT8) {
142  pKerPrivArgs->execute = VXLIB_convertDepth_exec_cn<uint8_t, int16_t>;
143  }
144  else if (bufParamsIn0->data_type == VXLIB_INT16) {
145  pKerPrivArgs->execute = VXLIB_convertDepth_exec_cn<int16_t, uint8_t>;
146  }
147  else {
148  status = VXLIB_ERR_INVALID_TYPE;
149 #if VXLIB_DEBUGPRINT
150  printf("VXLIB_DEBUGPRINT CP 2 status %d\n", status);
151 #endif
152  }
153  }
154  else {
155  if (bufParamsIn0->data_type == VXLIB_UINT8) {
156 #if VXLIB_DEBUGPRINT
157  printf("VXLIB_DEBUGPRINT bufParamsIn->data_type == VXLIB_UINT8\n");
158 #endif
159  pKerPrivArgs->execute = VXLIB_convertDepth_exec_ci<uint8_t, int16_t>;
160  status =
161  VXLIB_convertDepth_init_ci<VXLIB_UINT8, VXLIB_INT16>(handle, bufParamsIn0, bufParamsOut, pKerInitArgs);
162  }
163  else if (bufParamsIn0->data_type == VXLIB_INT16) {
164  pKerPrivArgs->execute = VXLIB_convertDepth_exec_ci<int16_t, uint8_t>;
165  status =
166  VXLIB_convertDepth_init_ci<VXLIB_INT16, VXLIB_UINT8>(handle, bufParamsIn0, bufParamsOut, pKerInitArgs);
167  }
168  else {
169  status = VXLIB_ERR_INVALID_TYPE;
170  }
171  }
172 #if VXLIB_DEBUGPRINT
173  printf("VXLIB_DEBUGPRINT CP 3 status %d\n", status);
174 #endif
175  return status;
176 }
