59 // this method initializes the kernel-specific parameters
60 // mainly, the streaming engine and streaming address generators
61 template <uint32_t dTypeIn0, uint32_t dTypeIn1, uint32_t dTypeOut>
62 VXLIB_STATUS VXLIB_magnitude_init_ci(VXLIB_kernelHandle handle,
63  const VXLIB_bufParams2D_t *bufParamsIn0,
64  const VXLIB_bufParams2D_t *bufParamsIn1,
65  const VXLIB_bufParams2D_t *bufParamsOut,
66  const VXLIB_magnitude_InitArgs *pKerInitArgs)
67 {
68  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
69 
70  // structs to hold SE and SA parameters
71  __SE_TEMPLATE_v1 se0Params = __gen_SE_TEMPLATE_v1();
72  __SE_TEMPLATE_v1 se1Params = __gen_SE_TEMPLATE_v1();
73  __SA_TEMPLATE_v1 sa0Params = __gen_SA_TEMPLATE_v1();
74 
75  // typecast handle (void) to struct pointer type associated to kernel
76  VXLIB_magnitude_PrivArgs *pKerPrivArgs = (VXLIB_magnitude_PrivArgs *) handle;
77 
78  uint8_t *pBlock = pKerPrivArgs->bufPblock; // address to retrieve to store SE/SA params
79 
80  // set SE0, SE1, and SA0 params
81  pKerPrivArgs->numBlocks = VXLIB_SE0SE1SA0_init<dTypeIn0, dTypeIn1, dTypeOut>(
82  &se0Params, &se1Params, &sa0Params, bufParamsIn0, bufParamsIn1, bufParamsOut);
83 
84  /**************************/
85  /* Store SE and SA params */
86  /**************************/
87 
88  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_PARAM_OFFSET) = se0Params;
89  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE1_PARAM_OFFSET) = se1Params;
90  *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_PARAM_OFFSET) = sa0Params;
91 
92  return status;
93 }

139 // this method performs pixel-wise addition of two input images
140 template <typename dTypeIn0, typename dTypeIn1, typename dTypeOut>
141 VXLIB_STATUS
142 VXLIB_magnitude_exec_ci(VXLIB_kernelHandle handle, void *restrict pIn0, void *restrict pIn1, void *restrict pOut)
143 {
144  // typecast handle (void) to struct pointer type associated to kernel
145  VXLIB_magnitude_PrivArgs *pKerPrivArgs = (VXLIB_magnitude_PrivArgs *) handle;
146 
147  // structs to retrieve SE and SA paramters
148  __SE_TEMPLATE_v1 se0Params;
149  __SE_TEMPLATE_v1 se1Params;
150  __SA_TEMPLATE_v1 sa0Params;
151 
152  // create local pointers
153  dTypeIn0 *restrict pInLocal0 = (dTypeIn0 *) pIn0;
154  dTypeIn1 *restrict pInLocal1 = (dTypeIn1 *) pIn1;
155  dTypeOut *restrict pOutLocal = (dTypeOut *) pOut;
156 
157 #if VXLIB_DEBUGPRINT
158  printf("Enter VXLIB_magnitude_exec_ci\n");
159 #endif
160 
161  // address of SE and SA parameters
162  uint8_t *pBlock = pKerPrivArgs->bufPblock;
163 
164  // retrieve SE and SA parameters
165  se0Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_PARAM_OFFSET);
166  se1Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE1_PARAM_OFFSET);
167  sa0Params = *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_PARAM_OFFSET);
168 
169  // open SEs to fetch Input samples
170  __SE0_OPEN(pInLocal0, se0Params);
171  __SE1_OPEN(pInLocal1, se1Params);
172 
173  // open SA0 to write output samples
174  __SA0_OPEN(sa0Params);
175 
176  size_t numBlocks = pKerPrivArgs->numBlocks;
177 
178  // derive c7x vector type from template typename
179  typedef typename c7x::make_full_vector<dTypeIn0>::type vec;
180 
181  // int_32 intermediate variables
182  c7x::int_vec a_sqrl;
183  c7x::int_vec a_sqrh;
184 
185  c7x::int_vec b_sqrl;
186  c7x::int_vec b_sqrh;
187 
188  c7x::int_vec suml_i;
189  c7x::int_vec sumh_i;
190 
191  c7x::int_vec sqrtl_i;
192  c7x::int_vec sqrth_i;
193 
194  // float intermediate variables
195  c7x::float_vec suml_f;
196  c7x::float_vec sumh_f;
197 
198  // define constants for square root
199  c7x::float_vec half, OneP5;
200  half = (c7x::float_vec) 0.5;
201  OneP5 = (c7x::float_vec) 1.5;
202 
203  // Create variables employed on sqrt computation
204  c7x::float_vec sqrtl_f, sqrth_f, p0l, p1l, d0l, p0h, p1h, d0h;
205 
206  /***********************************************************************/
207  /* Main Loop */
208  /***********************************************************************/
209  for (uint32_t counter = 0; counter < numBlocks; counter++) {
210  vec a = c7x::strm_eng<0, vec>::get_adv(); // fetch input 0 via SE0
211  vec b = c7x::strm_eng<1, vec>::get_adv(); // fetch input 1 via SE1
212  /***********************************************************************/
213  /* Square and summation computation */
214  /***********************************************************************/
215  __vmpyhw_vvw(a, a, a_sqrh, a_sqrl);
216  __vmpyhw_vvw(b, b, b_sqrh, b_sqrl);
217 
218  suml_i = __add_sat(a_sqrl, b_sqrl);
219  sumh_i = __add_sat(a_sqrh, b_sqrh);
220 
221  /***********************************************************************/
222  /* Square root computation */
223  /***********************************************************************/
224  // convert to float for square root calculation
225  suml_f = __int_to_float(suml_i);
226  sumh_f = __int_to_float(sumh_i);
227 
228  // Sqrt computation
229  // Reciprocal square root calculation for first half
230  p0l = __recip_sqrt(suml_f);
231  d0l = p0l * suml_f;
232  p1l = OneP5 - d0l * p0l * half;
233  sqrtl_f = suml_f * p0l * p1l;
234 
235  // Reciprocal square root calculation for second half
236  p0h = __recip_sqrt(sumh_f);
237  d0h = p0h * sumh_f;
238  p1h = OneP5 - d0h * p0h * half;
239  sqrth_f = sumh_f * p0h * p1h;
240 
241  /***********************************************************************/
242  /* Final data conversion and store */
243  /***********************************************************************/
244  // convert float to int_32
245  sqrtl_i = __float_to_int(sqrtl_f);
246  sqrth_i = __float_to_int(sqrth_f);
247 
248  // convert int_32 to int_16 with saturation
249  vec out = __pack_sat(sqrtl_i, sqrth_i);
250 
251  // store output via SA0
252  __vpred tmp = c7x::strm_agen<0, vec>::get_vpred(); // generate predicate
253  vec *addr = c7x::strm_agen<0, vec>::get_adv(pOutLocal); // generate address
254  __vstore_pred(tmp, addr, out); // store value
255  }
256 
257  // close SE/SA
258  __SE0_CLOSE();
259  __SE1_CLOSE();
260  __SA0_CLOSE();
261 
262  return VXLIB_SUCCESS;
263 }
