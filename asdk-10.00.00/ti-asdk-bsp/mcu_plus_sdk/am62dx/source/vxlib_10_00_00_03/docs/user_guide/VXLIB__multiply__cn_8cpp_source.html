49 template <typename dTypeIn0, typename dTypeIn1, typename dTypeOut>
50 VXLIB_STATUS
51 VXLIB_multiply_exec_cn(VXLIB_kernelHandle handle, void *restrict pIn0, void *restrict pIn1, void *restrict pOut)
52 {
53  VXLIB_DEBUGPRINTFN(0, "%s\n", "Enter Function");
54  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
55 
56  typedef typename c7x::make_full_vector<dTypeIn0>::type vec;
57  typedef
58  typename std::conditional<ELEM_COUNT(c7x::uchar_vec) == ELEM_COUNT(vec), int32_t, int64_t>::type typeResultInt;
59 
60  // typecast handle (void) to struct pointer type associated to kernel
61  VXLIB_multiply_PrivArgs *pKerPrivArgs = (VXLIB_multiply_PrivArgs *) handle;
62 
63  // obtain image parameters and overflow policy
64  size_t width = pKerPrivArgs->width;
65  size_t height = pKerPrivArgs->height;
66  size_t strideIn0Elements = pKerPrivArgs->strideIn0Elements;
67  size_t strideIn1Elements = pKerPrivArgs->strideIn1Elements;
68  size_t strideOutElements = pKerPrivArgs->strideOutElements;
69  VXLIB_OVERFLOW_POLICY overFlowPolicy = pKerPrivArgs->pKerInitArgs.overFlowPolicy;
70  VXLIB_F32 scale = pKerPrivArgs->pKerInitArgs.scale;
71 
72  // create local pointers
73  dTypeIn0 *restrict pInLocal0 = (dTypeIn0 *) pIn0;
74  dTypeIn1 *restrict pInLocal1 = (dTypeIn1 *) pIn1;
75  dTypeOut *restrict pOutLocal = (dTypeOut *) pOut;
76 
77  // variables to hold temporary values
78  uint32_t x, y, src0Index, src1Index, dstIndex;
79  int32_t unscaledIntResult, finalResult;
80  typeResultInt scaledIntResult;
81  VXLIB_D64 unscaledDoubleResult, scaledDoubleResut;
82 
83  {
84  for (y = 0; y < height; y++) {
85 
86  for (x = 0; x < width; x++) {
87  // printf("\n\nrow: %d col: %d\n", y, x);
88  src0Index = y * strideIn0Elements + x;
89  src1Index = y * strideIn1Elements + x;
90  dstIndex = y * strideOutElements + x;
91 
92  unscaledIntResult = pInLocal0[src0Index] * pInLocal1[src1Index];
93  unscaledDoubleResult = (double) unscaledIntResult;
94  scaledDoubleResut = scale * unscaledDoubleResult;
95 
96  scaledIntResult = (typeResultInt) scaledDoubleResut;
97 
98  /* Finally, overflow-check as per the target type and policy. */
99  if (overFlowPolicy == VXLIB_SATURATE) {
100  if (scaledIntResult > VXLIB_MULTIPLY_NUMERIC_MAX(dTypeOut)) {
101  finalResult = VXLIB_MULTIPLY_NUMERIC_MAX(dTypeOut);
102  }
103  else if (scaledIntResult < VXLIB_MULTIPLY_NUMERIC_MIN(dTypeOut)) {
104  finalResult = VXLIB_MULTIPLY_NUMERIC_MIN(dTypeOut);
105  }
106  else {
107  finalResult = (dTypeOut) scaledIntResult;
108  }
109  }
110  else {
111  /* Just for show: the final assignment will wrap too. */
112  finalResult = (dTypeOut) scaledIntResult;
113  }
114  pOutLocal[dstIndex] = (dTypeOut) finalResult;
115  }
116  }
117  }
118 
119  VXLIB_DEBUGPRINTFN(0, "%s\n", "Exit Function");
120  return (status);
121 }
