53 // this method initializes the kernel-specific parameters
54 // mainly, the streaming engine and streaming address generators
55 template <uint32_t dTypeIn0, uint32_t dTypeIn1, uint32_t dTypeOut>
56 VXLIB_STATUS VXLIB_subtract_init_ci(VXLIB_kernelHandle handle,
57  const VXLIB_bufParams2D_t * bufParamsIn0,
58  const VXLIB_bufParams2D_t * bufParamsIn1,
59  const VXLIB_bufParams2D_t * bufParamsOut,
60  const VXLIB_subtract_InitArgs *pKerInitArgs)
61 {
62  VXLIB_DEBUGPRINTFN(0, "%s\n", "Enter function");
63  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
64 
65  // structs to hold SE and SA parameters
66  __SE_TEMPLATE_v1 se0Params = __gen_SE_TEMPLATE_v1();
67  __SE_TEMPLATE_v1 se1Params = __gen_SE_TEMPLATE_v1();
68  __SA_TEMPLATE_v1 sa0Params = __gen_SA_TEMPLATE_v1();
69 
70  // typecast handle (void) to struct pointer type associated to kernel
71  VXLIB_subtract_PrivArgs *pKerPrivArgs = (VXLIB_subtract_PrivArgs *) handle;
72  uint8_t * pBlock = pKerPrivArgs->bufPblock; // address to retrieve to store SE/SA params
73 
74  // set SE0, SE1, and SA0 params
75  pKerPrivArgs->numBlocks = VXLIB_SE0SE1SA0_init<dTypeIn0, dTypeIn1, dTypeOut>(
76  &se0Params, &se1Params, &sa0Params, bufParamsIn0, bufParamsIn1, bufParamsOut);
77 
78  /**************************/
79  /* Store SE and SA params */
80  /**************************/
81 
82  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_PARAM_OFFSET) = se0Params;
83  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE1_PARAM_OFFSET) = se1Params;
84  *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_PARAM_OFFSET) = sa0Params;
85 
86  VXLIB_DEBUGPRINTFN(0, "%s\n", "Exit function");
87 
88  return status;
89 }

133 template <>
134 inline c7x::short_vec
135 subtractSaturate<c7x::short_vec>(c7x::short_vec vecIn1, c7x::short_vec vecIn2, c7x::uchar_vec zeroVec)
136 {
137  return __sub_sat(vecIn1, vecIn2);
138 }

140 template <>
141 inline c7x::uchar_vec
142 subtractSaturate<c7x::uchar_vec>(c7x::uchar_vec vecIn1, c7x::uchar_vec vecIn2, c7x::uchar_vec zeroVec)
143 {
144  c7x::uchar_vec vecSub = vecIn1 - vecIn2;
145  __vpred predComp = __cmp_gt_pred(vecIn1, vecIn2);
146  return __select(predComp, vecSub, zeroVec);
147 }

149 // this method subtracts two inputs, saturates the result, and writes the result to output buffer
150 // it is assumed that SEs and SA are open before calling this method
151 
152 template <typename dType> static inline void VXLIB_subtract_util_sat(dType *pOut, size_t size)
153 {
154  VXLIB_DEBUGPRINTFN(0, "%s\n", "Enter function");
155 
156  typedef typename c7x::make_full_vector<dType>::type vec;
157 
158  dType * pOutLocal = (dType *) pOut;
159  c7x::uchar_vec zeroVec = (c7x::uchar_vec) 0;
160 
161  for (uint32_t counter = 0; counter < size; counter++) {
162  vec a = c7x::strm_eng<0, vec>::get_adv(); // fetch input 0 via SE0
163  vec b = c7x::strm_eng<1, vec>::get_adv(); // fetch input 1 via SE1
164 
165  vec out = subtractSaturate<vec>(a, b, zeroVec);
166 
167  // store output via SA0
168  __vpred tmp = c7x::strm_agen<0, vec>::get_vpred(); // generate predicate
169  vec * addr = c7x::strm_agen<0, vec>::get_adv(pOutLocal); // generate address
170  __vstore_pred(tmp, addr, out); // store value
171  }
172  VXLIB_DEBUGPRINTFN(0, "%s\n", "Exit function");
173 
174  return;
175 }

183 template <typename dType> static inline void VXLIB_subtract_util_wrap(dType *pOut, size_t size)
184 {
185  VXLIB_DEBUGPRINTFN(0, "%s\n", "Enter function");
186  // derive c7x vector type from template typename
187  typedef typename c7x::make_full_vector<dType>::type vec;
188 
189  for (uint32_t counter = 0; counter < size; counter++) {
190  vec a = c7x::strm_eng<0, vec>::get_adv(); // fetch input 0 via SE0
191  vec b = c7x::strm_eng<1, vec>::get_adv(); // fetch input 1 via SE1
192 
193  vec out = a - b;
194 
195  // store output via SA0
196  __vpred tmp = c7x::strm_agen<0, vec>::get_vpred(); // generate predicate
197  vec * addr = c7x::strm_agen<0, vec>::get_adv(pOut); // generate address
198  __vstore_pred(tmp, addr, out); // store value
199  }
200  VXLIB_DEBUGPRINTFN(0, "%s\n", "Exit function");
201 
202  return;

211 // this method performs pixel-wise addition of two input images
212 template <typename dTypeIn0, typename dTypeIn1, typename dTypeOut>
213 VXLIB_STATUS
214 VXLIB_subtract_exec_ci(VXLIB_kernelHandle handle, void *restrict pIn0, void *restrict pIn1, void *restrict pOut)
215 {
216  VXLIB_DEBUGPRINTFN(0, "%s\n", "Enter function");
217  // typecast handle (void) to struct pointer type associated to kernel
218  VXLIB_subtract_PrivArgs *pKerPrivArgs = (VXLIB_subtract_PrivArgs *) handle;
219 
220  // structs to retrieve SE and SA paramters
221  __SE_TEMPLATE_v1 se0Params;
222  __SE_TEMPLATE_v1 se1Params;
223  __SA_TEMPLATE_v1 sa0Params;
224 
225  // create local pointers
226  dTypeIn0 *restrict pInLocal0 = (dTypeIn0 *) pIn0;
227  dTypeIn1 *restrict pInLocal1 = (dTypeIn1 *) pIn1;
228  dTypeOut *restrict pOutLocal = (dTypeOut *) pOut;
229 
230  // address of SE and SA parameters
231  uint8_t *pBlock = pKerPrivArgs->bufPblock;
232 
233  // retrieve SE and SA parameters
234  se0Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_PARAM_OFFSET);
235  se1Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE1_PARAM_OFFSET);
236  sa0Params = *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_PARAM_OFFSET);
237 
238  // open SEs to fetch Input samples
239  uint8_t subtractPolicy = pKerPrivArgs->pKerInitArgs.subtractPolicy;
240 
241  if (subtractPolicy == 0) {
242  __SE0_OPEN(pInLocal0, se0Params);
243  __SE1_OPEN(pInLocal1, se1Params);
244  }
245  else {
246  __SE0_OPEN(pInLocal1, se1Params);
247  __SE1_OPEN(pInLocal0, se0Params);
248  }
249 
250  // open SA0 to write output samples
251  __SA0_OPEN(sa0Params);
252 
253  VXLIB_OVERFLOW_POLICY overFlowPolicy = pKerPrivArgs->pKerInitArgs.overFlowPolicy;
254  size_t numBlocks = pKerPrivArgs->numBlocks;
255 
256  // call appropriate compute loop based on overflow policy
257  if (overFlowPolicy == VXLIB_SATURATE) {
258  VXLIB_subtract_util_sat<dTypeOut>(pOutLocal, numBlocks);
259  }
260  else {
261  VXLIB_subtract_util_wrap<dTypeOut>(pOutLocal, numBlocks);
262  }
263 
264  // close SE/SA
265  __SE0_CLOSE();
266  __SE1_CLOSE();
267  __SA0_CLOSE();
268  VXLIB_DEBUGPRINTFN(0, "%s\n", "Exit function");
269 
270  return VXLIB_SUCCESS;
271 }

307 void VXLIB_subtract_perfEst(VXLIB_kernelHandle handle, size_t *archCycles, size_t *estCycles)
308 {
309 
310  // typecast handle (void) to struct pointer type associated to kernel
311  VXLIB_subtract_PrivArgs *pKerPrivArgs = (VXLIB_subtract_PrivArgs *) handle;
312 
313  // obtain loop count for compute loop

315  size_t overheadCnt = 56; // profiled code before entering compute loop
