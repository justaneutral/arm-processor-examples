48 // this method computes or in via natural C code
49 template <typename dTypeIn0>
50 VXLIB_STATUS VXLIB_or_exec_cn(VXLIB_kernelHandle handle, void *restrict pIn0, void *restrict pIn1, void *restrict pOut)
51 {
52 
53  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
54 
55  // typecast handle (void) to struct pointer type associated to kernel
56  VXLIB_or_PrivArgs *pKerPrivArgs = (VXLIB_or_PrivArgs *) handle;
57 
58  // obtain image parameters and overflow policy
59  size_t width = pKerPrivArgs->width;
60  size_t height = pKerPrivArgs->height;
61  size_t strideIn0Elements = pKerPrivArgs->strideIn0Elements;
62  size_t strideIn1Elements = pKerPrivArgs->strideIn1Elements;
63  size_t strideOutElements = pKerPrivArgs->strideOutElements;
64 
65 #if VXLIB_DEBUGPRINT
66  printf("Enter VXLIB_or_exec_cn\n");
67 #endif
68 
69  // create local pointers
70  dTypeIn0 *restrict pInLocal0 = (dTypeIn0 *) pIn0;
71  dTypeIn0 *restrict pInLocal1 = (dTypeIn0 *) pIn1;
72  dTypeIn0 *restrict pOutLocal = (dTypeIn0 *) pOut;
73 
74  // variables to hold temporary values
75  dTypeIn0 a;
76  dTypeIn0 b;
77 
78 #if VXLIB_DEBUGPRINT
79  printf("In VXLIB_or_exec_cn, width: %d, height: %d\n", width, height);
80 #endif
81 
82  // perform or
83  for (uint32_t i = 0; i < height; i++) {
84  for (uint32_t j = 0; j < width; j++) {
85  a = pInLocal0[i * strideIn0Elements + j];
86  b = pInLocal1[i * strideIn1Elements + j];
87 
88  pOutLocal[i * strideOutElements + j] = a | b;
89  }
90  }
91 
92  return (status);
93 }
