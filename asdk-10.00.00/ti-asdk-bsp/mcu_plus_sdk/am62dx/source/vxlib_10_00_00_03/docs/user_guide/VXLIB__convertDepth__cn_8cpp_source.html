40 inline uint8_t VXLIB_convertDepth_sat_cn(int16_t A, uint8_t shift)
41 {
42  A = (int16_t)((uint32_t)A >> (uint32_t)shift);
43 
44  A = A < 0 ? 0 : A;
45  A = A > 255 ? 255 : A;
46 
47  uint8_t res = A;
48  return res;
49 }

67 template <typename dTypeIn0, typename dTypeOut>
68 VXLIB_STATUS VXLIB_convertDepth_exec_cn(VXLIB_kernelHandle handle, void *restrict pIn0, void *restrict pOut)
69 {
70  VXLIB_STATUS status = VXLIB_SUCCESS;
71 
72  VXLIB_convertDepth_PrivArgs *pKerPrivArgs = (VXLIB_convertDepth_PrivArgs *) handle;
73 
74  int32_t m = 0;
75  int32_t n = 0;
76 
77  int32_t height = pKerPrivArgs->height;
78  int32_t width = pKerPrivArgs->width;
79  int32_t strideIn0 = pKerPrivArgs->strideIn0Elements;
80  int32_t strideOut = pKerPrivArgs->strideOutElements;
81  uint8_t shift = pKerPrivArgs->shift;
82 
83 #if VXLIB_DEBUGPRINT
84  printf("Enter VXLIB_convertDepth_exec_cn\n");
85 #endif
86 
87 #if VXLIB_DEBUGPRINT
88  printf("VXLIB_convertDepth_exec_cn: height = %d, width = %d\n", height, width);
89 #endif
90 
91  const dTypeIn0 *restrict A = (const dTypeIn0 *) pIn0;
92  dTypeOut *restrict C = (dTypeOut *) pOut;
93 
94  if (pKerPrivArgs->overFlowPolicy == VXLIB_SATURATE) {
95  for (m = 0; m < height; m++) {
96  for (n = 0; n < width; n++) {
97  C[n + m * strideOut] = VXLIB_convertDepth_sat_cn(A[n + m * strideIn0], shift);
98  }
99  }
100  }
101  else {
102  for (m = 0; m < height; m++) {
103  for (n = 0; n < width; n++) {
104  C[n + m * strideOut] = VXLIB_convertDepth_wrap_cn(A[n + m * strideIn0], C[n + m * strideOut], shift);
105  }
106  }
107  }
108 
109  return status;
110 }
