61 // this method checks the initialization parameters for the VXLIB_subtract kernel
62 VXLIB_STATUS
63 VXLIB_subtract_init_checkParams(VXLIB_kernelHandle handle,
64  const VXLIB_bufParams2D_t * bufParamsIn0,
65  const VXLIB_bufParams2D_t * bufParamsIn1,
66  const VXLIB_bufParams2D_t * bufParamsOut,
67  const VXLIB_subtract_InitArgs *pKerInitArgs)
68 {
69  VXLIB_STATUS status = VXLIB_SUCCESS;
70 
71  // check for dimensions and datatype combinations
72  if ((handle == NULL) || (bufParamsIn0 == NULL) || (bufParamsIn1 == NULL) || (bufParamsOut == NULL) || (pKerInitArgs == NULL)) {
73  status = VXLIB_ERR_NULL_POINTER;
74  }
75 
76  if (status == VXLIB_SUCCESS) {
77  // obtain input0 buffer parameters
78  uint32_t dTypeIn0 = bufParamsIn0->data_type;
79  uint32_t widthIn0 = bufParamsIn0->dim_x;
80  uint32_t heightIn0 = bufParamsIn0->dim_y;
81  uint32_t strideIn0 = bufParamsIn0->stride_y;
82 
83  // obtain input1 buffer parameters
84  uint32_t dTypeIn1 = bufParamsIn1->data_type;
85  uint32_t widthIn1 = bufParamsIn1->dim_x;
86  uint32_t heightIn1 = bufParamsIn1->dim_y;
87  uint32_t strideIn1 = bufParamsIn1->stride_y;
88 
89  // obtain output buffer parameters
90  uint32_t dTypeOut = bufParamsOut->data_type;
91  uint32_t widthOut = bufParamsOut->dim_x;
92  uint32_t heightOut = bufParamsOut->dim_y;
93  uint32_t strideOut = bufParamsOut->stride_y;
94 
95  uint32_t strideIn0Elements = strideIn0 / VXLIB_sizeof(dTypeIn0);
96  uint32_t strideIn1Elements = strideIn1 / VXLIB_sizeof(dTypeIn1);
97  uint32_t strideOutElements = strideOut / VXLIB_sizeof(dTypeOut);
98 
99  if ((widthIn0 != widthIn1) || (widthIn0 != widthOut) || (heightIn0 != heightIn1) ||
100  (heightIn0 != heightOut)) {
101  status = VXLIB_ERR_INVALID_DIMENSION;
102  }
103  else if ((strideIn0Elements < widthIn0) || (strideIn1Elements < widthIn1) || (strideOutElements < widthOut)) {
104  status = VXLIB_ERR_NOT_EQUAL_WIDTH_STRIDE;
105  }
106  else if (!(VXLIB_SUBTRACT_I8U_I8U_O8U) && !(VXLIB_SUBTRACT_I8U_I8U_O16S) && !(VXLIB_SUBTRACT_I8U_I16S_O16S) &&
107  !(VXLIB_SUBTRACT_I16S_I16S_O16S)) {
108  status = VXLIB_ERR_INVALID_TYPE;
109  }
110  else {
111  status = VXLIB_SUCCESS;
112  }
113 
114  }
115 
116  return status;
117 }

152 // this method is the user-level initialization function for the VXLIB_subtract kernel
153 VXLIB_STATUS VXLIB_subtract_init(VXLIB_kernelHandle handle,
154  VXLIB_bufParams2D_t * bufParamsIn0,
155  VXLIB_bufParams2D_t * bufParamsIn1,
156  VXLIB_bufParams2D_t * bufParamsOut,
157  const VXLIB_subtract_InitArgs *pKerInitArgs)
158 {
159  VXLIB_STATUS status = VXLIB_SUCCESS;
160  VXLIB_subtract_PrivArgs *pKerPrivArgs = (VXLIB_subtract_PrivArgs *) handle;
161 
162  // copy pKerinitArgs into pKerPrivargs
163  pKerPrivArgs->pKerInitArgs = *pKerInitArgs;
164 
165  // set width and height of image
166  pKerPrivArgs->width = bufParamsIn0->dim_x;
167  pKerPrivArgs->height = bufParamsIn0->dim_y;
168 
169  // compute stride in elements as SE/SA params are set to work with given element type
170  pKerPrivArgs->strideIn0Elements = bufParamsIn0->stride_y / VXLIB_sizeof(bufParamsIn0->data_type);
171  pKerPrivArgs->strideIn1Elements = bufParamsIn1->stride_y / VXLIB_sizeof(bufParamsIn1->data_type);
172  pKerPrivArgs->strideOutElements = bufParamsOut->stride_y / VXLIB_sizeof(bufParamsOut->data_type);
173 
174 #if VXLIB_DEBUGPRINT
175  printf("VXLIB_DEBUGPRINT Enter VXLIB_subtract_init\n");
176 #endif
177 
178  // obtain buffer datatypes
179  uint32_t dTypeIn0 = bufParamsIn0->data_type;
180  uint32_t dTypeIn1 = bufParamsIn1->data_type;
181  uint32_t dTypeOut = bufParamsOut->data_type;
182 
183  // determine natural C vs optimized function call
184  if (pKerInitArgs->funcStyle == VXLIB_FUNCTION_NATC) {
185 
186  // set function pointer for natural C function with appropriate template parameters based on datatypes
187  if (VXLIB_SUBTRACT_I8U_I8U_O8U) {
188  pKerPrivArgs->execute = VXLIB_subtract_exec_cn<VXLIB_SUBTRACT_TYPENAME_I8U_I8U_O8U>;
189  }
190  else if (VXLIB_SUBTRACT_I8U_I8U_O16S) {
191  pKerPrivArgs->execute = VXLIB_subtract_exec_cn<VXLIB_SUBTRACT_TYPENAME_I8U_I8U_O16S>;
192  }
193  else if (VXLIB_SUBTRACT_I8U_I16S_O16S) {
194  pKerPrivArgs->execute = VXLIB_subtract_exec_cn<VXLIB_SUBTRACT_TYPENAME_I8U_I16S_O16S>;
195  }
196  else if (VXLIB_SUBTRACT_I16S_I16S_O16S) {
197  pKerPrivArgs->execute = VXLIB_subtract_exec_cn<VXLIB_SUBTRACT_TYPENAME_I16S_I16S_O16S>;
198  }
199  else {
200  status = VXLIB_ERR_INVALID_TYPE;
201  }
202  }
203  else { // Optimized function
204 
205  // set function pointer for natural C function with appropriate template parameters based on datatypes
206  if (VXLIB_SUBTRACT_I8U_I8U_O8U) {
207  pKerPrivArgs->execute = VXLIB_subtract_exec_ci<VXLIB_SUBTRACT_TYPENAME_I8U_I8U_O8U>;
208  status = VXLIB_subtract_init_ci<VXLIB_SUBTRACT_DTYPE_I8U_I8U_O8U>(handle, bufParamsIn0, bufParamsIn1,
209  bufParamsOut, pKerInitArgs);
210  }
211  else if (VXLIB_SUBTRACT_I8U_I8U_O16S) {
212  pKerPrivArgs->execute = VXLIB_subtract_exec_ci<VXLIB_SUBTRACT_TYPENAME_I8U_I8U_O16S>;
213  status = VXLIB_subtract_init_ci<VXLIB_SUBTRACT_DTYPE_I8U_I8U_O16S>(handle, bufParamsIn0, bufParamsIn1,
214  bufParamsOut, pKerInitArgs);
215  }
216  else if (VXLIB_SUBTRACT_I8U_I16S_O16S) {
217  pKerPrivArgs->execute = VXLIB_subtract_exec_ci<VXLIB_SUBTRACT_TYPENAME_I8U_I16S_O16S>;
218  status = VXLIB_subtract_init_ci<VXLIB_SUBTRACT_DTYPE_I8U_I16S_O16S>(handle, bufParamsIn0, bufParamsIn1,
219  bufParamsOut, pKerInitArgs);
220  }
221  else if (VXLIB_SUBTRACT_I16S_I16S_O16S) {
222  pKerPrivArgs->execute = VXLIB_subtract_exec_ci<VXLIB_SUBTRACT_TYPENAME_I16S_I16S_O16S>;
223  status = VXLIB_subtract_init_ci<VXLIB_SUBTRACT_DTYPE_I16S_I16S_O16S>(handle, bufParamsIn0, bufParamsIn1,
224  bufParamsOut, pKerInitArgs);
225  }
226  else {
227  status = VXLIB_ERR_INVALID_TYPE;
228  }
229  }
230 
231  return status;
232 }
