56 // this method computes kxk gaussianFilter in via natural C code
57 template <typename dTypeIn, typename dTypeOut>
58 VXLIB_STATUS VXLIB_gaussian_exec_cn(VXLIB_kernelHandle handle, void *restrict pIn, void *restrict pOut)
59 {
60 
61  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
62 
63 #if VXLIB_DEBUGPRINT
64  printf("Enter VXLIB_gaussian_exec_cn\n");
65 #endif
66 
67  // typecast handle (void) to struct pointer type associated to kernel
68  VXLIB_gaussian_PrivArgs *pKerPrivArgs = (VXLIB_gaussian_PrivArgs *) handle;
69 
70  // obtain image parameters and overflow policy
71  size_t width = pKerPrivArgs->width;
72  size_t height = pKerPrivArgs->height;
73  size_t strideInElements = pKerPrivArgs->strideInElements;
74  size_t strideOutElements = pKerPrivArgs->strideOutElements;
75 
76  size_t padLeft = pKerPrivArgs->pKerInitArgs.padLeft;
77  size_t padRight = pKerPrivArgs->pKerInitArgs.padRight;
78  size_t padTop = pKerPrivArgs->pKerInitArgs.padTop;
79  size_t padBottom = pKerPrivArgs->pKerInitArgs.padBottom;
80  bool isNotPadded = (padLeft == 0) && (padRight == 0) && (padTop == 0) && (padBottom == 0);
81 
82  size_t filterDim = pKerPrivArgs->pKerInitArgs.filterSize;
83  uint8_t shift = pKerPrivArgs->pKerInitArgs.shift;
84 
85  // create local pointers
86  dTypeIn *restrict pInLocal = (dTypeIn *) pIn;
87  dTypeOut *restrict pOutLocal = (dTypeOut *) pOut;
88 
89 #if VXLIB_DEBUGPRINT
90  printf("In VXLIB_gaussian_exec_cn, width: %d, height: %d\n", width, height);
91 #endif
92 
93  if (filterDim == VXLIB_GAUSSIAN_FILTER_3x3) {
94 
95  // non-padded natc implementation
96  if (isNotPadded) {
97 
98  int32_t x, y, i, j;
99  int32_t sum;
100  int32_t outWidth = (width - filterDim + 1);
101  int32_t outHeight = (height - filterDim + 1);
102  int8_t Q_BIT = 15;
103 
104  // Creating gaussian filter 3x3 matrix
105  // gaussianFilter3x3 = [[1, 2, 1], [2, 4, 2], [1, 2, 1]] * 1/16
106  // Q_FACTOR = 1 << Q_BIT
107  // gaussianFilter3x3 = gaussianFilter3x3 * Q_FACTOR
108  int16_t gaussianFilter3x3[3][3] = {{2048, 4096, 2048}, {4096, 8192, 4096}, {2048, 4096, 2048}};
109 
110  for (y = 0; y < outHeight; y++) {
111  for (x = 0; x < outWidth; x++) {
112 
113  sum = 0;
114 
115  for (j = 0; j < filterDim; j++) {
116  for (i = 0; i < filterDim; i++) {
117  sum += (pInLocal[(y + j) * strideInElements + (x + i)]) * gaussianFilter3x3[j][i];
118  }
119  }
120 
121  pOutLocal[y * strideOutElements + x] = (dTypeOut)((uint32_t)sum >> (uint32_t)Q_BIT);
122  }
123  }
124  }
125  // padded natc implementation
126  else {
127  status = VXLIB_ERR_NOT_IMPLEMENTED;
128  }
129  }
130  // 5x5 implementation
131  else if (filterDim == VXLIB_GAUSSIAN_FILTER_5x5) {
132  // non-padded natc implementation
133  if (isNotPadded) {
134 
135  int32_t x, y, i, j;
136  int32_t sum;
137  int32_t outWidth = (width < filterDim) ? 1 : (width - filterDim + 1);
138  int32_t outHeight = (height < filterDim) ? 1 : (height - filterDim + 1);
139 
140  int16_t gaussianFilter5x5[5][5] = {
141  {1, 4, 6, 4, 1}, {4, 16, 24, 16, 4}, {6, 24, 36, 24, 6}, {4, 16, 24, 16, 4}, {1, 4, 6, 4, 1}};
142 
143  for (y = 0; y < outHeight; y++) {
144  for (x = 0; x < outWidth; x++) {
145 
146  sum = 0;
147 
148  for (j = 0; j < filterDim; j++) {
149  for (i = 0; i < filterDim; i++) {
150 
151  if ((y + j < height) && (x + i < width)) {
152  sum += (pInLocal[(y + j) * strideInElements + (x + i)]) * gaussianFilter5x5[j][i];
153  }
154  }
155  }
156 
157  pOutLocal[y * strideOutElements + x] = (dTypeOut)((uint32_t)sum >> (uint32_t)shift);
158  }
159  }
160  }
161  // padded natc implementation
162  else {
163  status = VXLIB_ERR_NOT_IMPLEMENTED;
164  }
165  }
166  // other filter dimensions
167  else {
168  status = VXLIB_ERR_NOT_IMPLEMENTED;
169  }
170 
171  return (status);
172 }
