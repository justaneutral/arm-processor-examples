57 // this method computes add in via natural C code
58 template <typename dTypeIn, typename dTypeOut>
59 VXLIB_STATUS VXLIB_meanStdDev_exec_cn(VXLIB_kernelHandle handle,
60  void *restrict pIn,
61  void *restrict pOut0,
62  void *restrict pOut1,
63  void *restrict pPixelsProcessed,
64  void *restrict pCurrentSum,
65  void *restrict pCurrentSqSum)
66 {
67 
68  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
69 
70  // typecast handle (void) to struct pointer type associated to kernel
71  VXLIB_meanStdDev_PrivArgs *pKerPrivArgs = (VXLIB_meanStdDev_PrivArgs *) handle;
72 
73  // obtain image parameters and overflow policy
74  size_t width = pKerPrivArgs->width;
75  size_t height = pKerPrivArgs->height;
76  size_t strideInElements = pKerPrivArgs->strideInElements;
77 
78 #if VXLIB_DEBUGPRINT
79  printf("Enter VXLIB_meanStdDev_exec_cn\n");
80 #endif
81 
82  typedef typename std::conditional<std::is_same<dTypeIn, uint8_t>::value, uint32_t, uint64_t>::type dTypeAcc;
83 
84  // create local pointers
85  dTypeIn *restrict pInLocal = (dTypeIn *) pIn;
86  dTypeOut *restrict pOut0Local = (dTypeOut *) pOut0;
87  dTypeOut *restrict pOut1Local = (dTypeOut *) pOut1;
88  uint32_t *restrict pPixelsProcessedLocal = (uint32_t *) pPixelsProcessed;
89  dTypeAcc *restrict pCurrentSumLocal = (dTypeAcc *) pCurrentSum;
90  dTypeAcc *restrict pCurrentSqSumLocal = (dTypeAcc *) pCurrentSqSum;
91 
92  // variables to hold temporary values
93  size_t N = width * height;
94 
95 #if VXLIB_DEBUGPRINT
96  printf("In VXLIB_meanStdDev_exec_cn, width: %d, height: %d\n", width, height);
97 #endif
98 
99  for (uint32_t i = 0; i < height; i++) {
100  for (uint32_t j = 0; j < width; j++) {
101  *pCurrentSumLocal += (dTypeAcc) pInLocal[i * strideInElements + j];
102  *pCurrentSqSumLocal +=
103  ((dTypeAcc) pInLocal[i * strideInElements + j] * (dTypeAcc) pInLocal[i * strideInElements + j]);
104  }
105  }
106 
107  // calculate mean
108  *pPixelsProcessedLocal += N;
109 
110  *pOut0Local = (dTypeOut) (*pCurrentSumLocal) / (*pPixelsProcessedLocal);
111 
112  // calculate sqrt
113  dTypeOut variance = ((dTypeOut) (*pCurrentSqSumLocal) -
114  ((dTypeOut) (*pCurrentSumLocal) * (*pCurrentSumLocal) / (*pPixelsProcessedLocal))) /
115  (*pPixelsProcessedLocal);
116 
117  *pOut1Local = (dTypeOut) sqrt(variance);
118 
119  return (status);
120 }
