54 // this method computes addWeight in via natural C code
55 template <typename dTypeIn0, typename dTypeIn1, typename dTypeOut>
56 VXLIB_STATUS VXLIB_addWeight_exec_cn(VXLIB_kernelHandle handle,
57  void *restrict pIn0,
58  void *restrict pIn1,
59  void *restrict pOut,
60  float alpha)
61 {
62 
63  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
64 
65  // typecast handle (void) to struct pointer type associated to kernel
66  VXLIB_addWeight_PrivArgs *pKerPrivArgs = (VXLIB_addWeight_PrivArgs *) handle;
67 
68  // obtain image parameters
69  size_t width = pKerPrivArgs->width;
70  size_t height = pKerPrivArgs->height;
71  size_t strideIn0Elements = pKerPrivArgs->strideIn0Elements;
72  size_t strideIn1Elements = pKerPrivArgs->strideIn1Elements;
73  size_t strideOutElements = pKerPrivArgs->strideOutElements;
74 
75 #if VXLIB_DEBUGPRINT
76  printf("Enter VXLIB_addWeight_exec_cn\n");
77 #endif
78 
79  // create local pointers
80  dTypeIn0 *restrict pInLocal0 = (dTypeIn0 *) pIn0;
81  dTypeIn1 *restrict pInLocal1 = (dTypeIn1 *) pIn1;
82  dTypeOut *restrict pOutLocal = (dTypeOut *) pOut;
83 
84  // variables to hold temporary values
85  dTypeIn0 a;
86  dTypeIn1 b;
87  dTypeOut out;
88 
89  // float containers for computation values
90  float a_, b_, out_;
91 
92 #if VXLIB_DEBUGPRINT
93  printf("In VXLIB_addWeight_exec_cn, width: %d, height: %d\n", width, height);
94 #endif
95 
96  // perform addWeight
97  for (uint32_t i = 0; i < height; i++) {
98  for (uint32_t j = 0; j < width; j++) {
99  a = pInLocal0[i * strideIn0Elements + j];
100  b = pInLocal1[i * strideIn1Elements + j];
101 
102  // move to float containers for computation
103  a_ = (float) (int32_t) a;
104  b_ = (float) (int32_t) b;
105  out_ = alpha * a_ + (1 - alpha) * b_;
106 
107  // move back to uint8_t container
108  out = (dTypeOut) (int8_t) out_;
109 
110  pOutLocal[i * strideOutElements + j] = out;
111  }
112  }
113 
114  return (status);
115 }
