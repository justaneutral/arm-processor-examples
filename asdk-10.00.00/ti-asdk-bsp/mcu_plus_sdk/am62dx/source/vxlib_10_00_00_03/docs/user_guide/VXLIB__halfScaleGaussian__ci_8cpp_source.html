82 // this method initializes the kernel-specific parameters
83 // mainly, the streaming engine and streaming address generators
84 template <uint32_t dTypeIn, uint32_t dTypeOut>
85 VXLIB_STATUS VXLIB_halfScaleGaussian_init_ci(VXLIB_kernelHandle handle,
86  const VXLIB_bufParams2D_t *bufParamsIn,
87  const VXLIB_bufParams2D_t *bufParamsOut,
88  const VXLIB_halfScaleGaussian_InitArgs *pKerInitArgs)
89 {
90  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
91 
92  // structs to hold SE and SA parameters
93  __SE_TEMPLATE_v1 se0Params = __gen_SE_TEMPLATE_v1();
94  __SE_TEMPLATE_v1 se1Params = __gen_SE_TEMPLATE_v1();
95  __SA_TEMPLATE_v1 sa0Params = __gen_SA_TEMPLATE_v1();
96 
97  // fetch + SE promote, store all half SIMD width
98  __SE_ELETYPE SE_ELETYPE = c7x::se_eletype<c7x::uchar_vec>::value;
99  __SE_VECLEN SE_VECLEN = c7x::se_veclen<c7x::uchar_hvec>::value;
100  __SA_VECLEN SA_VECLEN = c7x::sa_veclen<c7x::uchar_hvec>::value;
101 
102  // typecast handle (void) to struct pointer type associated to kernel
103  VXLIB_halfScaleGaussian_PrivArgs *pKerPrivArgs = (VXLIB_halfScaleGaussian_PrivArgs *) handle;
104 
105  // number of elements fetched from input each SE get
106  size_t elemCount = c7x::element_count_of<c7x::char_hvec>::value;
107 
108  // get init args
109  size_t width = bufParamsIn->dim_x;
110  size_t height = bufParamsIn->dim_y;
111  size_t stride = bufParamsIn->stride_y;
112 
113  size_t filterSize = pKerInitArgs->filterSize;
114  size_t padLeft = pKerInitArgs->padLeft;
115  size_t padRight = pKerInitArgs->padRight;
116  size_t padTop = pKerInitArgs->padTop;
117  size_t padBottom = pKerInitArgs->padBottom;
118 
119  // determine non-padded vs padded implementation
120  bool isNotPadded = (padLeft == 0) && (padRight == 0) && (padTop == 0) && (padBottom == 0);
121 
122  uint8_t *pBlock = pKerPrivArgs->bufPblock;
123  pKerPrivArgs->numBlocks = width * VXLIB_ceilingDiv(width, elemCount);
124 
125  // setup se/sa for k x k implementation
126 
127  if (filterSize == VXLIB_HALF_SCALE_GAUSSIAN_FILTER_5x5) {
128  int16_t gaussianFilter5x5[15] = {1, 4, 6, 4, 1, 4, 16, 24, 16, 4, 6, 24, 36, 24, 6};
129 
130  // call 5x5 gaussian init non-padded implementation
131  if (isNotPadded) {
132  size_t elemCountOut = c7x::element_count_of<c7x::uchar_qvec>::value;
133 
134  // set SE0 Params
135  se0Params.DIMFMT = __SE_DIMFMT_4D;
136  se0Params.ELETYPE = SE_ELETYPE;
137  se0Params.VECLEN = SE_VECLEN;
138  se0Params.PROMOTE = __SE_PROMOTE_2X_ZEROEXT;
139 
140  // 0: number of elements to be computed per SE fetch
141  se0Params.ICNT0 = elemCount;
142 
143  // 1: number of accumulation rows in single convolution window
144  se0Params.DIM1 = stride;
145  se0Params.ICNT1 = filterSize;
146 
147  // 2: vertical iterations
148  se0Params.DIM2 = stride * 2;
149  se0Params.ICNT2 = (height - filterSize + 1) / 2;
150 
151  // 3: horizontal iterations
152  se0Params.DIM3 = elemCount / 2; // vfir8 is half-vector fetch, quarter-vec adv
153  se0Params.ICNT3 = VXLIB_ceilingDiv(width, (elemCount / 2));
154 
155  // input column dimension DECDIM
156  se0Params.DECDIM1 = __SE_DECDIM_DIM3;
157  se0Params.DECDIM1_WIDTH = width;
158  se0Params.DECDIM1SD = __SE_DECDIMSD_DIM0;
159 
160  // set SA0 Params
161  sa0Params.DIMFMT = __SA_DIMFMT_3D;
162  sa0Params.VECLEN = SA_VECLEN;
163 
164  // 0: number of elements to store per SE fetch
165  sa0Params.ICNT0 = elemCountOut / 2;
166 
167  // 1: vertical iterations (1 store per convolution window)
168  sa0Params.DIM1 = pKerPrivArgs->strideOutElements; // SA DIM jumps are given in num elements
169  sa0Params.ICNT1 = bufParamsOut->dim_y;
170 
171  // 2: horizontal iterations
172  sa0Params.DIM2 = elemCountOut / 2;
173  sa0Params.ICNT2 = VXLIB_ceilingDiv(bufParamsOut->dim_x, (elemCountOut / 2));
174 
175  // output column dimension DECDIM
176  sa0Params.DECDIM1 = __SA_DECDIM_DIM2;
177  sa0Params.DECDIM1_WIDTH = bufParamsOut->dim_x;
178  sa0Params.DECDIM1SD = __SA_DECDIMSD_DIM0;
179 
180  // store gaussian filter 5x5 coefficients
181  // 1st row of coefficients split into low and high for VFIR8
182 
183  /* 0 64b */
184  /* +-----------------------+ */
185  /* | 1 | 4 | 6 | 4 | */
186  /* +-----------------------+ */
187 
188  /* 0 64b */
189  /* +-----------------------+ */
190  /* | 1 | 0 | 0 | 0 | */
191  /* +-----------------------+ */
192 
193  // Store first row (low) of coefficient matrix
194  uint64_t filter = gaussianFilter5x5[3];
195  filter = (filter << 16LLU) | (uint64_t)gaussianFilter5x5[2];
196  filter = (filter << 16LLU) | (uint64_t)gaussianFilter5x5[1];
197  filter = (filter << 16LLU) | (uint64_t)gaussianFilter5x5[0];
198 
199  size_t filter_offset = SA0_PARAM_OFFSET + VXLIB_GAUSSIANFILTER3x3_COEFF_SIZE;
200 
201  *(uint64_t *) ((uint8_t *) pBlock + filter_offset) = filter;
202 
203  // high
204  filter = gaussianFilter5x5[4];
205  filter_offset = filter_offset + VXLIB_GAUSSIANFILTER3x3_COEFF_SIZE;
206 
207  *(uint64_t *) ((uint8_t *) pBlock + filter_offset) = filter;
208 
209  // 2nd row of coefficients split into low and high for VFIR8
210 
211  /* 0 64b */
212  /* +--------------------------+ */
213  /* | 4 | 16 | 24 | 16 | */
214  /* +--------------------------+ */
215 
216  /* 0 64b */
217  /* +-----------------------+ */
218  /* | 4 | 0 | 0 | 0 | */
219  /* +-----------------------+ */
220 
221  // Store second row (low) of the coefficient matrix
222  filter = gaussianFilter5x5[8];
223  filter = (filter << 16LLU) | (uint64_t)gaussianFilter5x5[7];
224  filter = (filter << 16LLU) | (uint64_t)gaussianFilter5x5[6];
225  filter = (filter << 16LLU) | (uint64_t)gaussianFilter5x5[5];
226 
227  filter_offset = filter_offset + VXLIB_GAUSSIANFILTER3x3_COEFF_SIZE;
228 
229  *(uint64_t *) ((uint8_t *) pBlock + filter_offset) = filter;
230 
231  // high
232  filter = gaussianFilter5x5[9];
233  filter_offset = filter_offset + VXLIB_GAUSSIANFILTER3x3_COEFF_SIZE;
234 
235  *(uint64_t *) ((uint8_t *) pBlock + filter_offset) = filter;
236 
237  // 3rd row of coefficients split into low and high for VFIR8
238 
239  /* 0 64b */
240  /* +--------------------------+ */
241  /* | 6 | 24 | 36 | 24 | */
242  /* +--------------------------+ */
243 
244  /* 0 64b */
245  /* +-----------------------+ */
246  /* | 6 | 0 | 0 | 0 | */
247  /* +-----------------------+ */
248 
249  // Store third row (low) of the coefficient matrix
250  filter = gaussianFilter5x5[13];
251  filter = (filter << 16LLU) | (uint64_t)gaussianFilter5x5[12];
252  filter = (filter << 16LLU) | (uint64_t)gaussianFilter5x5[11];
253  filter = (filter << 16LLU) | (uint64_t)gaussianFilter5x5[10];
254 
255  filter_offset = filter_offset + VXLIB_GAUSSIANFILTER3x3_COEFF_SIZE;
256 
257  *(uint64_t *) ((uint8_t *) pBlock + filter_offset) = filter;
258 
259  // high
260  filter = gaussianFilter5x5[14];
261  filter_offset = filter_offset + VXLIB_GAUSSIANFILTER3x3_COEFF_SIZE;
262 
263  *(uint64_t *) ((uint8_t *) pBlock + filter_offset) = filter;
264 
265  // 4th and 5th row not stored as same as 1 and 2 of coefficient matrix
266  }
267 
268  // return error for padded version; not implemented
269  else {
270 
271  status = VXLIB_ERR_NOT_IMPLEMENTED;
272  }
273  }
274 
275  else {
276  // return error for any other filter size (only 3x3 supported for now)
277  status = VXLIB_ERR_NOT_IMPLEMENTED;
278  }
279 
280  // store SE and SA params
281  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_PARAM_OFFSET) = se0Params;
282  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE1_PARAM_OFFSET) = se1Params;
283  *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_PARAM_OFFSET) = sa0Params;
284 
285  return status;
286 }

306 template <typename dTypeIn, typename dTypeOut>
307 void VXLIB_halfScaleGaussian_5x5_exec_ci(VXLIB_kernelHandle handle, void *restrict pIn, void *restrict pOut)
308 {
309  // get kernel handle
310  VXLIB_halfScaleGaussian_PrivArgs *pKerPrivArgs = (VXLIB_halfScaleGaussian_PrivArgs *) handle;
311 
312  // SE/SA params template
313  __SE_TEMPLATE_v1 se0Params;
314  __SA_TEMPLATE_v1 sa0Params;
315 
316  // create local pointers
317  dTypeIn *restrict pInLocal = (dTypeIn *) pIn;
318  dTypeOut *restrict pOutLocal = (dTypeOut *) pOut;
319 
320  uint8_t *pBlock = pKerPrivArgs->bufPblock;
321  size_t filterOffset = SA0_PARAM_OFFSET + VXLIB_GAUSSIANFILTER3x3_COEFF_SIZE;
322 
323  // get SE/SA params
324  se0Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_PARAM_OFFSET);
325  sa0Params = *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_PARAM_OFFSET);
326 
327  // get coeff from memory
328  uint64_t scalarCoeff = *(uint64_t *) ((uint8_t *) pBlock + filterOffset);
329 
330  // duplicate scalar coeff to full vector length
331  c7x::ulong_vec ulongvCoeff = __vload_dup(&scalarCoeff);
332 
333  // reinterpret to 16b element coeffs
334  c7x::short_vec vCoeff1_lo = c7x::reinterpret<c7x::short_vec>(ulongvCoeff);
335 
336  // Repeat for remaining coefficient vectors
337  scalarCoeff = *(uint64_t *) ((uint8_t *) pBlock + (filterOffset + VXLIB_GAUSSIANFILTER3x3_COEFF_SIZE));
338  ulongvCoeff = __vload_dup(&scalarCoeff);
339  c7x::short_vec vCoeff1_hi = c7x::reinterpret<c7x::short_vec>(ulongvCoeff);
340 
341  scalarCoeff = *(uint64_t *) ((uint8_t *) pBlock + (filterOffset + 2 * VXLIB_GAUSSIANFILTER3x3_COEFF_SIZE));
342  ulongvCoeff = __vload_dup(&scalarCoeff);
343  c7x::short_vec vCoeff2_lo = c7x::reinterpret<c7x::short_vec>(ulongvCoeff);
344 
345  scalarCoeff = *(uint64_t *) ((uint8_t *) pBlock + (filterOffset + 3 * VXLIB_GAUSSIANFILTER3x3_COEFF_SIZE));
346  ulongvCoeff = __vload_dup(&scalarCoeff);
347  c7x::short_vec vCoeff2_hi = c7x::reinterpret<c7x::short_vec>(ulongvCoeff);
348 
349  scalarCoeff = *(uint64_t *) ((uint8_t *) pBlock + (filterOffset + 4 * VXLIB_GAUSSIANFILTER3x3_COEFF_SIZE));
350  ulongvCoeff = __vload_dup(&scalarCoeff);
351  c7x::short_vec vCoeff3_lo = c7x::reinterpret<c7x::short_vec>(ulongvCoeff);
352 
353  scalarCoeff = *(uint64_t *) ((uint8_t *) pBlock + (filterOffset + 5 * VXLIB_GAUSSIANFILTER3x3_COEFF_SIZE));
354  ulongvCoeff = __vload_dup(&scalarCoeff);
355  c7x::short_vec vCoeff3_hi = c7x::reinterpret<c7x::short_vec>(ulongvCoeff);
356 
357  // open streams
358  __SE0_OPEN(pInLocal, se0Params);
359  __SA0_OPEN(sa0Params);
360 
361  size_t filterSize = pKerPrivArgs->pKerInitArgs.filterSize;
362 
363  size_t elemCountLoop = c7x::element_count_of<c7x::uchar_hvec>::value;
364  size_t wBlocks = VXLIB_ceilingDiv(pKerPrivArgs->widthIn, (elemCountLoop / 2));
365  size_t hBlocks = (pKerPrivArgs->heightIn - filterSize + 1) / 2;
366 
367  c7x::int_vec vResA, vResB, vResC, vResD, vResE, vRes1ABCDE;
368 
369  uint8_t shift = 8;
370  c7x::int_vec shiftVec = (c7x::int_vec) shift;
371 
372  // TODO: nested loop coalescing (NLC compiler pragma)
373  for (int32_t wCounter = 0; wCounter < wBlocks; wCounter++) {
374 
375  for (int32_t hCounter = 0; hCounter < hBlocks; hCounter++) {
376  vResA = __vfir8hw_wvv(vCoeff1_lo, vCoeff1_hi, __SE_REG_0_ADV);
377  vResB = __vfir8hw_wvv(vCoeff2_lo, vCoeff2_hi, __SE_REG_0_ADV);
378  vResC = __vfir8hw_wvv(vCoeff3_lo, vCoeff3_hi, __SE_REG_0_ADV);
379  vResD = __vfir8hw_wvv(vCoeff2_lo, vCoeff2_hi, __SE_REG_0_ADV);
380  vResE = __vfir8hw_wvv(vCoeff1_lo, vCoeff1_hi, __SE_REG_0_ADV);
381 
382  vRes1ABCDE = vResA + vResB + vResC + vResD + vResE;
383  vRes1ABCDE = __shift_right(vRes1ABCDE, shiftVec);
384 
385  // saturate from 32b signed to 8b unsigned
386  c7x::uint_vec vURes1ABCDE = c7x::convert<c7x::uint_vec>(vRes1ABCDE);
387 
388  c7x::uint_vec zero = (c7x::uint_vec) 0;
389  // permute
390  c7x::uint_vec vUResABCDE = c7x::as_uint_vec(__permute_even_even_int(
391  VXLIB_vperm_data, c7x::as_uchar_vec(zero), c7x::as_uchar_vec(vURes1ABCDE)));
392 
393  __vpred tmp = c7x::strm_agen<0, c7x::uint_vec>::get_vpred();
394  c7x::uchar_qvec *addr = c7x::strm_agen<0, c7x::uchar_qvec>::get_adv(pOutLocal);
395  __vstore_pred_pack_byte(tmp, addr, vUResABCDE);
396  }
397  }
398 
399  __SE0_CLOSE();
400  __SA0_CLOSE();
401 }
