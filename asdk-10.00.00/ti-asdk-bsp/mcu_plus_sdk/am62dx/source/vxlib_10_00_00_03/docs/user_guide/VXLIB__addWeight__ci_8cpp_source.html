62 // this method initializes the kernel-specific parameters
63 // mainly, the streaming engine and streaming address generators
64 template <uint32_t dTypeIn0, uint32_t dTypeIn1, uint32_t dTypeOut>
65 VXLIB_STATUS VXLIB_addWeight_init_ci(VXLIB_kernelHandle handle,
66  const VXLIB_bufParams2D_t *bufParamsIn0,
67  const VXLIB_bufParams2D_t *bufParamsIn1,
68  const VXLIB_bufParams2D_t *bufParamsOut,
69  const VXLIB_addWeight_InitArgs *pKerInitArgs)
70 {
71  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
72 
73  // structs to hold SE and SA parameters
74  __SE_TEMPLATE_v1 se0Params = __gen_SE_TEMPLATE_v1();
75  __SE_TEMPLATE_v1 se1Params = __gen_SE_TEMPLATE_v1();
76  __SA_TEMPLATE_v1 sa0Params = __gen_SA_TEMPLATE_v1();
77 
78  // typecast handle (void) to struct pointer type associated to kernel
79  VXLIB_addWeight_PrivArgs *pKerPrivArgs = (VXLIB_addWeight_PrivArgs *) handle;
80 
81  // obtain image size and compute number of blocks to process
82  size_t width_ = pKerPrivArgs->width;
83  size_t height_ = pKerPrivArgs->height;
84  size_t elemCount_ = VXLIB_max_simd<VXLIB_UINT32>::value;
85  size_t wBlocks_ = (width_ + (elemCount_ - 1)) / (elemCount_);
86  size_t numBlocks_ = height_ * wBlocks_;
87  pKerPrivArgs->numBlocks = numBlocks_;
88 
89  uint8_t *pBlock = pKerPrivArgs->bufPblock; // address to retrieve to store SE/SA params
90 
91  // TODO: set custom params to promot 4x from se
92 
93  // SE/SA parameter containers
94  __SE_ELETYPE SE_ELETYPE;
95  __SE_VECLEN SE_VECLEN;
96  __SA_VECLEN SA_VECLEN;
97 
98  // assign SE/SA parameters
99  SE_ELETYPE = c7x::se_eletype<c7x::uchar_vec>::value;
100  SE_VECLEN = c7x::se_veclen<c7x::uint_vec>::value;
101  SA_VECLEN = c7x::sa_veclen<c7x::uint_vec>::value;
102 
103  size_t width = bufParamsIn0->dim_x;
104  size_t height = bufParamsIn0->dim_y;
105 
106  size_t strideIn0Elems = pKerPrivArgs->strideIn0Elements;
107  size_t strideIn1Elems = pKerPrivArgs->strideIn1Elements;
108  size_t strideOutElems = pKerPrivArgs->strideOutElements;
109 
110  bool isLinear = false;
111 
112  if ((width == strideIn0Elems) && (strideIn0Elems == strideIn1Elems) && (strideIn0Elems == strideOutElems)) {
113  isLinear = true;
114  }
115 
116  // data cannot be accessed in 1D pattern
117  if (!isLinear) {
118 
119  /***************************************/
120  /* Create SE params for reading input0 */
121  /***************************************/
122 
123  se0Params.ELETYPE = SE_ELETYPE;
124  se0Params.VECLEN = SE_VECLEN;
125  se0Params.DIMFMT = __SE_DIMFMT_2D;
126  se0Params.PROMOTE = __SE_PROMOTE_4X_ZEROEXT;
127 
128  se0Params.ICNT0 = width;
129  se0Params.ICNT1 = height;
130  se0Params.DIM1 = strideIn0Elems;
131 
132  /***************************************/
133  /* Create SE params for reading input1 */
134  /***************************************/
135 
136  se1Params.ELETYPE = SE_ELETYPE;
137  se1Params.VECLEN = SE_VECLEN;
138  se1Params.DIMFMT = __SE_DIMFMT_2D;
139  se1Params.PROMOTE = __SE_PROMOTE_4X_ZEROEXT;
140 
141  se1Params.ICNT0 = width;
142  se1Params.ICNT1 = height;
143  se1Params.DIM1 = strideIn1Elems;
144 
145  /****************************************/
146  /* Create SA0 params for writing output */
147  /****************************************/
148 
149  sa0Params.ICNT0 = width;
150  sa0Params.ICNT1 = height;
151  sa0Params.DIM1 = strideOutElems;
152  sa0Params.VECLEN = SA_VECLEN;
153  sa0Params.DIMFMT = __SA_DIMFMT_2D;
154  }
155 
156  // data can be accessed in 1D pattern
157  else {
158 
159  /***************************************/
160  /* Create SE params for reading input0 */
161  /***************************************/
162 
163  se0Params.ELETYPE = SE_ELETYPE;
164  se0Params.VECLEN = SE_VECLEN;
165  se0Params.DIMFMT = __SE_DIMFMT_1D;
166  se0Params.PROMOTE = __SE_PROMOTE_4X_ZEROEXT;
167 
168  se0Params.ICNT0 = width * height;
169 
170  /***************************************/
171  /* Create SE params for reading input1 */
172  /***************************************/
173 
174  se1Params.ELETYPE = SE_ELETYPE;
175  se1Params.VECLEN = SE_VECLEN;
176  se1Params.DIMFMT = __SE_DIMFMT_1D;
177  se1Params.PROMOTE = __SE_PROMOTE_4X_ZEROEXT;
178 
179  se1Params.ICNT0 = width * height;
180 
181  /****************************************/
182  /* Create SA0 params for writing output */
183  /****************************************/
184 
185  sa0Params.VECLEN = SA_VECLEN;
186  sa0Params.DIMFMT = __SA_DIMFMT_1D;
187 
188  sa0Params.ICNT0 = width * height;
189  }
190 
191  /**************************/
192  /* Store SE and SA params */
193  /**************************/
194 
195  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_PARAM_OFFSET) = se0Params;
196  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE1_PARAM_OFFSET) = se1Params;
197  *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_PARAM_OFFSET) = sa0Params;
198 
199  return status;
200 }

221 // this method performs pixel-wise weighted addition of two input images
222 template <typename dTypeIn0, typename dTypeIn1, typename dTypeOut>
223 VXLIB_STATUS VXLIB_addWeight_exec_ci(VXLIB_kernelHandle handle,
224  void *restrict pIn0,
225  void *restrict pIn1,
226  void *restrict pOut,
227  float alpha)
228 {
229  // typecast handle (void) to struct pointer type associated to kernel
230  VXLIB_addWeight_PrivArgs *pKerPrivArgs = (VXLIB_addWeight_PrivArgs *) handle;
231 
232  // structs to retrieve SE and SA paramters
233  __SE_TEMPLATE_v1 se0Params;
234  __SE_TEMPLATE_v1 se1Params;
235  __SA_TEMPLATE_v1 sa0Params;
236 
237  // create local pointers
238  dTypeIn0 *restrict pInLocal0 = (dTypeIn0 *) pIn0;
239  dTypeIn1 *restrict pInLocal1 = (dTypeIn1 *) pIn1;
240  dTypeOut *restrict pOutLocal = (dTypeOut *) pOut;
241 
242 #if VXLIB_DEBUGPRINT
243  printf("Enter VXLIB_addWeight_exec_ci\n");
244 #endif
245 
246  // address of SE and SA parameters
247  uint8_t *pBlock = pKerPrivArgs->bufPblock;
248 
249  // num blocks for kernel iteration
250  int32_t numBlocks = pKerPrivArgs->numBlocks;
251 
252  // define vector type
253  /* typedef typename c7x::make_full_vector<dTypeOut>::type vec; */
254 
255  // define computation vector type - float
256  typedef typename c7x::make_full_vector<float>::type fvec;
257 
258  // define 32b se promote vector type - uint
259  typedef typename c7x::make_full_vector<uint32_t>::type uvec;
260 
261  // define int vec
262  typedef typename c7x::make_full_vector<int32_t>::type ivec;
263 
264  // retrieve SE and SA parameters
265  se0Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_PARAM_OFFSET);
266  se1Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE1_PARAM_OFFSET);
267  sa0Params = *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_PARAM_OFFSET);
268 
269  // open SEs to fetch Input samples
270  __SE0_OPEN(pInLocal0, se0Params);
271  __SE1_OPEN(pInLocal1, se1Params);
272 
273  // open SA0 to write output samples
274  __SA0_OPEN(sa0Params);
275 
276  // vectorize alpha scalar input
277  fvec vAlpha = (fvec) alpha;
278  fvec vOne = (fvec) 1.0;
279  fvec vOneMinusAlpha = vOne - vAlpha;
280 
281  // define sa vector address type based on architecture
282  typedef typename std::conditional<__C7X_VEC_SIZE_BITS__ == 512, __uchar16, __uchar8>::type __saType;
283 
284  // perform weighted addition of two images
285  for (int32_t counter = 0; counter < numBlocks; counter += UNROLL_FACTOR) {
286 
287  // fetch first 4th of promoted input vector
288  // =================================================
289  uvec a0 = c7x::strm_eng<0, uvec>::get_adv();
290  uvec b0 = c7x::strm_eng<1, uvec>::get_adv();
291 
292  // move to float computation containers
293  fvec fa0 = __int_to_float(a0); // vintspu (.C, .S, .L)
294  fvec fb0 = __int_to_float(b0); // vintspu (.C, .S, .L)
295  fvec fout0 = vAlpha * fa0 + vOneMinusAlpha * fb0; // vmpysp (.M, .N), vaddsp (.C, .S, .L)
296 
297  // move to 32b int container
298  ivec out0 = __float_to_int_rtz(fout0); // vsptrunc (.S, .L)
299 
300  // convert output to unsigned
301  uvec uout0 = c7x::reinterpret<uvec>(out0);
302 
303  // predicated pack store (lowest byte of every word) first 4th of output vector
304  __vpred tmp = c7x::strm_agen<0, uvec>::get_vpred();
305  __saType *addr = c7x::strm_agen<0, __saType>::get_adv(pOutLocal);
306  __vstore_pred_pack_byte(tmp, addr, uout0); // vstpwsvpackb (.D)
307 
308  // fetch second 4th of promoted input vector
309  // =================================================
310  uvec a1 = c7x::strm_eng<0, uvec>::get_adv();
311  uvec b1 = c7x::strm_eng<1, uvec>::get_adv();
312 
313  fvec fa1 = __int_to_float(a1);
314  fvec fb1 = __int_to_float(b1);
315  fvec fout1 = vAlpha * fa1 + vOneMinusAlpha * fb1;
316 
317  ivec out1 = __float_to_int_rtz(fout1);
318  uvec uout1 = c7x::reinterpret<uvec>(out1);
319 
320  // predicated pack store (lowest byte of every word) second 4th of output vector
321  tmp = c7x::strm_agen<0, uvec>::get_vpred();
322  addr = c7x::strm_agen<0, __saType>::get_adv(pOutLocal);
323  __vstore_pred_pack_byte(tmp, addr, uout1);
324 
325  // fetch third 4th of promoted input vector
326  // =================================================
327  uvec a2 = c7x::strm_eng<0, uvec>::get_adv();
328  uvec b2 = c7x::strm_eng<1, uvec>::get_adv();
329 
330  fvec fa2 = __int_to_float(a2);
331  fvec fb2 = __int_to_float(b2);
332  fvec fout2 = vAlpha * fa2 + vOneMinusAlpha * fb2;
333 
334  ivec out2 = __float_to_int_rtz(fout2);
335  uvec uout2 = c7x::reinterpret<uvec>(out2);
336 
337  // predicated pack store (lowest byte of every word) third 4th of output vector
338  tmp = c7x::strm_agen<0, uvec>::get_vpred();
339  addr = c7x::strm_agen<0, __saType>::get_adv(pOutLocal);
340  __vstore_pred_pack_byte(tmp, addr, uout2);
341 
342  // fetch last 4th of promoted input vector
343  // =================================================
344  uvec a3 = c7x::strm_eng<0, uvec>::get_adv();
345  uvec b3 = c7x::strm_eng<1, uvec>::get_adv();
346 
347  fvec fa3 = __int_to_float(a3);
348  fvec fb3 = __int_to_float(b3);
349  fvec fout3 = vAlpha * fa3 + vOneMinusAlpha * fb3;
350 
351  ivec out3 = __float_to_int_rtz(fout3);
352  uvec uout3 = c7x::reinterpret<uvec>(out3);
353 
354  // predicated pack store (lowest byte of every word) last 4th of output vector
355  tmp = c7x::strm_agen<0, uvec>::get_vpred();
356  addr = c7x::strm_agen<0, __saType>::get_adv(pOutLocal);
357  __vstore_pred_pack_byte(tmp, addr, uout3);
358  }
359 
360  // close SE/SA
361  __SE0_CLOSE();
362  __SE1_CLOSE();
363  __SA0_CLOSE();
364 
365  return VXLIB_SUCCESS;
366 }
