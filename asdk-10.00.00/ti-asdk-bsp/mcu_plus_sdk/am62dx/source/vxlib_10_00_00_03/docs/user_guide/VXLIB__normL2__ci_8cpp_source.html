71  __SE_TEMPLATE_v1 * se0Params,
72  __SE_TEMPLATE_v1 * se1Params,
73  __SA_TEMPLATE_v1 * sa0Params,
74  const VXLIB_bufParams2D_t *bufParamsIn0,
75  const VXLIB_bufParams2D_t *bufParamsIn1,
76  const VXLIB_bufParams2D_t *bufParamsOut)
77 {
78  VXLIB_normL2_PrivArgs *pKerPrivArgs = (VXLIB_normL2_PrivArgs *) handle;
79  size_t numBlocks = 0;
80 
81  // obtain image size and compute number of blocks to process
82  size_t width = pKerPrivArgs->width;
83  size_t height = pKerPrivArgs->height;
84 
85  size_t strideIn0Elems = pKerPrivArgs->strideIn0Elements;
86  size_t strideIn1Elems = pKerPrivArgs->strideIn1Elements;
87  size_t strideOutElems = pKerPrivArgs->strideOutElements;
88  size_t elemCount = VXLIB_max_simd<VXLIB_INT8>::value;
89 
90  int32_t nTiles = (width + (elemCount - 1)) / elemCount;
91 
92  numBlocks = VXLIB_SE0SE1SA0_init<VXLIB_I8S_I8S_O8U_DTYPE>(se0Params, se1Params, sa0Params, bufParamsIn0,
93  bufParamsIn1, bufParamsOut);
94 
95  if ((width == strideIn0Elems) && (strideIn0Elems == strideIn1Elems) && (strideIn0Elems == strideOutElems)) {
96  se0Params->VECLEN = c7x::se_veclen<c7x::short_vec>::value;
97  se0Params->PROMOTE = __SE_PROMOTE_2X_SIGNEXT;
98  se1Params->VECLEN = c7x::se_veclen<c7x::short_vec>::value;
99  se1Params->PROMOTE = __SE_PROMOTE_2X_SIGNEXT;
100  }
101  else {
102  se0Params->ICNT0 = elemCount;
103  se0Params->DIM1 = elemCount;
104  se0Params->ICNT1 = nTiles;
105  se0Params->ICNT2 = height;
106  se0Params->DIM2 = strideIn0Elems;
107  se0Params->DIMFMT = __SE_DIMFMT_3D;
108  se0Params->DECDIM1 = __SE_DECDIM_DIM1;
109  se0Params->DECDIM1SD = __SE_DECDIMSD_DIM0;
110  se0Params->VECLEN = c7x::se_veclen<c7x::short_vec>::value;
111  se0Params->DECDIM1_WIDTH = width;
112  se0Params->PROMOTE = __SE_PROMOTE_2X_SIGNEXT;
113 
114  se1Params->ICNT0 = elemCount;
115  se1Params->DIM1 = elemCount;
116  se1Params->ICNT1 = nTiles;
117  se1Params->ICNT2 = height;
118  se1Params->DIM2 = strideIn1Elems;
119  se1Params->DIMFMT = __SE_DIMFMT_3D;
120  se1Params->DECDIM1 = __SE_DECDIM_DIM1;
121  se1Params->DECDIM1SD = __SE_DECDIMSD_DIM0;
122  se1Params->VECLEN = c7x::se_veclen<c7x::short_vec>::value;
123  se1Params->DECDIM1_WIDTH = width;
124  se1Params->PROMOTE = __SE_PROMOTE_2X_SIGNEXT;
125  }
126  return numBlocks;
127 }

195 template <typename vec> static inline vec getSqrt(vec a)
196 {
197  vec zeroVec = (vec) 0;
198  vec Half = (vec) 0.5f;
199  vec OneP5 = (vec) 1.5f;
200  __vpred zeroPred = __cmp_eq_pred(zeroVec, a);
201  vec nonZero = (vec) 1e-12f;
202 
203  a = __select(zeroPred, nonZero, a);
204 
205  vec x = __recip_sqrt(a);
206 
207  vec tmp = x * a;
208  vec tmp1 = OneP5 - tmp * x * Half;
209  vec y = x * tmp1;
210 
211  tmp = y * a;
212  tmp1 = OneP5 - tmp * y * Half;
213  y = a * y * tmp1;
214 
215  return (y);
216 }
217 // this method performs pixel-wise bitwise normL2 of two input images
218 template <typename dTypeIn, typename dTypeOut>
219 VXLIB_STATUS
220 VXLIB_normL2_exec_ci(VXLIB_kernelHandle handle, void *restrict pIn0, void *restrict pIn1, void *restrict pOut)
221 {
222  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
223  // typecast handle (void) to struct pointer type associated to kernel
224  VXLIB_normL2_PrivArgs *pKerPrivArgs = (VXLIB_normL2_PrivArgs *) handle;
225 
226  // structs to retrieve SE and SA paramters
227  __SE_TEMPLATE_v1 se0Params;
228  __SE_TEMPLATE_v1 se1Params;
229  __SA_TEMPLATE_v1 sa0Params;
230 
231  // create local pointers
232  dTypeIn *restrict pInLocal0 = (dTypeIn *) pIn0;
233  dTypeIn *restrict pInLocal1 = (dTypeIn *) pIn1;
234  dTypeOut *restrict pOutLocal = (dTypeOut *) pOut;
235 
236 #if VXLIB_DEBUGPRINT
237  printf("Enter VXLIB_normL2_exec_ci\n");
238 #endif
239 
240  // address of SE and SA parameters
241  uint8_t *pBlock = pKerPrivArgs->bufPblock;
242 
243  // retrieve SE and SA parameters
244  se0Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_PARAM_OFFSET);
245  se1Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE1_PARAM_OFFSET);
246  sa0Params = *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_PARAM_OFFSET);
247 
248  // open SEs to fetch Input samples
249  __SE0_OPEN(pInLocal0, se0Params);
250  __SE1_OPEN(pInLocal1, se1Params);
251 
252  // open SA0 to write output samples
253  __SA0_OPEN(sa0Params);
254 
255  size_t numBlocks = pKerPrivArgs->numBlocks;
256 
257  typedef typename c7x::make_full_vector<dTypeIn>::type inVec;
258  typedef typename c7x::make_full_vector<dTypeOut>::type outVec;
259 
260  c7x::int_vec sqrVec00, sqrVec01, sqrVec10, sqrVec11;
261  c7x::int_vec outVal0, outVal1;
262  c7x::uint_vec sumVec0, sumVec1;
263 
264  c7x::float_vec sumfVec0, sumfVec1, sqrtVec0, sqrtVec1;
265  for (uint32_t counter = 0; counter < numBlocks; counter++) {
266 
267  inVec val0 = c7x::strm_eng<0, inVec>::get_adv(); // fetch input 0 via SE0
268  inVec val1 = c7x::strm_eng<1, inVec>::get_adv(); // fetch input 1 via SE1
269 
270  __vmpyhw_vvw(val0, val0, sqrVec00, sqrVec01);
271  __vmpyhw_vvw(val1, val1, sqrVec10, sqrVec11);
272 
273  sumVec0 = c7x::as_uint_vec(sqrVec00) + c7x::as_uint_vec(sqrVec10);
274  sumVec1 = c7x::as_uint_vec(sqrVec01) + c7x::as_uint_vec(sqrVec11);
275 
276  sumfVec0 = __int_to_float(sumVec0);
277  sumfVec1 = __int_to_float(sumVec1);
278 
279  sqrtVec0 = getSqrt(sumfVec0);
280  sqrtVec1 = getSqrt(sumfVec1);
281 
282  outVal0 = __float_to_int(sqrtVec0);
283  outVal1 = __float_to_int(sqrtVec1);
284 
285  // interleave store.
286  __vpred tmp = c7x::strm_agen<0, outVec>::get_vpred(); // generate predicate
287  outVec *addr = c7x::strm_agen<0, outVec>::get_adv(pOutLocal); // generate address
288  __vstore_pred_interleave(tmp, addr, c7x::as_ushort_vec(outVal0), c7x::as_ushort_vec(outVal1)); // store value
289  }
290 
291  // close SE/SA
292  __SE0_CLOSE();
293  __SE1_CLOSE();
294  __SA0_CLOSE();
295 
296  return status;
297 }

363  sumfVec01 = __high_half_int_to_float(sumVec0);

366  val1 = c7x::strm_eng<1, c7x::short_vec>::get_adv();

374  sumfVec11 = __high_half_int_to_float(sumVec1);
375 
376  sqrtVec00 = getSqrt(sumfVec00);
377  sqrtVec01 = getSqrt(sumfVec01);
378  sqrtVec10 = getSqrt(sumfVec10);
379  sqrtVec11 = getSqrt(sumfVec11);
380 
381  outVal00 = c7x::as_ushort_vec(__vspinthpk_vv(sqrtVec00));
382  outVal01 = c7x::as_ushort_vec(__vspinthpk_vv(sqrtVec01));
383 
384  outVal0 = c7x::as_ushort_vec(__vpackw_vvv(c7x::as_uint_vec(outVal01), c7x::as_uint_vec(outVal00)));
385 
386  outVal10 = c7x::as_ushort_vec(__vspinthpk_vv(sqrtVec10));
387  outVal11 = c7x::as_ushort_vec(__vspinthpk_vv(sqrtVec11));
388 
389  outVal1 = c7x::as_ushort_vec(__vpackw_vvv(c7x::as_uint_vec(outVal11), c7x::as_uint_vec(outVal10)));
390 
391  // pack and store.
392  __vpred tmp = c7x::strm_agen<0, outVec>::get_vpred(); // generate predicate
393  outVec *addr = c7x::strm_agen<0, outVec>::get_adv(pOutLocal); // generate address
394  __vstore_pred_packl_2src(tmp, addr, outVal0, outVal1); // store value
395  }
396 
397  // close SE/SA
398  __SE0_CLOSE();
399  __SE1_CLOSE();
400  __SA0_CLOSE();

402  return status;
