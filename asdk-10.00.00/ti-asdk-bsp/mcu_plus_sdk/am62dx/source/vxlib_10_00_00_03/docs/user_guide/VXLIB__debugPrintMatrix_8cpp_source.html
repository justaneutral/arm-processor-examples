49 template <typename dataType> void VXLIB_debugPrintMatrix_helper(dataType *matrix, const VXLIB_bufParams2D_t *params)
50 {
51  uint32_t x, y;
52  dataType *xPtr;
53  dataType *yPtr = matrix;
54 
55  // stride_y is stored in bytes, but easier to use in elements
56  uint32_t stride_y_elements = params->stride_y / VXLIB_sizeof(params->data_type);
57 
58  for (y = 0; y < params->dim_y; y++) {
59  xPtr = yPtr;
60  VXLIB_PRINTF("%p |", xPtr);
61  for (x = 0; x < params->dim_x; x++) {
62  VXLIB_PRINTF("%3d ", *(xPtr));
63  xPtr++;
64  }
65  VXLIB_PRINTF("%s", "|\n");
66  yPtr += stride_y_elements;
67  }
68 
69  return;
70 }

85 void VXLIB_debugPrintMatrix(void *matrix, const VXLIB_bufParams2D_t *params)
86 {
87  switch (params->data_type) {

90  break;
91  case VXLIB_INT16:

93  break;
94  case VXLIB_UINT16:

96  break;

104  VXLIB_PRINTF("\nERROR: Unrecognized data type in %s.\n", __FUNCTION__);
105  }
