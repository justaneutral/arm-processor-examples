63 // this method initializes the kernel-specific parameters
64 // mainly, the streaming engine and streaming address generators
65 template <uint32_t dTypeIn, uint32_t dTypeOut>
66 VXLIB_STATUS VXLIB_dilate_init_ci(VXLIB_kernelHandle handle,
67  const VXLIB_bufParams2D_t *bufParamsIn,
68  const VXLIB_bufParams2D_t *bufParamsOut,
69  const VXLIB_dilate_InitArgs *pKerInitArgs)
70 {
71  VXLIB_STATUS status = VXLIB_SUCCESS; // assign status to success by default
72 
73  // structs to hold SE and SA parameters
74  __SE_TEMPLATE_v1 se0Params = __gen_SE_TEMPLATE_v1();
75  __SE_TEMPLATE_v1 se1Params = __gen_SE_TEMPLATE_v1();
76  __SA_TEMPLATE_v1 sa0Params = __gen_SA_TEMPLATE_v1();
77  __SA_TEMPLATE_v1 sa1Params = __gen_SA_TEMPLATE_v1();
78 
79  // typecast handle (void) to struct pointer type associated to kernel
80  VXLIB_dilate_PrivArgs *pKerPrivArgs = (VXLIB_dilate_PrivArgs *) handle;
81 
82  typedef typename std::conditional<dTypeIn == VXLIB_UINT8, uint8_t, uint16_t>::type elemType;
83 
84  typedef typename std::conditional<dTypeIn == VXLIB_UINT8, c7x::uchar_vec, c7x::ushort_vec>::type vecType;
85 
86  // number of elements fetched from input each SE get
87  size_t elemCountIn = c7x::element_count_of<vecType>::value;
88 
89  // get init args
90  size_t widthIn = bufParamsIn->dim_x;
91  size_t heightIn = bufParamsIn->dim_y;
92  size_t strideIn = pKerPrivArgs->strideInElements;
93 
94  size_t widthOut = bufParamsOut->dim_x;
95  size_t heightOut = bufParamsOut->dim_y;
96  size_t strideOut = pKerPrivArgs->strideOutElements;
97 
98  size_t filterWidth = pKerInitArgs->filterWidth;
99  size_t filterHeight = pKerInitArgs->filterHeight;
100 
101  uint8_t *pBlock = pKerPrivArgs->bufPblock;
102  pKerPrivArgs->numBlocks = widthIn * VXLIB_ceilingDiv(widthIn, elemCountIn);
103 
104  se0Params.DIMFMT = __SE_DIMFMT_5D;
105  se0Params.ELETYPE = c7x::se_eletype<vecType>::value;
106  se0Params.VECLEN = c7x::se_veclen<vecType>::value;
107 
108  se0Params.ICNT0 = elemCountIn;
109 
110  se0Params.DIM1 = 1;
111  se0Params.ICNT1 = filterWidth;
112 
113  se0Params.DIM2 = strideIn;
114  se0Params.ICNT2 = filterHeight;
115 
116  se0Params.DIM3 = strideIn * 2;
117  se0Params.ICNT3 = VXLIB_ceilingDiv(heightOut, 2);
118 
119  se0Params.DIM4 = elemCountIn;
120  se0Params.ICNT4 = VXLIB_ceilingDiv(widthIn, elemCountIn);
121 
122  se0Params.DECDIM1 = __SE_DECDIM_DIM4;
123  se0Params.DECDIM1_WIDTH = widthIn;
124  se0Params.DECDIM1SD = __SE_DECDIMSD_DIM1;
125 
126  se0Params.DECDIM2 = __SE_DECDIM_DIM3;
127  se0Params.DECDIM2_WIDTH = strideIn * heightIn;
128 
129  se1Params.DIMFMT = __SE_DIMFMT_5D;
130  se1Params.ELETYPE = c7x::se_eletype<vecType>::value;
131  se1Params.VECLEN = c7x::se_veclen<vecType>::value;
132 
133  se1Params.ICNT0 = elemCountIn;
134 
135  se1Params.DIM1 = 1;
136  se1Params.ICNT1 = filterWidth;
137 
138  se1Params.DIM2 = strideIn;
139  se1Params.ICNT2 = filterHeight;
140 
141  se1Params.DIM3 = strideIn * 2;
142  se1Params.ICNT3 = (heightOut % 2 == 0) ? (VXLIB_ceilingDiv(heightOut, 2)) : (VXLIB_ceilingDiv(heightOut, 2) - 1);
143 
144  se1Params.DIM4 = elemCountIn;
145  se1Params.ICNT4 = VXLIB_ceilingDiv(widthIn, elemCountIn);
146 
147  se1Params.DECDIM1 = __SE_DECDIM_DIM4;
148  se1Params.DECDIM1_WIDTH = widthIn;
149  se1Params.DECDIM1SD = __SE_DECDIMSD_DIM1;
150 
151  se1Params.DECDIM2 = __SE_DECDIM_DIM3;
152  se1Params.DECDIM2_WIDTH = heightIn * strideIn;
153 
154  sa0Params.DIMFMT = __SA_DIMFMT_3D;
155  sa0Params.VECLEN = c7x::sa_veclen<vecType>::value;
156 
157  sa0Params.ICNT0 = elemCountIn;
158 
159  sa0Params.DIM1 = strideOut * 2;
160  sa0Params.ICNT1 = VXLIB_ceilingDiv(heightOut, 2);
161 
162  sa0Params.DIM2 = elemCountIn;
163  sa0Params.ICNT2 = VXLIB_ceilingDiv(widthIn, elemCountIn);
164 
165  sa0Params.DECDIM1 = __SA_DECDIM_DIM2;
166  sa0Params.DECDIM1_WIDTH = widthOut;
167  sa0Params.DECDIM1SD = __SA_DECDIMSD_DIM0;
168 
169  sa0Params.DECDIM2 = __SA_DECDIM_DIM1;
170  sa0Params.DECDIM2_WIDTH = strideOut * heightOut;
171 
172  sa1Params.DIMFMT = __SA_DIMFMT_3D;
173  sa1Params.VECLEN = c7x::sa_veclen<vecType>::value;
174 
175  sa1Params.ICNT0 = elemCountIn;
176 
177  sa1Params.DIM1 = strideOut * 2;
178  sa1Params.ICNT1 = (heightOut % 2 == 0) ? (VXLIB_ceilingDiv(heightOut, 2)) : (VXLIB_ceilingDiv(heightOut, 2) - 1);
179 
180  sa1Params.DIM2 = elemCountIn;
181  sa1Params.ICNT2 = VXLIB_ceilingDiv(widthIn, elemCountIn);
182 
183  sa1Params.DECDIM1 = __SA_DECDIM_DIM2;
184  sa1Params.DECDIM1_WIDTH = widthOut;
185  sa1Params.DECDIM1SD = __SA_DECDIMSD_DIM0;
186 
187  sa1Params.DECDIM2 = __SA_DECDIM_DIM1;
188  sa1Params.DECDIM2_WIDTH = strideOut * heightOut;
189 
190  // store SE and SA params
191  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_PARAM_OFFSET) = se0Params;
192  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE1_PARAM_OFFSET) = se1Params;
193  *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_PARAM_OFFSET) = sa0Params;
194  *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA1_PARAM_OFFSET) = sa1Params;
195 
196  return status;
197 }

221 // this method performs the M x N dilation filter on an input image
222 template <typename dTypeIn, typename dTypeOut>
223 VXLIB_STATUS VXLIB_dilate_exec_ci(VXLIB_kernelHandle handle, void *restrict pIn, void *restrict pOut)
224 {
225 
226  VXLIB_STATUS status = VXLIB_SUCCESS;
227 
228  // get kernel handle
229  VXLIB_dilate_PrivArgs *pKerPrivArgs = (VXLIB_dilate_PrivArgs *) handle;
230 
231  // SE/SA params template
232  __SE_TEMPLATE_v1 se0Params;
233  __SE_TEMPLATE_v1 se1Params;
234  __SA_TEMPLATE_v1 sa0Params;
235  __SA_TEMPLATE_v1 sa1Params;
236 
237  // create local pointers
238  dTypeIn *restrict pInLocal = (dTypeIn *) pIn;
239  dTypeOut *restrict pOutLocal = (dTypeOut *) pOut;
240 
241  size_t widthIn = pKerPrivArgs->width;
242  size_t heightIn = pKerPrivArgs->height;
243  size_t strideIn = pKerPrivArgs->strideInElements;
244 
245  size_t filterWidth = pKerPrivArgs->pKerInitArgs.filterWidth;
246  size_t filterHeight = pKerPrivArgs->pKerInitArgs.filterHeight;
247 
248  size_t heightOut = heightIn - filterHeight + 1;
249  size_t strideOut = pKerPrivArgs->strideOutElements;
250 
251  uint8_t *pBlock = pKerPrivArgs->bufPblock;
252 
253  se0Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE0_PARAM_OFFSET);
254  se1Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE1_PARAM_OFFSET);
255  sa0Params = *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA0_PARAM_OFFSET);
256  sa1Params = *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SA1_PARAM_OFFSET);
257 
258  typedef
259  typename std::conditional<std::is_same<dTypeIn, uint8_t>::value, c7x::uchar_vec, c7x::ushort_vec>::type vecType;
260 
261  size_t elemCountIn = c7x::element_count_of<vecType>::value;
262 
263  // open streams
264  __SE0_OPEN(pInLocal, se0Params);
265  __SE1_OPEN(pInLocal + strideIn, se1Params);
266  __SA0_OPEN(sa0Params);
267  __SA1_OPEN(sa1Params);
268 
269  size_t wBlocks = VXLIB_ceilingDiv(widthIn, elemCountIn);
270  size_t hBlocks = VXLIB_ceilingDiv(heightOut, 2);
271 
272  for (int32_t wCounter = 0; wCounter < wBlocks; wCounter++) {
273 
274  for (int32_t hCounter = 0; hCounter < hBlocks; hCounter++) {
275 
276  vecType vMaxA = (vecType) 0;
277  vecType vMaxB = (vecType) 0;
278 
279  for (int32_t fCounter = 0; fCounter < (filterWidth * filterHeight); fCounter++) {
280 
281  vecType a = c7x::strm_eng<0, vecType>::get_adv();
282  vecType b = c7x::strm_eng<1, vecType>::get_adv();
283 
284  vMaxA = __max(vMaxA, a);
285  vMaxB = __max(vMaxB, b);
286  }
287 
288  __vpred tmp0 = c7x::strm_agen<0, vecType>::get_vpred();
289  __vpred tmp1 = c7x::strm_agen<1, vecType>::get_vpred();
290 
291  vecType *addr0 = c7x::strm_agen<0, vecType>::get_adv(pOutLocal);
292  vecType *addr1 = c7x::strm_agen<1, vecType>::get_adv(pOutLocal + strideOut);
293 
294  __vstore_pred(tmp0, addr0, vMaxA);
295  __vstore_pred(tmp1, addr1, vMaxB);
296  }
297  }
298  __SE0_CLOSE();
299  __SE1_CLOSE();
300  __SA0_CLOSE();
301  __SA1_CLOSE();
302 
303  return status;
304 }
