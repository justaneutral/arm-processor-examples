/*
 *  Sends the test parameter structure and number of tests
 */
void VXLIB_convertDepth_getTestParams (
    VXLIB_convertDepth_testParams_t **params,
    int32_t *                                 numTests)
{
   *params = testParams;
   *numTests =
       sizeof (testParams) / sizeof (VXLIB_convertDepth_testParams_t);
}
