/*
 *  Sends the test parameter structure and number of tests
 */
void VXLIB_meanStdDev_getTestParams (
    VXLIB_meanStdDev_testParams_t **params,
    int32_t *                                 numTests)
{
   *params = testParams;
   *numTests =
       sizeof (testParams) / sizeof (VXLIB_meanStdDev_testParams_t);
}
