/*
 *  Sends the test parameter structure and number of tests
 */
void VXLIB_absDiff_getTestParams (
    VXLIB_absDiff_testParams_t **params,
    int32_t *                                 numTests)
{
   *params = testParams;
   *numTests =
       sizeof (testParams) / sizeof (VXLIB_absDiff_testParams_t);
}
