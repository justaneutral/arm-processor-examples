/*
 *  Sends the test parameter structure and number of tests
 */
void VXLIB_addWeight_getTestParams (
    VXLIB_addWeight_testParams_t **params,
    int32_t *                                 numTests)
{
   *params = testParams;
   *numTests =
       sizeof (testParams) / sizeof (VXLIB_addWeight_testParams_t);
}
