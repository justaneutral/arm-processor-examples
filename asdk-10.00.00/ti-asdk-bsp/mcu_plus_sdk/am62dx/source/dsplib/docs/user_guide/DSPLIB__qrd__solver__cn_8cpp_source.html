71  DSPLIB_STATUS status = DSPLIB_SUCCESS;
72 
73  DSPLIB_qrd_solver_PrivArgs *pKerPrivArgs = (DSPLIB_qrd_solver_PrivArgs *) handle;
74  int32_t heightR = pKerPrivArgs->heightR;
75  int32_t widthR = pKerPrivArgs->widthR;
76  int32_t strideQ = pKerPrivArgs->strideQ;
77  int32_t strideR = pKerPrivArgs->strideR;
78  int32_t dataSize = sizeof(dataType);
79 
80  /* Typecast void pointers to respective data type */
81  dataType *pLocalQ = (dataType *) pQ;
82  dataType *pLocalR = (dataType *) pR;
83  dataType *pLocalB = (dataType *) pB;
84  dataType *pLocalY = (dataType *) pY;
85  dataType *pLocalX = (dataType *) pX;
86 
87  int32_t colRstride = strideR / dataSize;
88  int32_t colQstride = strideQ / dataSize;
89  DSPLIB_DEBUGPRINTFN(0, "pLocalQ: %p pLocalR: %p pLocalB: %p pLocalY: %p pLocalX: %p widthR: %d heightR: %d\n",
90  pLocalQ, pLocalR, pLocalB, pLocalY, pLocalX, widthR, heightR);
91 
92  /* ------------------------------------------------------------------- */
93  /* Write each column of 'pLocal' to a row of 'pLocalX'. */
94  /* ------------------------------------------------------------------- */
95 
96  int32_t row;
97  int32_t col;
98  int32_t loopCnt;
99  dataType sum;
100 
101  /* generate y=Q'*b */
102  for (row = 0; row < heightR; row++) {
103  sum = 0.0;
104  for (col = 0; col < heightR; col++) {
105  sum += pLocalQ[row + col * colQstride] * pLocalB[col];
106  }
107  pLocalY[row] = sum;
108  }
109 
110  /* use backward substitution to solve x=inv(R)*y */
111  if (heightR >= widthR){
112  loopCnt = widthR;
113  }
114  else{
115  loopCnt = heightR;
116  }
117  memset(pLocalX, 0, sizeof(dataType) * widthR);
118  for (row = loopCnt - 1; row >= 0; row--) {
119  sum = 0.0;
120  for (col = row + 1; col < loopCnt; col++) {
121  sum += pLocalR[col + row * colRstride] * pLocalX[col];
122  }
123  pLocalX[row] = (pLocalY[row] - sum) / pLocalR[row + row * colRstride];
124  }
125 
126  DSPLIB_DEBUGPRINTFN(0, "Exiting function with return status: %d\n", status);
127 
128  return (status);
129 }
