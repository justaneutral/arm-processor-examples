81  SA_VECLEN = c7x::sa_veclen<vec>::value;
82  SE_ELETYPE = c7x::se_eletype<vec>::value;

87 #endif
88 
89  /**********************************************************************/
90  /* Prepare streaming engine 1 to fetch the input */
91  /**********************************************************************/
92  se0Params = __gen_SE_TEMPLATE_v1();
93 
94  se0Params.ICNT0 = blockSize;
95  se0Params.ELETYPE = SE_ELETYPE;
96  se0Params.VECLEN = SE_VECLEN;
97  se0Params.DIMFMT = __SE_DIMFMT_1D;
98 
99  /**********************************************************************/
100  /* Prepare SA template to store output */
101  /**********************************************************************/
102  sa0Params = __gen_SA_TEMPLATE_v1();

104  sa0Params.ICNT0 = blockSize;
105  sa0Params.DIM1 = blockSize;
106  sa0Params.VECLEN = SA_VECLEN;
107  sa0Params.DIMFMT = __SA_DIMFMT_1D;
108 
109  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE_SE0_PARAM_OFFSET) = se0Params;

155 template <typename dataType>
156 DSPLIB_STATUS
157 DSPLIB_mul_exec_ci(DSPLIB_kernelHandle handle, void *restrict pIn1, void *restrict pIn2, void *restrict pOut)
158 {
159  DSPLIB_mul_PrivArgs *pKerPrivArgs = (DSPLIB_mul_PrivArgs *) handle;
160  uint32_t blockSize = pKerPrivArgs->blockSize;
161 
162  __SE_TEMPLATE_v1 se0Params;
163  __SA_TEMPLATE_v1 sa0Params;
164 
165  dataType *restrict pInLocal1 = (dataType *) pIn1;
166  dataType *restrict pInLocal2 = (dataType *) pIn2;
167  dataType *restrict pOutLocal = (dataType *) pOut;
168 
169 #if DSPLIB_DEBUGPRINT
170  printf("Enter DSPLIB_mul_exec_ci\n");
171 #endif
172 
173  typedef typename c7x::make_full_vector<dataType>::type vec;
174  int32_t eleCount = c7x::element_count_of<vec>::value;
175 
176 #if DSPLIB_DEBUGPRINT
177  printf("Enter eleCount %d\n", eleCount);
178 #endif
179  uint8_t *pBlock = pKerPrivArgs->bufPblock;
180 
181  se0Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE_SE0_PARAM_OFFSET);
182 
183  sa0Params = *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE_SA0_PARAM_OFFSET);
184  __SA0_OPEN(sa0Params);
185 
186  // Input samples
187  __SE0_OPEN(pInLocal1, se0Params);
188  __SE1_OPEN(pInLocal2, se0Params);
189 
190  // Output samples
191  __SA0_OPEN(sa0Params);
192 
193 #if DSPLIB_DEBUGPRINT
194  printf("DSPLIB_DEBUGPRINT blockSize %d\n", blockSize);
195 #endif
196 
197  vec out;
198  for (int32_t counter = 0; counter < blockSize; counter += eleCount) {
199  vec a = c7x::strm_eng<0, vec>::get_adv();
200  vec b = c7x::strm_eng<1, vec>::get_adv();
201 
202  out = a * b;
203 
204  __vpred tmp = c7x::strm_agen<0, vec>::get_vpred();
205  vec *VB1 = c7x::strm_agen<0, vec>::get_adv(pOutLocal);
206 
207  __vstore_pred(tmp, VB1, out);
208  }
209  __SE0_CLOSE();
210  __SE1_CLOSE();
211  __SA0_CLOSE();
212 
213  return DSPLIB_SUCCESS;
214 }
