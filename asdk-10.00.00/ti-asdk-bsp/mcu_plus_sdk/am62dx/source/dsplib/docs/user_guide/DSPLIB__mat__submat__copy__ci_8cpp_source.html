179 template <typename dataType>
180 DSPLIB_STATUS DSPLIB_mat_submat_copy_exec_ci(DSPLIB_kernelHandle handle, void *restrict pX, void *restrict pY)
181 {
182  DSPLIB_DEBUGPRINTFN(0, "%s\n", "Entering function");
183  DSPLIB_STATUS status = DSPLIB_SUCCESS;
184  DSPLIB_mat_submat_copy_PrivArgs *pKerPrivArgs = (DSPLIB_mat_submat_copy_PrivArgs *) handle;
185  int32_t strideX = pKerPrivArgs->strideX;
186  int32_t strideY = pKerPrivArgs->strideY;
187  int32_t dir = pKerPrivArgs->dir;
188  uint8_t *pBlock = pKerPrivArgs->bufPblock;
189  int32_t stRow = pKerPrivArgs->stRow;
190  int32_t stCol = pKerPrivArgs->stCol;
191  int32_t nCols = pKerPrivArgs->nCols;
192  int32_t colStrideX = strideX / sizeof(dataType);
193  int32_t colStrideY = strideY / sizeof(dataType);
194 
195  /* Typecast void pointers to respective data type */
196  dataType *yLocal = (dataType *) pY;
197  dataType *xLocal = (dataType *) pX;
198 
199  DSPLIB_DEBUGPRINTFN(0, "yLocal: %p xLocal: %p nRows : %d nCols: %d stRow: %d stCol:%d dir: %d \n", yLocal, xLocal,pKerPrivArgs->nRows, nCols, stRow, stCol , dir);
200  uint32_t i, j;
201  /* ------------------------------------------------------------------- */
202  /* Copy sub matrix */
203  /* ------------------------------------------------------------------- */
204 
205  typedef typename c7x::make_full_vector<dataType>::type vec;
206 
207  uint32_t eleCount = c7x::element_count_of<vec>::value;
208 
209  dataType *ptrSA0 = NULL;
210  dataType *ptrSA1 = NULL;
211  dataType *ptrSE0 = NULL;
212  dataType *ptrSE1 = NULL;
213 
214  __SE_TEMPLATE_v1 se0MatParams = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock);
215  __SE_TEMPLATE_v1 se1MatParams = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + (1 * SE_PARAM_SIZE));
216  __SA_TEMPLATE_v1 sa0MatParams = *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + (2 * SE_PARAM_SIZE));
217  __SA_TEMPLATE_v1 sa1MatParams = *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + (3 * SE_PARAM_SIZE));
218 
219  if (dir == 0) {
220  ptrSA0 = &yLocal[0];
221  ptrSA1 = &yLocal[colStrideY];
222  ptrSE0 = &xLocal[stRow * colStrideX + stCol];
223  ptrSE1 = &xLocal[(stRow + 1) * colStrideX + stCol];
224  }
225  else {
226  ptrSE0 = &yLocal[0];
227  ptrSE1 = &yLocal[colStrideY];
228  ptrSA0 = &xLocal[stRow * colStrideX + stCol];
229  ptrSA1 = &xLocal[(stRow + 1) * colStrideX + stCol];
230  }
231 
232  int32_t nTile = DSPLIB_ceilingDiv(nCols, eleCount);
233  int32_t loopCnt = se1MatParams.ICNT1 * nTile;
234 
235  __SE0_OPEN(ptrSE0, se0MatParams);
236  __SA0_OPEN(sa0MatParams);
237 
238  vec v1;
239  vec v2;
240 
241  if (se1MatParams.ICNT1) {
242  __SE1_OPEN(ptrSE1, se1MatParams);
243  __SA1_OPEN(sa1MatParams);
244  }
245 
246  for (i = 0; i < (uint32_t)loopCnt; i++) {
247  /* 1 + trip_cnt * 2 */
248  v1 = c7x::strm_eng<0, vec>::get_adv();
249  v2 = c7x::strm_eng<1, vec>::get_adv();
250 
251  __vpred pred = c7x::strm_agen<0, vec>::get_vpred();
252  vec *pStoreVec = c7x::strm_agen<0, vec>::get_adv(ptrSA0);
253  __vstore_pred(pred, pStoreVec, v1);
254 
255  pred = c7x::strm_agen<1, vec>::get_vpred();
256  pStoreVec = c7x::strm_agen<1, vec>::get_adv(ptrSA1);
257  __vstore_pred(pred, pStoreVec, v2);
258  }
259  /* To handle odd number of rows */
260  if (se0MatParams.ICNT1 != se1MatParams.ICNT1) {
261  for (j = 0; j < (uint32_t)nTile; j ++) {
262  v1 = c7x::strm_eng<0, vec>::get_adv();
263 
264  __vpred pred = c7x::strm_agen<0, vec>::get_vpred();
265  vec *pStoreVec = c7x::strm_agen<0, vec>::get_adv(ptrSA0);
266  __vstore_pred(pred, pStoreVec, v1);
267  }
268  }
269 
270 
271  __SE0_CLOSE();
272  __SA0_CLOSE();
273  if (se1MatParams.ICNT1) {
274  __SE1_CLOSE();
275  __SA1_CLOSE();
276  }
277 
278  DSPLIB_DEBUGPRINTFN(0, "Exiting function with return status: %d\n", status);
279 
280  return (status);
281 }
