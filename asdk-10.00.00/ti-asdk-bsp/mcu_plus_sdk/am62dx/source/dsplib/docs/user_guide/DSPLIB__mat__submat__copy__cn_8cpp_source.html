59 {
60  DSPLIB_DEBUGPRINTFN(0, "%s\n", "Entering function");
61  DSPLIB_STATUS status = DSPLIB_SUCCESS;
62  DSPLIB_mat_submat_copy_PrivArgs *pKerPrivArgs = (DSPLIB_mat_submat_copy_PrivArgs *) handle;
63  int32_t strideX = pKerPrivArgs->strideX;
64  int32_t strideY = pKerPrivArgs->strideY;
65  int32_t dir = pKerPrivArgs->dir;
66  int32_t stRow = pKerPrivArgs->stRow;
67  int32_t stCol = pKerPrivArgs->stCol;
68  int32_t nRows = pKerPrivArgs->nRows;
69  int32_t nCols = pKerPrivArgs->nCols;
70  int32_t colStrideX = strideX / sizeof(dataType);
71  int32_t colStrideY = strideY / sizeof(dataType);
72 
73  /* Typecast void pointers to respective data type */
74  dataType *yLocal = (dataType *) pY;
75  dataType *xLocal = (dataType *) pX;
76 
77  DSPLIB_DEBUGPRINTFN(0, "yLocal: %p xLocal: %p nRows : %d nCols: %d stRow: %d stCol:%d dir: %d \n", yLocal, xLocal,nRows, nCols, stRow, stCol , dir);
78  uint32_t i, j;
79  /* ------------------------------------------------------------------- */
80  /* Write each column of 'pXLocal' to a row of 'pYLocal'. */
81  /* ------------------------------------------------------------------- */
82  for (i = stRow; i < (uint32_t)(stRow + nRows); i++) {
83  for (j = stCol; j < (uint32_t)(stCol + nCols); j++) {
84  if (dir == 0){
85  yLocal[(i - stRow) * colStrideY + (j - stCol)] = xLocal[i * colStrideX + j];
86 
87  }
88  else {
89  xLocal[i * colStrideX + j] = yLocal[(i - stRow) * colStrideY + (j - stCol)];
90  }
91  }
92  }
93 
94  DSPLIB_DEBUGPRINTFN(0, "Exiting function with return status: %d\n", status);
95 
96  return (status);
97 }
