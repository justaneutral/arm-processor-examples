127 template <typename vec> static inline vec getSqrt(vec a)
128 {
129  vec zeroVec = (vec) 0;
130  vec Half = (vec) 0.5f;
131  vec OneP5 = (vec) 1.5f;
132  __vpred zeroPred = __cmp_eq_pred(zeroVec, a);
133  vec nonZero = (vec) 1e-12f;
134 
135  a = __select(zeroPred, nonZero, a);
136 
137  vec x = __recip_sqrt(a);
138 
139  vec tmp = x * a;
140  vec tmp1 = OneP5 - tmp * x * Half;
141  vec y = x * tmp1;
142 
143  tmp = y * a;
144  tmp1 = OneP5 - tmp * y * Half;
145  y = a * y * tmp1;
146 
147  return (y);
148 }

457  (nRows - col), pBlock);

459  }
460  } /* if (sum!=0) */
461  } /* for (col=0;col<=loopCount;col++) */
462 
463  DSPLIB_DEBUGPRINTFN(0, "Exiting function with return status: %d\n", status);
464  return (status);
