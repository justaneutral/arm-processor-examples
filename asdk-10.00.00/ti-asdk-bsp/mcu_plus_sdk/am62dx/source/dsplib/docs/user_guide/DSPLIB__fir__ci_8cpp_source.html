125 template <typename dataType>
126 DSPLIB_STATUS DSPLIB_fir_init_ci(DSPLIB_kernelHandle handle,
127  const DSPLIB_bufParams2D_t *bufParamsIn,
128  const DSPLIB_bufParams2D_t *bufParamsFilter,
129  const DSPLIB_bufParams2D_t *bufParamsOut,
130  const DSPLIB_fir_InitArgs *pKerInitArgs)
131 {
132  DSPLIB_STATUS status = DSPLIB_SUCCESS;
133  __SE_TEMPLATE_v1 se0Params =__gen_SE_TEMPLATE_v1();
134  __SE_TEMPLATE_v1 se1Params =__gen_SE_TEMPLATE_v1();
135  __SA_TEMPLATE_v1 sa0Params =__gen_SA_TEMPLATE_v1();
136  __SA_TEMPLATE_v1 sa1Params =__gen_SA_TEMPLATE_v1();
137 
138  DSPLIB_fir_PrivArgs *pKerPrivArgs = (DSPLIB_fir_PrivArgs *) handle;
139  uint8_t *pBlock = pKerPrivArgs->bufPblock;
140  uint32_t dataSize = pKerPrivArgs->initArgs.dataSize;
141  uint32_t batchSize = pKerPrivArgs->initArgs.batchSize;
142  uint32_t filterSize = pKerPrivArgs->initArgs.filterSize;
143  uint32_t enableNchCoefs = pKerPrivArgs->initArgs.enableNchCoefs;
144  uint32_t dataBufferInPitch = pKerPrivArgs->dataBufferInPitch;
145  uint32_t dataBufferOutPitch = pKerPrivArgs->dataBufferOutPitch;
146  uint32_t filterPitch = pKerPrivArgs->filterPitch;
147  int32_t outputSize = pKerPrivArgs->outputSize;
148 
149  typedef typename c7x::make_full_vector<dataType>::type vec;
150  __SE_VECLEN SE_VECLEN = c7x::se_veclen<vec>::value;
151  __SA_VECLEN SA_VECLEN = c7x::sa_veclen<vec>::value;
152  __SE_ELETYPE SE_ELETYPE = c7x::se_eletype<vec>::value;
153  __SE_ELEDUP SE_ELEDUP = c7x::se_eledup<dataType>::value;
154  int32_t eleCount = c7x::element_count_of<vec>::value;
155  uint32_t flag;
156  int32_t blkElements;
157 
158  if (dataSize >= eleCount * 16 && filterSize <= (dataSize - 16 * eleCount + 1)) {
159  flag = 1;
160  }
161  else {
162  flag = 0;
163  }
164  pKerPrivArgs->flag = flag;
165 
166 #if DSPLIB_DEBUGPRINT
167  printf("DSPLIB_DEBUGPRINT FIR size flag: %d\n", flag);
168 #endif
169 
170 #if DSPLIB_DEBUGPRINT
171  printf("DSPLIB_DEBUGPRINT SE_VECLEN: %d, SA_VECLEN: %d, SE_ELETYPE: %d "
172  "batchSize %d outputSize %d\n",
173  SE_VECLEN, SA_VECLEN, SE_ELETYPE, batchSize, outputSize);
174 #endif
175 
176 #if DSPLIB_DEBUGPRINT
177  printf("DSPLIB_DEBUGPRINT SE_VECLEN: %d, SA_VECLEN: %d, SE_ELETYPE: %d "
178  "SE_ELEDUP %d eleCount %d\n",
179  SE_VECLEN, SA_VECLEN, SE_ELETYPE, SE_ELEDUP, eleCount);
180 #endif
181 
182  int32_t dataLoop = 0;
183 
184  if (flag == 1) {
185  blkElements = 16 * eleCount;
186  dataLoop = (outputSize + blkElements - 1 ) / blkElements;
187  dataLoop = (dataLoop == 0) ? 1 : dataLoop;
188 
189 #if DSPLIB_DEBUGPRINT
190  printf("DSPLIB_DEBUGPRINT dataLoop: %d (dataSize - filterSize) %d 16*eleCount "
191  "%d blkElements %d ((dataSize - filterSize) / blkElements) %d\n",
192  dataLoop, (dataSize - filterSize), 16 * eleCount, blkElements, ((dataSize - filterSize) / blkElements));
193 #endif
194  }
195  else {
196  blkElements = (eleCount > filterSize) ? filterSize : eleCount;
197  dataLoop = (outputSize + blkElements - 1 ) / blkElements;
198  dataLoop = (dataLoop == 0) ? 1 : dataLoop;
199  }
200 
201  pKerPrivArgs->dataLoop = dataLoop;
202 
203  if (flag == 0) {
204  /**********************************************************************/
205  /* Prepare streaming engine 1 to fetch the filter values */
206  /**********************************************************************/
207  se1Params = __gen_SE_TEMPLATE_v1();
208  se1Params.ICNT0 = 1;
209  se1Params.ICNT1 = filterSize;
210  se1Params.DIM1 = -(int32_t) 1;
211  se1Params.ICNT2 = dataLoop;
212  se1Params.DIM2 = 0;
213  se1Params.ICNT3 = batchSize;
214  se1Params.DIM3 = (enableNchCoefs == 0) ? 0 : filterPitch;
215  se1Params.ELETYPE = SE_ELETYPE;
216  se1Params.VECLEN = SE_VECLEN;
217  se1Params.ELEDUP = SE_ELEDUP;
218  se1Params.DIMFMT = __SE_DIMFMT_4D;
219  se1Params.DIR = __SE_DIR_DEC;
220  }
221  else {
222  /*********************************************
223  * Prepare SA template to read filter values *
224  *********************************************/
225  sa1Params = __gen_SA_TEMPLATE_v1();
226  sa1Params.ICNT0 = (uint32_t) 1;
227  sa1Params.ICNT1 = (uint32_t) filterSize;
228  sa1Params.DIM1 = -(int32_t) 1;
229  sa1Params.ICNT2 = dataLoop;
230  sa1Params.DIM2 = 0;
231  sa1Params.ICNT3 = batchSize;
232  sa1Params.DIM3 = (enableNchCoefs == 0) ? 0 : filterPitch;
233  sa1Params.DIMFMT = __SA_DIMFMT_4D;
234  sa1Params.VECLEN = __SA_VECLEN_1ELEM;
235  }
236  /**********************************************************************/
237  /* Prepare streaming engine 0 to fetch data */
238  /**********************************************************************/
239  if (flag == 0) {
240  se0Params.ICNT0 = blkElements;
241  se0Params.ICNT1 = filterSize;
242  se0Params.DIM1 = (int32_t) 1;
243  se0Params.ICNT2 = dataLoop;
244  se0Params.DIM2 = blkElements;
245  se0Params.ICNT3 = batchSize;
246  se0Params.DIM3 = dataBufferInPitch;
247  se0Params.DECDIM1 = __SE_DECDIM_DIM2;
248  se0Params.DECDIM1SD = __SE_DECDIMSD_DIM1;
249  se0Params.DECDIM1_WIDTH = dataSize;
250  se0Params.ELETYPE = SE_ELETYPE;
251  se0Params.VECLEN = SE_VECLEN;
252  se0Params.DIMFMT = __SE_DIMFMT_4D;
253  }
254  else {
255  se0Params.ICNT0 = 8*8;
256  se0Params.ICNT1 = filterSize;
257  se0Params.DIM1 = (int32_t) 1;
258  se0Params.ICNT2 = dataLoop;
259  se0Params.DIM2 = (int32_t) 16 * eleCount;
260  se0Params.ICNT3 = batchSize;
261  se0Params.DIM3 = dataBufferInPitch;
262  se0Params.DECDIM1 = __SE_DECDIM_DIM2;
263  se0Params.DECDIM1SD = __SE_DECDIMSD_DIM1;
264  se0Params.DECDIM1_WIDTH = dataSize;
265  se0Params.ELETYPE = SE_ELETYPE;
266  se0Params.VECLEN = SE_VECLEN;
267  se0Params.DIMFMT = __SE_DIMFMT_4D;
268  }
269 
270  /**********************************************************************/
271  /* Prepare SA template to store output */
272  /**********************************************************************/
273  if (flag == 0) {
274  sa0Params.ICNT0 = blkElements;
275  sa0Params.ICNT1 = dataLoop;
276  sa0Params.DIM1 = blkElements;
277  sa0Params.ICNT2 = batchSize;
278  sa0Params.DIM2 = dataBufferOutPitch;
279  sa0Params.DECDIM1_WIDTH = outputSize;
280  sa0Params.DECDIM1 = __SA_DECDIM_DIM1;
281  sa0Params.VECLEN = SA_VECLEN;
282  sa0Params.DIMFMT = __SA_DIMFMT_3D;
283  }
284  else {
285  sa0Params.ICNT0 = eleCount;
286  sa0Params.ICNT1 = 16;
287  sa0Params.DIM1 = (int32_t) eleCount;
288  sa0Params.ICNT2 = dataLoop;
289  sa0Params.DIM2 = (int32_t) 16 * eleCount;
290  sa0Params.ICNT3 = batchSize;
291  sa0Params.DIM3 = dataBufferOutPitch;
292  sa0Params.DECDIM1 = __SA_DECDIM_DIM2;
293  sa0Params.DECDIM1SD = __SA_DECDIMSD_DIM1;
294  sa0Params.DECDIM1_WIDTH = outputSize;
295  sa0Params.VECLEN = SA_VECLEN;
296  sa0Params.DIMFMT = __SA_DIMFMT_4D;
297  }
298 
299  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE_SE0_PARAM_OFFSET) = se0Params;
300  *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE_SE1_PARAM_OFFSET) = se1Params;
301  *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE_SA0_PARAM_OFFSET) = sa0Params;
302  if (flag == 1) {
303  *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE_SA1_PARAM_OFFSET) = sa1Params;
304  }
305  else {
306  }
307 
308  return status;
309 }

340 template <typename dataType>
341 DSPLIB_STATUS
342 DSPLIB_fir_exec_ci(DSPLIB_kernelHandle handle, void *restrict pIn, void *restrict pFilter, void *restrict pOut)
343 {
344  __SE_TEMPLATE_v1 se0Params;
345  __SE_TEMPLATE_v1 se1Params;
346  __SA_TEMPLATE_v1 sa0Params;
347  __SA_TEMPLATE_v1 sa1Params;
348 
349  DSPLIB_fir_PrivArgs *pKerPrivArgs = (DSPLIB_fir_PrivArgs *) handle;
350  uint8_t *pBlock = pKerPrivArgs->bufPblock;
351  uint32_t batchSize = pKerPrivArgs->initArgs.batchSize;
352  uint32_t filterSize = pKerPrivArgs->initArgs.filterSize;
353  uint32_t flag = pKerPrivArgs->flag;
354  int32_t cntData;
355 
356  dataType *pInLocal = (dataType *) pIn;
357  dataType *pFilterLocal = (dataType *) pFilter;
358  dataType *pOutLocal = (dataType *) pOut;
359 
360 #if DSPLIB_DEBUGPRINT
361  printf("Enter DSPLIB_fir_exec_ci\n");
362 #endif
363 
364  typedef typename c7x::make_full_vector<dataType>::type vec;
365  int eleCount = c7x::element_count_of<vec>::value;
366 
367  vec accumulator_0, accumulator_1, accumulator_2, accumulator_3;
368  vec accumulator_x, accumulator_y;
369 
370  vec accumulator_0_0;
371  vec accumulator_0_1;
372  vec accumulator_0_2;
373  vec accumulator_0_3;
374  vec accumulator_0_4;
375  vec accumulator_0_5;
376  vec accumulator_0_6;
377  vec accumulator_0_7;
378  vec accumulator_1_0;
379  vec accumulator_1_1;
380  vec accumulator_1_2;
381  vec accumulator_1_3;
382  vec accumulator_1_4;
383  vec accumulator_1_5;
384  vec accumulator_1_6;
385  vec accumulator_1_7;
386 
387 #if DSPLIB_DEBUGPRINT
388  printf("Enter pFilter %p\n", pFilter);
389 #endif
390 
391  if (flag == 1) {
392  pFilterLocal = pFilterLocal + filterSize - 1;
393  }
394  else {
395  pFilterLocal = pFilterLocal + filterSize;
396  }
397 
398 #if DSPLIB_DEBUGPRINT
399  printf("Enter pFilter %p\n", pFilterLocal);
400 #endif
401 
402  se0Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE_SE0_PARAM_OFFSET);
403  __SE0_OPEN(pInLocal, se0Params);
404 
405  if (flag == 0) {
406  se1Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE_SE1_PARAM_OFFSET);
407  __SE1_OPEN(pFilterLocal, se1Params);
408  }
409  else {
410 
411  pInLocal = pInLocal + 8 * eleCount;
412  se1Params = *(__SE_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE_SE0_PARAM_OFFSET);
413  se1Params.DECDIM1_WIDTH -= 8*8; // SE1 is advanced w.r.t. SE0 by 64 samples, adjust decdim width accordingly
414  __SE1_OPEN(pInLocal, se1Params);
415 
416  // Filter coefficient
417  sa1Params = *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE_SA1_PARAM_OFFSET);
418  __SA1_OPEN(sa1Params);
419  }
420 
421  sa0Params = *(__SA_TEMPLATE_v1 *) ((uint8_t *) pBlock + SE_SA0_PARAM_OFFSET);
422  __SA0_OPEN(sa0Params);
423 
424  cntData = pKerPrivArgs->dataLoop;
425 
426 #if DSPLIB_DEBUGPRINT
427  printf("DSPLIB_DEBUGPRINT dataSize %d filterSize %d cntData %d\n", pKerPrivArgs->initArgs.dataSize, filterSize, cntData);
428 #endif
429 
430 #if DSPLIB_DEBUGPRINT
431  printf("DSPLIB_DEBUGPRINT filterSize %d\n", filterSize);
432 #endif
433 
434  if (flag == 0) {
435  uint32_t filterSizeDiv4 = filterSize/4;
436  uint32_t filterSizeMod4 = filterSize%4;
437 
438  for (int32_t batchSizeCount = 0; batchSizeCount < batchSize; batchSizeCount++) {
439  for (int32_t dataLoopCount = 0; dataLoopCount < cntData; dataLoopCount++) {
440  accumulator_0 = (vec) 0;
441  accumulator_1 = (vec) 0;
442  accumulator_2 = (vec) 0;
443  accumulator_3 = (vec) 0;
444 
445  for (int32_t filterLoopCount = 0; filterLoopCount < filterSizeDiv4; filterLoopCount++) {
446  vec tmp0 = c7x::strm_eng<0, vec>::get_adv();
447  vec tmp1 = c7x::strm_eng<1, vec>::get_adv();
448 #if DSPLIB_DEBUGPRINT
449 // DSPLIB_debugPrintVector(tmp0);
450 // DSPLIB_debugPrintVector(tmp1);
451 #endif
452  accumulator_0 += tmp0 * tmp1;
453 
454  vec tmp2 = c7x::strm_eng<0, vec>::get_adv();
455  vec tmp3 = c7x::strm_eng<1, vec>::get_adv();
456  accumulator_1 += tmp2 * tmp3;
457 
458  vec tmp4 = c7x::strm_eng<0, vec>::get_adv();
459  vec tmp5 = c7x::strm_eng<1, vec>::get_adv();
460  accumulator_2 += tmp4 * tmp5;
461 
462  vec tmp6 = c7x::strm_eng<0, vec>::get_adv();
463  vec tmp7 = c7x::strm_eng<1, vec>::get_adv();
464  accumulator_3 += tmp6 * tmp7;
465  }
466 
467  for (int32_t filterLoopCount = 0; filterLoopCount < filterSizeMod4; filterLoopCount++)
468  {
469  vec tmp0 = c7x::strm_eng<0, vec>::get_adv();
470  vec tmp1 = c7x::strm_eng<1, vec>::get_adv();
471  accumulator_0 += tmp0 * tmp1;
472  }
473 
474  accumulator_x = accumulator_0 + accumulator_1;
475  accumulator_y = accumulator_2 + accumulator_3;
476  accumulator_x = accumulator_x + accumulator_y;
477  __vpred tmp = c7x::strm_agen<0, vec>::get_vpred();
478  vec *VB1 = c7x::strm_agen<0, vec>::get_adv(pOutLocal);
479  __vstore_pred(tmp, VB1, accumulator_x);
480 
481 #if DSPLIB_DEBUGPRINT
482  DSPLIB_debugPrintVector(accumulator_x);
483 #endif
484  }
485  }
486  }
487  else {
488  for (int32_t batchSizeCount = 0; batchSizeCount < batchSize; batchSizeCount++) {
489  for (int32_t dataLoopCount = 0; dataLoopCount < cntData; dataLoopCount++) {
490  accumulator_0_0 = (vec) 0;
491  accumulator_0_1 = (vec) 0;
492  accumulator_0_2 = (vec) 0;
493  accumulator_0_3 = (vec) 0;
494  accumulator_0_4 = (vec) 0;
495  accumulator_0_5 = (vec) 0;
496  accumulator_0_6 = (vec) 0;
497  accumulator_0_7 = (vec) 0;
498  accumulator_1_0 = (vec) 0;
499  accumulator_1_1 = (vec) 0;
500  accumulator_1_2 = (vec) 0;
501  accumulator_1_3 = (vec) 0;
502  accumulator_1_4 = (vec) 0;
503  accumulator_1_5 = (vec) 0;
504  accumulator_1_6 = (vec) 0;
505  accumulator_1_7 = (vec) 0;
506 
507  for (int32_t filterLoopCount = 0; filterLoopCount < (filterSize); filterLoopCount++) {
508  dataType *VB1Dup = c7x::strm_agen<1, dataType>::get_adv(pFilterLocal);
509  vec flt = __vload_dup(VB1Dup);
510 
511  vec tmp0_0 = c7x::strm_eng<0, vec>::get_adv();
512  accumulator_0_0 += tmp0_0 * flt;
513 
514  vec tmp0_1 = c7x::strm_eng<0, vec>::get_adv();
515  accumulator_0_1 += tmp0_1 * flt;
516 
517  vec tmp0_2 = c7x::strm_eng<0, vec>::get_adv();
518  accumulator_0_2 += tmp0_2 * flt;
519 
520  vec tmp0_3 = c7x::strm_eng<0, vec>::get_adv();
521  accumulator_0_3 += tmp0_3 * flt;
522 
523  vec tmp0_4 = c7x::strm_eng<0, vec>::get_adv();
524  accumulator_0_4 += tmp0_4 * flt;
525 
526  vec tmp0_5 = c7x::strm_eng<0, vec>::get_adv();
527  accumulator_0_5 += tmp0_5 * flt;
