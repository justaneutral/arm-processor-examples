/*
 *  Sends the test parameter structure and number of tests
 */
void DSPLIB_sqrAdd_getTestParams (
    DSPLIB_sqrAdd_testParams_t **params,
    int32_t *                                 numTests)
{
   *params = testParams;
   *numTests =
       sizeof (testParams) / sizeof (DSPLIB_sqrAdd_testParams_t);
}
