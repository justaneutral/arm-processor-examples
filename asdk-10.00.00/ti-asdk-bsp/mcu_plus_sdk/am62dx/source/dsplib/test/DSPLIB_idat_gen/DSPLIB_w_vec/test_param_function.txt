/*
 *  Sends the test parameter structure and number of tests
 */
void DSPLIB_w_vec_getTestParams (
    DSPLIB_w_vec_testParams_t **params,
    int32_t *                                 numTests)
{
   *params = testParams;
   *numTests =
       sizeof (testParams) / sizeof (DSPLIB_w_vec_testParams_t);
}
