/*
 *  Sends the test parameter structure and number of tests
 */
void DSPLIB_dotp_sqr_getTestParams (
    DSPLIB_dotp_sqr_testParams_t **params,
    int32_t *                                 numTests)
{
   *params = testParams;
   *numTests =
       sizeof (testParams) / sizeof (DSPLIB_dotp_sqr_testParams_t);
}
